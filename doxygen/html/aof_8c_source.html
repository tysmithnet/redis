46 /* ----------------------------------------------------------------------------
47  * AOF rewrite buffer implementation.
48  *
49  * The following code implement a simple buffer used in order to accumulate
50  * changes while the background process is rewriting the AOF file.
51  *
52  * We only need to append, but can't just use realloc with a large block
53  * because 'huge' reallocs are not always handled as one could expect
54  * (via remapping of pages at OS level) but may involve copying data.
55  *
56  * For this reason we use a list of blocks, every block is
57  * AOF_RW_BUF_BLOCK_SIZE bytes.
58  * ------------------------------------------------------------------------- */
59 
60 #define AOF_RW_BUF_BLOCK_SIZE (1024*1024*10) /* 10 MB per block */
61 
62 typedef struct aofrwblock {
63  unsigned long used, free;
64  char buf[AOF_RW_BUF_BLOCK_SIZE];
65 } aofrwblock;

92 /* Event handler used to send data to the child process doing the AOF
93  * rewrite. We send pieces of our AOF differences buffer so that the final
94  * write when the child finishes the rewrite will be small. */
95 void aofChildWriteDiffData(aeEventLoop *el, int fd, void *privdata, int mask) {
96  listNode *ln;
97  aofrwblock *block;
98  ssize_t nwritten;
99  UNUSED(el);
100  UNUSED(fd);
101  UNUSED(privdata);
102  UNUSED(mask);
103 
104  while(1) {
105  ln = listFirst(server.aof_rewrite_buf_blocks);
106  block = ln ? ln->value : NULL;
107  if (server.aof_stop_sending_diff || !block) {
108  aeDeleteFileEvent(server.el,server.aof_pipe_write_data_to_child,
109  AE_WRITABLE);
110  return;
111  }
112  if (block->used > 0) {
113  nwritten = write(server.aof_pipe_write_data_to_child,
114  block->buf,block->used);
115  if (nwritten <= 0) return;
116  memmove(block->buf,block->buf+nwritten,block->used-nwritten);
117  block->used -= nwritten;
118  block->free += nwritten;
119  }
120  if (block->used == 0) listDelNode(server.aof_rewrite_buf_blocks,ln);
121  }
122 }

124 /* Append data to the AOF rewrite buffer, allocating new blocks if needed. */
125 void aofRewriteBufferAppend(unsigned char *s, unsigned long len) {
126  listNode *ln = listLast(server.aof_rewrite_buf_blocks);
127  aofrwblock *block = ln ? ln->value : NULL;
128 
129  while(len) {
130  /* If we already got at least an allocated block, try appending
131  * at least some piece into it. */
132  if (block) {
133  unsigned long thislen = (block->free < len) ? block->free : len;
134  if (thislen) { /* The current block is not already full. */
135  memcpy(block->buf+block->used, s, thislen);
136  block->used += thislen;
137  block->free -= thislen;
138  s += thislen;
139  len -= thislen;
140  }
141  }
142 
143  if (len) { /* First block to allocate, or need another block. */
144  int numblocks;
145 
146  block = zmalloc(sizeof(*block));
147  block->free = AOF_RW_BUF_BLOCK_SIZE;
148  block->used = 0;
149  listAddNodeTail(server.aof_rewrite_buf_blocks,block);
150 
151  /* Log every time we cross more 10 or 100 blocks, respectively
152  * as a notice or warning. */
153  numblocks = listLength(server.aof_rewrite_buf_blocks);
154  if (((numblocks+1) % 10) == 0) {
155  int level = ((numblocks+1) % 100) == 0 ? LL_WARNING :
156  LL_NOTICE;
157  serverLog(level,"Background AOF buffer size: %lu MB",
158  aofRewriteBufferSize()/(1024*1024));
159  }
160  }
161  }
162 
163  /* Install a file event to send data to the rewrite child if there is
164  * not one already. */
165  if (aeGetFileEvents(server.el,server.aof_pipe_write_data_to_child) == 0) {
166  aeCreateFileEvent(server.el, server.aof_pipe_write_data_to_child,
167  AE_WRITABLE, aofChildWriteDiffData, NULL);
168  }
169 }

171 /* Write the buffer (possibly composed of multiple blocks) into the specified
172  * fd. If a short write or any other error happens -1 is returned,
173  * otherwise the number of bytes written is returned. */
174 ssize_t aofRewriteBufferWrite(int fd) {
175  listNode *ln;
176  listIter li;
177  ssize_t count = 0;
178 
179  listRewind(server.aof_rewrite_buf_blocks,&li);
180  while((ln = listNext(&li))) {
181  aofrwblock *block = listNodeValue(ln);
182  ssize_t nwritten;
183 
184  if (block->used) {
185  nwritten = write(fd,block->buf,block->used);
186  if (nwritten != (ssize_t)block->used) {
187  if (nwritten == 0) errno = EIO;
188  return -1;
189  }
190  count += nwritten;
191  }
192  }
193  return count;
194 }

284 /* This is a wrapper to the write syscall in order to retry on short writes
285  * or if the syscall gets interrupted. It could look strange that we retry
286  * on short writes given that we are writing to a block device: normally if
287  * the first call is short, there is a end-of-space condition, so the next
288  * is likely to fail. However apparently in modern systems this is no longer
289  * true, and in general it looks just more resilient to retry the write. If
290  * there is an actual error condition we'll get it at the next try. */
291 ssize_t aofWrite(int fd, const char *buf, size_t len) {
292  ssize_t nwritten = 0, totwritten = 0;
293 
294  while(len) {
295  nwritten = write(fd, buf, len);
296 
297  if (nwritten < 0) {
298  if (errno == EINTR) {
299  continue;
300  }
301  return totwritten ? totwritten : -1;
302  }
303 
304  len -= nwritten;
305  buf += nwritten;
306  totwritten += nwritten;
307  }
308 
309  return totwritten;
310 }

312 /* Write the append only file buffer on disk.
313  *
314  * Since we are required to write the AOF before replying to the client,
315  * and the only way the client socket can get a write is entering when the
316  * the event loop, we accumulate all the AOF writes in a memory
317  * buffer and write it on disk using this function just before entering
318  * the event loop again.
319  *
320  * About the 'force' argument:
321  *
322  * When the fsync policy is set to 'everysec' we may delay the flush if there
323  * is still an fsync() going on in the background thread, since for instance
324  * on Linux write(2) will be blocked by the background fsync anyway.
325  * When this happens we remember that there is some aof buffer to be
326  * flushed ASAP, and will try to do that in the serverCron() function.
327  *
328  * However if force is set to 1 we'll write regardless of the background
329  * fsync. */
330 #define AOF_WRITE_LOG_ERROR_RATE 30 /* Seconds between errors logging. */
331 void flushAppendOnlyFile(int force) {
332  ssize_t nwritten;
333  int sync_in_progress = 0;
334  mstime_t latency;
335 
336  if (sdslen(server.aof_buf) == 0) return;
337 
338  if (server.aof_fsync == AOF_FSYNC_EVERYSEC)
339  sync_in_progress = bioPendingJobsOfType(BIO_AOF_FSYNC) != 0;
340 
341  if (server.aof_fsync == AOF_FSYNC_EVERYSEC && !force) {
342  /* With this append fsync policy we do background fsyncing.
343  * If the fsync is still in progress we can try to delay
344  * the write for a couple of seconds. */
345  if (sync_in_progress) {
346  if (server.aof_flush_postponed_start == 0) {
347  /* No previous write postponing, remember that we are
348  * postponing the flush and return. */
349  server.aof_flush_postponed_start = server.unixtime;
350  return;
351  } else if (server.unixtime - server.aof_flush_postponed_start < 2) {
352  /* We were already waiting for fsync to finish, but for less
353  * than two seconds this is still ok. Postpone again. */
354  return;
355  }
356  /* Otherwise fall trough, and go write since we can't wait
357  * over two seconds. */
358  server.aof_delayed_fsync++;
359  serverLog(LL_NOTICE,"Asynchronous AOF fsync is taking too long (disk is busy?). Writing the AOF buffer without waiting for fsync to complete, this may slow down Redis.");
360  }
361  }
362  /* We want to perform a single write. This should be guaranteed atomic
363  * at least if the filesystem we are writing is a real physical one.
364  * While this will save us against the server being killed I don't think
365  * there is much to do about the whole server stopping for power problems
366  * or alike */
367 
368  latencyStartMonitor(latency);
369  nwritten = aofWrite(server.aof_fd,server.aof_buf,sdslen(server.aof_buf));
370  latencyEndMonitor(latency);
371  /* We want to capture different events for delayed writes:
372  * when the delay happens with a pending fsync, or with a saving child
373  * active, and when the above two conditions are missing.
374  * We also use an additional event name to save all samples which is
375  * useful for graphing / monitoring purposes. */
376  if (sync_in_progress) {
377  latencyAddSampleIfNeeded("aof-write-pending-fsync",latency);
378  } else if (server.aof_child_pid != -1 || server.rdb_child_pid != -1) {
379  latencyAddSampleIfNeeded("aof-write-active-child",latency);
380  } else {
381  latencyAddSampleIfNeeded("aof-write-alone",latency);
382  }
383  latencyAddSampleIfNeeded("aof-write",latency);
384 
385  /* We performed the write so reset the postponed flush sentinel to zero. */
386  server.aof_flush_postponed_start = 0;
387 
388  if (nwritten != (ssize_t)sdslen(server.aof_buf)) {
389  static time_t last_write_error_log = 0;
390  int can_log = 0;
391 
392  /* Limit logging rate to 1 line per AOF_WRITE_LOG_ERROR_RATE seconds. */
393  if ((server.unixtime - last_write_error_log) > AOF_WRITE_LOG_ERROR_RATE) {
394  can_log = 1;
395  last_write_error_log = server.unixtime;
396  }
397 
398  /* Log the AOF write error and record the error code. */
399  if (nwritten == -1) {
400  if (can_log) {
401  serverLog(LL_WARNING,"Error writing to the AOF file: %s",
402  strerror(errno));
403  server.aof_last_write_errno = errno;
404  }
405  } else {
406  if (can_log) {
407  serverLog(LL_WARNING,"Short write while writing to "
408  "the AOF file: (nwritten=%lld, "
409  "expected=%lld)",
410  (long long)nwritten,
411  (long long)sdslen(server.aof_buf));
412  }
413 
414  if (ftruncate(server.aof_fd, server.aof_current_size) == -1) {
415  if (can_log) {
416  serverLog(LL_WARNING, "Could not remove short write "
417  "from the append-only file. Redis may refuse "
418  "to load the AOF the next time it starts. "
419  "ftruncate: %s", strerror(errno));
420  }
421  } else {
422  /* If the ftruncate() succeeded we can set nwritten to
423  * -1 since there is no longer partial data into the AOF. */
424  nwritten = -1;
425  }
426  server.aof_last_write_errno = ENOSPC;
427  }
428 
429  /* Handle the AOF write error. */
430  if (server.aof_fsync == AOF_FSYNC_ALWAYS) {
431  /* We can't recover when the fsync policy is ALWAYS since the
432  * reply for the client is already in the output buffers, and we
433  * have the contract with the user that on acknowledged write data
434  * is synced on disk. */
435  serverLog(LL_WARNING,"Can't recover from AOF write error when the AOF fsync policy is 'always'. Exiting...");
436  exit(1);
437  } else {
438  /* Recover from failed write leaving data into the buffer. However
439  * set an error to stop accepting writes as long as the error
440  * condition is not cleared. */
441  server.aof_last_write_status = C_ERR;
442 
443  /* Trim the sds buffer if there was a partial write, and there
444  * was no way to undo it with ftruncate(2). */
445  if (nwritten > 0) {
446  server.aof_current_size += nwritten;
447  sdsrange(server.aof_buf,nwritten,-1);
448  }
449  return; /* We'll try again on the next call... */
450  }
451  } else {
452  /* Successful write(2). If AOF was in error state, restore the
453  * OK state and log the event. */
454  if (server.aof_last_write_status == C_ERR) {
455  serverLog(LL_WARNING,
456  "AOF write error looks solved, Redis can write again.");
457  server.aof_last_write_status = C_OK;
458  }
459  }
460  server.aof_current_size += nwritten;
461 
462  /* Re-use AOF buffer when it is small enough. The maximum comes from the
463  * arena size of 4k minus some overhead (but is otherwise arbitrary). */
464  if ((sdslen(server.aof_buf)+sdsavail(server.aof_buf)) < 4000) {
465  sdsclear(server.aof_buf);
466  } else {
467  sdsfree(server.aof_buf);
468  server.aof_buf = sdsempty();
469  }
470 
471  /* Don't fsync if no-appendfsync-on-rewrite is set to yes and there are
472  * children doing I/O in the background. */
473  if (server.aof_no_fsync_on_rewrite &&
474  (server.aof_child_pid != -1 || server.rdb_child_pid != -1))
475  return;
476 
477  /* Perform the fsync if needed. */
478  if (server.aof_fsync == AOF_FSYNC_ALWAYS) {
479  /* redis_fsync is defined as fdatasync() for Linux in order to avoid
480  * flushing metadata. */
481  latencyStartMonitor(latency);
482  redis_fsync(server.aof_fd); /* Let's try to get this data on the disk */
483  latencyEndMonitor(latency);
484  latencyAddSampleIfNeeded("aof-fsync-always",latency);
485  server.aof_last_fsync = server.unixtime;
486  } else if ((server.aof_fsync == AOF_FSYNC_EVERYSEC &&
487  server.unixtime > server.aof_last_fsync)) {
488  if (!sync_in_progress) aof_background_fsync(server.aof_fd);
489  server.aof_last_fsync = server.unixtime;
490  }
491 }

555 void feedAppendOnlyFile(struct redisCommand *cmd, int dictid, robj **argv, int argc) {
556  sds buf = sdsempty();
557  robj *tmpargv[3];
558 
559  /* The DB this command was targeting is not the same as the last command
560  * we appended. To issue a SELECT command is needed. */
561  if (dictid != server.aof_selected_db) {
562  char seldb[64];
563 
564  snprintf(seldb,sizeof(seldb),"%d",dictid);
565  buf = sdscatprintf(buf,"*2\r\n$6\r\nSELECT\r\n$%lu\r\n%s\r\n",
566  (unsigned long)strlen(seldb),seldb);
567  server.aof_selected_db = dictid;
568  }
569 
570  if (cmd->proc == expireCommand || cmd->proc == pexpireCommand ||
571  cmd->proc == expireatCommand) {
572  /* Translate EXPIRE/PEXPIRE/EXPIREAT into PEXPIREAT */
573  buf = catAppendOnlyExpireAtCommand(buf,cmd,argv[1],argv[2]);
574  } else if (cmd->proc == setexCommand || cmd->proc == psetexCommand) {
575  /* Translate SETEX/PSETEX to SET and PEXPIREAT */
576  tmpargv[0] = createStringObject("SET",3);
577  tmpargv[1] = argv[1];
578  tmpargv[2] = argv[3];
579  buf = catAppendOnlyGenericCommand(buf,3,tmpargv);

670 /* Replay the append log file. On success C_OK is returned. On non fatal
671  * error (the append only file is zero-length) C_ERR is returned. On
672  * fatal error an error message is logged and the program exists. */
673 int loadAppendOnlyFile(char *filename) {
674  struct client *fakeClient;
675  FILE *fp = fopen(filename,"r");
676  struct redis_stat sb;
677  int old_aof_state = server.aof_state;
678  long loops = 0;
679  off_t valid_up_to = 0; /* Offset of latest well-formed command loaded. */
680  off_t valid_before_multi = 0; /* Offset before MULTI command loaded. */
681 
682  if (fp == NULL) {
683  serverLog(LL_WARNING,"Fatal error: can't open the append log file for reading: %s",strerror(errno));
684  exit(1);
685  }
686 
687  /* Handle a zero-length AOF file as a special case. An empty AOF file
688  * is a valid AOF because an empty server with AOF enabled will create
689  * a zero length file at startup, that will remain like that if no write
690  * operation is received. */
691  if (fp && redis_fstat(fileno(fp),&sb) != -1 && sb.st_size == 0) {
692  server.aof_current_size = 0;
693  fclose(fp);
694  return C_ERR;
695  }
696 
697  /* Temporarily disable AOF, to prevent EXEC from feeding a MULTI
698  * to the same file we're about to read. */
699  server.aof_state = AOF_OFF;
700 
701  fakeClient = createFakeClient();
702  startLoading(fp);
703 
704  /* Check if this AOF file has an RDB preamble. In that case we need to
705  * load the RDB file and later continue loading the AOF tail. */
706  char sig[5]; /* "REDIS" */
707  if (fread(sig,1,5,fp) != 5 || memcmp(sig,"REDIS",5) != 0) {
708  /* No RDB preamble, seek back at 0 offset. */
709  if (fseek(fp,0,SEEK_SET) == -1) goto readerr;
710  } else {
711  /* RDB preamble. Pass loading the RDB functions. */
712  rio rdb;
713 
714  serverLog(LL_NOTICE,"Reading RDB preamble from AOF file...");
715  if (fseek(fp,0,SEEK_SET) == -1) goto readerr;
716  rioInitWithFile(&rdb,fp);
717  if (rdbLoadRio(&rdb,NULL,1) != C_OK) {
718  serverLog(LL_WARNING,"Error reading the RDB preamble of the AOF file, AOF loading aborted");
719  goto readerr;
720  } else {
721  serverLog(LL_NOTICE,"Reading the remaining AOF tail...");
722  }
723  }
724 
725  /* Read the actual AOF file, in REPL format, command by command. */
726  while(1) {
727  int argc, j;
728  unsigned long len;
729  robj **argv;
730  char buf[128];
731  sds argsds;
732  struct redisCommand *cmd;
733 
734  /* Serve the clients from time to time */
735  if (!(loops++ % 1000)) {
736  loadingProgress(ftello(fp));
737  processEventsWhileBlocked();
738  }
739 
740  if (fgets(buf,sizeof(buf),fp) == NULL) {
741  if (feof(fp))
742  break;
743  else
744  goto readerr;
745  }
746  if (buf[0] != '*') goto fmterr;
747  if (buf[1] == '\0') goto readerr;
748  argc = atoi(buf+1);
749  if (argc < 1) goto fmterr;
750 
751  argv = zmalloc(sizeof(robj*)*argc);
752  fakeClient->argc = argc;
753  fakeClient->argv = argv;
754 
755  for (j = 0; j < argc; j++) {
756  if (fgets(buf,sizeof(buf),fp) == NULL) {
757  fakeClient->argc = j; /* Free up to j-1. */
758  freeFakeClientArgv(fakeClient);
759  goto readerr;
760  }
761  if (buf[0] != '$') goto fmterr;
762  len = strtol(buf+1,NULL,10);
763  argsds = sdsnewlen(SDS_NOINIT,len);
764  if (len && fread(argsds,len,1,fp) == 0) {
765  sdsfree(argsds);
766  fakeClient->argc = j; /* Free up to j-1. */
767  freeFakeClientArgv(fakeClient);
768  goto readerr;
769  }
770  argv[j] = createObject(OBJ_STRING,argsds);
771  if (fread(buf,2,1,fp) == 0) {
772  fakeClient->argc = j+1; /* Free up to j. */
773  freeFakeClientArgv(fakeClient);
774  goto readerr; /* discard CRLF */
775  }
776  }
777 
778  /* Command lookup */
779  cmd = lookupCommand(argv[0]->ptr);
780  if (!cmd) {
781  serverLog(LL_WARNING,
782  "Unknown command '%s' reading the append only file",
783  (char*)argv[0]->ptr);
784  exit(1);
785  }
786 
787  if (cmd == server.multiCommand) valid_before_multi = valid_up_to;
788 
789  /* Run the command in the context of a fake client */
790  fakeClient->cmd = cmd;
791  if (fakeClient->flags & CLIENT_MULTI &&
792  fakeClient->cmd->proc != execCommand)
793  {
794  queueMultiCommand(fakeClient);
795  } else {
796  cmd->proc(fakeClient);
797  }
798 
799  /* The fake client should not have a reply */
800  serverAssert(fakeClient->bufpos == 0 &&
801  listLength(fakeClient->reply) == 0);
802 
803  /* The fake client should never get blocked */
804  serverAssert((fakeClient->flags & CLIENT_BLOCKED) == 0);
805 
806  /* Clean up. Command code may have changed argv/argc so we use the
807  * argv/argc of the client instead of the local variables. */
808  freeFakeClientArgv(fakeClient);
809  fakeClient->cmd = NULL;
810  if (server.aof_load_truncated) valid_up_to = ftello(fp);
811  }
812 
813  /* This point can only be reached when EOF is reached without errors.
814  * If the client is in the middle of a MULTI/EXEC, handle it as it was
815  * a short read, even if technically the protocol is correct: we want
816  * to remove the unprocessed tail and continue. */
817  if (fakeClient->flags & CLIENT_MULTI) {
818  serverLog(LL_WARNING,
819  "Revert incomplete MULTI/EXEC transaction in AOF file");
820  valid_up_to = valid_before_multi;
821  goto uxeof;
822  }
823 
824 loaded_ok: /* DB loaded, cleanup and return C_OK to the caller. */
825  fclose(fp);
826  freeFakeClient(fakeClient);
827  server.aof_state = old_aof_state;
828  stopLoading();
829  aofUpdateCurrentSize();
830  server.aof_rewrite_base_size = server.aof_current_size;
831  return C_OK;
832 
833 readerr: /* Read error. If feof(fp) is true, fall through to unexpected EOF. */
834  if (!feof(fp)) {
835  if (fakeClient) freeFakeClient(fakeClient); /* avoid valgrind warning */
836  serverLog(LL_WARNING,"Unrecoverable error reading the append only file: %s", strerror(errno));
837  exit(1);
838  }
839 
840 uxeof: /* Unexpected AOF end of file. */
841  if (server.aof_load_truncated) {
842  serverLog(LL_WARNING,"!!! Warning: short read while loading the AOF file !!!");
843  serverLog(LL_WARNING,"!!! Truncating the AOF at offset %llu !!!",
844  (unsigned long long) valid_up_to);
845  if (valid_up_to == -1 || truncate(filename,valid_up_to) == -1) {
846  if (valid_up_to == -1) {
847  serverLog(LL_WARNING,"Last valid command offset is invalid");
848  } else {
849  serverLog(LL_WARNING,"Error truncating the AOF file: %s",
850  strerror(errno));
851  }
852  } else {
853  /* Make sure the AOF file descriptor points to the end of the
854  * file after the truncate call. */
855  if (server.aof_fd != -1 && lseek(server.aof_fd,0,SEEK_END) == -1) {
856  serverLog(LL_WARNING,"Can't seek the end of the AOF file: %s",
857  strerror(errno));
858  } else {
859  serverLog(LL_WARNING,
860  "AOF loaded anyway because aof-load-truncated is enabled");
861  goto loaded_ok;
862  }
863  }
864  }
865  if (fakeClient) freeFakeClient(fakeClient); /* avoid valgrind warning */
866  serverLog(LL_WARNING,"Unexpected end of file reading the append only file. You can: 1) Make a backup of your AOF file, then use ./redis-check-aof --fix <filename>. 2) Alternatively you can set the 'aof-load-truncated' configuration option to yes and restart the server.");
867  exit(1);
868 
869 fmterr: /* Format error. */
870  if (fakeClient) freeFakeClient(fakeClient); /* avoid valgrind warning */
871  serverLog(LL_WARNING,"Bad file format reading the append only file: make a backup of your AOF file, then use ./redis-check-aof --fix <filename>");
872  exit(1);
873 }

1166  if (rioWriteBulkString(r,"XADD",4) == 0) return 0;

1215  ri_pel.key,nack) == 0)
1216  {
1217  return 0;
1218  }
1219  }

1221  }
1222  raxStop(&ri_cons);
1223  }
1224  raxStop(&ri);
