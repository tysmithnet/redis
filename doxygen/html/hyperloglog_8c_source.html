37 /* The Redis HyperLogLog implementation is based on the following ideas:
38  *
39  * * The use of a 64 bit hash function as proposed in [1], in order to don't
40  * limited to cardinalities up to 10^9, at the cost of just 1 additional
41  * bit per register.
42  * * The use of 16384 6-bit registers for a great level of accuracy, using
43  * a total of 12k per key.
44  * * The use of the Redis string data type. No new type is introduced.
45  * * No attempt is made to compress the data structure as in [1]. Also the
46  * algorithm used is the original HyperLogLog Algorithm as in [2], with
47  * the only difference that a 64 bit hash function is used, so no correction
48  * is performed for values near 2^32 as in [1].
49  *
50  * [1] Heule, Nunkesser, Hall: HyperLogLog in Practice: Algorithmic
51  * Engineering of a State of The Art Cardinality Estimation Algorithm.
52  *
53  * [2] P. Flajolet, Éric Fusy, O. Gandouet, and F. Meunier. Hyperloglog: The
54  * analysis of a near-optimal cardinality estimation algorithm.
55  *
56  * Redis uses two representations:
57  *
58  * 1) A "dense" representation where every entry is represented by
59  * a 6-bit integer.
60  * 2) A "sparse" representation using run length compression suitable
61  * for representing HyperLogLogs with many registers set to 0 in
62  * a memory efficient way.
63  *
64  *
65  * HLL header
66  * ===
67  *
68  * Both the dense and sparse representation have a 16 byte header as follows:
69  *
70  * +------+---+-----+----------+
71  * | HYLL | E | N/U | Cardin. |
72  * +------+---+-----+----------+
73  *
74  * The first 4 bytes are a magic string set to the bytes "HYLL".
75  * "E" is one byte encoding, currently set to HLL_DENSE or
76  * HLL_SPARSE. N/U are three not used bytes.
77  *
78  * The "Cardin." field is a 64 bit integer stored in little endian format
79  * with the latest cardinality computed that can be reused if the data
80  * structure was not modified since the last computation (this is useful
81  * because there are high probabilities that HLLADD operations don't
82  * modify the actual data structure and hence the approximated cardinality).
83  *
84  * When the most significant bit in the most significant byte of the cached
85  * cardinality is set, it means that the data structure was modified and
86  * we can't reuse the cached value that must be recomputed.
87  *
88  * Dense representation
89  * ===
90  *
91  * The dense representation used by Redis is the following:
92  *
93  * +--------+--------+--------+------// //--+
94  * |11000000|22221111|33333322|55444444 .... |
95  * +--------+--------+--------+------// //--+
96  *
97  * The 6 bits counters are encoded one after the other starting from the
98  * LSB to the MSB, and using the next bytes as needed.
99  *
100  * Sparse representation
101  * ===
102  *
103  * The sparse representation encodes registers using a run length
104  * encoding composed of three opcodes, two using one byte, and one using
105  * of two bytes. The opcodes are called ZERO, XZERO and VAL.
106  *
107  * ZERO opcode is represented as 00xxxxxx. The 6-bit integer represented
108  * by the six bits 'xxxxxx', plus 1, means that there are N registers set
109  * to 0. This opcode can represent from 1 to 64 contiguous registers set
110  * to the value of 0.
111  *
112  * XZERO opcode is represented by two bytes 01xxxxxx yyyyyyyy. The 14-bit
113  * integer represented by the bits 'xxxxxx' as most significant bits and
114  * 'yyyyyyyy' as least significant bits, plus 1, means that there are N
115  * registers set to 0. This opcode can represent from 0 to 16384 contiguous
116  * registers set to the value of 0.
117  *
118  * VAL opcode is represented as 1vvvvvxx. It contains a 5-bit integer
119  * representing the value of a register, and a 2-bit integer representing
120  * the number of contiguous registers set to that value 'vvvvv'.
121  * To obtain the value and run length, the integers vvvvv and xx must be
122  * incremented by one. This opcode can represent values from 1 to 32,
123  * repeated from 1 to 4 times.
124  *
125  * The sparse representation can't represent registers with a value greater
126  * than 32, however it is very unlikely that we find such a register in an
127  * HLL with a cardinality where the sparse representation is still more
128  * memory efficient than the dense representation. When this happens the
129  * HLL is converted to the dense representation.
130  *
131  * The sparse representation is purely positional. For example a sparse
132  * representation of an empty HLL is just: XZERO:16384.
133  *
134  * An HLL having only 3 non-zero registers at position 1000, 1020, 1021
135  * respectively set to 2, 3, 3, is represented by the following three
136  * opcodes:
137  *
138  * XZERO:1000 (Registers 0-999 are set to 0)
139  * VAL:2,1 (1 register set to value 2, that is register 1000)
140  * ZERO:19 (Registers 1001-1019 set to 0)
141  * VAL:3,2 (2 registers set to value 3, that is registers 1020,1021)
142  * XZERO:15362 (Registers 1022-16383 set to 0)
143  *
144  * In the example the sparse representation used just 7 bytes instead
145  * of 12k in order to represent the HLL registers. In general for low
146  * cardinality there is a big win in terms of space efficiency, traded
147  * with CPU time since the sparse representation is slower to access:
148  *
149  * The following table shows average cardinality vs bytes used, 100
150  * samples per cardinality (when the set was not representable because
151  * of registers with too big value, the dense representation size was used
152  * as a sample).
153  *
154  * 100 267
155  * 200 485
156  * 300 678
157  * 400 859
158  * 500 1033
159  * 600 1205
160  * 700 1375
161  * 800 1544
162  * 900 1713
163  * 1000 1882
164  * 2000 3480
165  * 3000 4879
166  * 4000 6089
167  * 5000 7138
168  * 6000 8042
169  * 7000 8823
170  * 8000 9500
171  * 9000 10088
172  * 10000 10591
173  *
174  * The dense representation uses 12288 bytes, so there is a big win up to
175  * a cardinality of ~2000-3000. For bigger cardinalities the constant times
176  * involved in updating the sparse representation is not justified by the
177  * memory savings. The exact maximum length of the sparse representation
178  * when this implementation switches to the dense representation is
179  * configured via the define server.hll_sparse_max_bytes.
180  */

212 /* Macros to access the dense representation.
213  *
214  * We need to get and set 6 bit counters in an array of 8 bit bytes.
215  * We use macros to make sure the code is inlined since speed is critical
216  * especially in order to compute the approximated cardinality in
217  * HLLCOUNT where we need to access all the registers at once.
218  * For the same reason we also want to avoid conditionals in this code path.
219  *
220  * +--------+--------+--------+------//
221  * |11000000|22221111|33333322|55444444
222  * +--------+--------+--------+------//
223  *
224  * Note: in the above representation the most significant bit (MSB)
225  * of every byte is on the left. We start using bits from the LSB to MSB,
226  * and so forth passing to the next byte.
227  *
228  * Example, we want to access to counter at pos = 1 ("111111" in the
229  * illustration above).
230  *
231  * The index of the first byte b0 containing our data is:
232  *
233  * b0 = 6 * pos / 8 = 0
234  *
235  * +--------+
236  * |11000000| <- Our byte at b0
237  * +--------+
238  *
239  * The position of the first bit (counting from the LSB = 0) in the byte
240  * is given by:
241  *
242  * fb = 6 * pos % 8 -> 6
243  *
244  * Right shift b0 of 'fb' bits.
245  *
246  * +--------+
247  * |11000000| <- Initial value of b0
248  * |00000011| <- After right shift of 6 pos.
249  * +--------+
250  *
251  * Left shift b1 of bits 8-fb bits (2 bits)
252  *
253  * +--------+
254  * |22221111| <- Initial value of b1
255  * |22111100| <- After left shift of 2 bits.
256  * +--------+
257  *
258  * OR the two bits, and finally AND with 111111 (63 in decimal) to
259  * clean the higher order bits we are not interested in:
260  *
261  * +--------+
262  * |00000011| <- b0 right shifted
263  * |22111100| <- b1 left shifted
264  * |22111111| <- b0 OR b1
265  * | 111111| <- (b0 OR b1) AND 63, our value.
266  * +--------+
267  *
268  * We can try with a different example, like pos = 0. In this case
269  * the 6-bit counter is actually contained in a single byte.
270  *
271  * b0 = 6 * pos / 8 = 0
272  *
273  * +--------+
274  * |11000000| <- Our byte at b0
275  * +--------+
276  *
277  * fb = 6 * pos % 8 = 0
278  *
279  * So we right shift of 0 bits (no shift in practice) and
280  * left shift the next byte of 8 bits, even if we don't use it,
281  * but this has the effect of clearing the bits so the result
282  * will not be affacted after the OR.
283  *
284  * -------------------------------------------------------------------------
285  *
286  * Setting the register is a bit more complex, let's assume that 'val'
287  * is the value we want to set, already in the right range.
288  *
289  * We need two steps, in one we need to clear the bits, and in the other
290  * we need to bitwise-OR the new bits.
291  *
292  * Let's try with 'pos' = 1, so our first byte at 'b' is 0,
293  *
294  * "fb" is 6 in this case.
295  *
296  * +--------+
297  * |11000000| <- Our byte at b0
298  * +--------+
299  *
300  * To create a AND-mask to clear the bits about this position, we just
301  * initialize the mask with the value 63, left shift it of "fs" bits,
302  * and finally invert the result.
303  *
304  * +--------+
305  * |00111111| <- "mask" starts at 63
306  * |11000000| <- "mask" after left shift of "ls" bits.
307  * |00111111| <- "mask" after invert.
308  * +--------+
309  *
310  * Now we can bitwise-AND the byte at "b" with the mask, and bitwise-OR
311  * it with "val" left-shifted of "ls" bits to set the new bits.
312  *
313  * Now let's focus on the next byte b1:
314  *
315  * +--------+
316  * |22221111| <- Initial value of b1
317  * +--------+
318  *
319  * To build the AND mask we start again with the 63 value, right shift
320  * it by 8-fb bits, and invert it.
321  *
322  * +--------+
323  * |00111111| <- "mask" set at 2&6-1
324  * |00001111| <- "mask" after the right shift by 8-fb = 2 bits
325  * |11110000| <- "mask" after bitwise not.
326  * +--------+
327  *
328  * Now we can mask it with b+1 to clear the old bits, and bitwise-OR
329  * with "val" left-shifted by "rs" bits to set the new value.
330  */

393 /* Our hash function is MurmurHash2, 64 bit version.
394  * It was modified for Redis in order to provide the same result in
395  * big and little endian archs (endian neutral). */
396 uint64_t MurmurHash64A (const void * key, int len, unsigned int seed) {
397  const uint64_t m = 0xc6a4a7935bd1e995;
398  const int r = 47;
399  uint64_t h = seed ^ (len * m);
400  const uint8_t *data = (const uint8_t *)key;
401  const uint8_t *end = data + (len-(len&7));
402 
403  while(data != end) {
404  uint64_t k;
405 
406 #if (BYTE_ORDER == LITTLE_ENDIAN)
407  #ifdef USE_ALIGNED_ACCESS
408  memcpy(&k,data,sizeof(uint64_t));
409  #else
410  k = *((uint64_t*)data);
411  #endif
412 #else
413  k = (uint64_t) data[0];
414  k |= (uint64_t) data[1] << 8;
415  k |= (uint64_t) data[2] << 16;
416  k |= (uint64_t) data[3] << 24;
417  k |= (uint64_t) data[4] << 32;
418  k |= (uint64_t) data[5] << 40;
419  k |= (uint64_t) data[6] << 48;
420  k |= (uint64_t) data[7] << 56;
421 #endif
422 
423  k *= m;
424  k ^= k >> r;
425  k *= m;
426  h ^= k;
427  h *= m;
428  data += 8;
429  }
430 
431  switch(len & 7) {
432  case 7: h ^= (uint64_t)data[6] << 48; /* fall-thru */
433  case 6: h ^= (uint64_t)data[5] << 40; /* fall-thru */
434  case 5: h ^= (uint64_t)data[4] << 32; /* fall-thru */
435  case 4: h ^= (uint64_t)data[3] << 24; /* fall-thru */
436  case 3: h ^= (uint64_t)data[2] << 16; /* fall-thru */
437  case 2: h ^= (uint64_t)data[1] << 8; /* fall-thru */
438  case 1: h ^= (uint64_t)data[0];
439  h *= m; /* fall-thru */
440  };
441 
442  h ^= h >> r;
443  h *= m;
444  h ^= h >> r;
445  return h;
446 }

448 /* Given a string element to add to the HyperLogLog, returns the length
449  * of the pattern 000..1 of the element hash. As a side effect 'regp' is
450  * set to the register index this element hashes to. */
451 int hllPatLen(unsigned char *ele, size_t elesize, long *regp) {
452  uint64_t hash, bit, index;
453  int count;
454 
455  /* Count the number of zeroes starting from bit HLL_REGISTERS
456  * (that is a power of two corresponding to the first bit we don't use
457  * as index). The max run can be 64-P+1 = Q+1 bits.
458  *
459  * Note that the final "1" ending the sequence of zeroes must be
460  * included in the count, so if we find "001" the count is 3, and
461  * the smallest count possible is no zeroes at all, just a 1 bit
462  * at the first position, that is a count of 1.
463  *
464  * This may sound like inefficient, but actually in the average case
465  * there are high probabilities to find a 1 after a few iterations. */
466  hash = MurmurHash64A(ele,elesize,0xadc83b19ULL);
467  index = hash & HLL_P_MASK; /* Register index. */
468  hash >>= HLL_P; /* Remove bits used to address the register. */
469  hash |= ((uint64_t)1<<HLL_Q); /* Make sure the loop terminates
470  and count will be <= Q+1. */
471  bit = 1;
472  count = 1; /* Initialized to 1 since we count the "00000...1" pattern. */
473  while((hash & bit) == 0) {
474  count++;
475  bit <<= 1;
476  }
477  *regp = (int) index;
478  return count;
479 }

518 /* Compute the register histogram in the dense representation. */
519 void hllDenseRegHisto(uint8_t *registers, int* reghisto) {
520  int j;
521 
522  /* Redis default is to use 16384 registers 6 bits each. The code works
523  * with other values by modifying the defines, but for our target value
524  * we take a faster path with unrolled loops. */
525  if (HLL_REGISTERS == 16384 && HLL_BITS == 6) {
526  uint8_t *r = registers;
527  unsigned long r0, r1, r2, r3, r4, r5, r6, r7, r8, r9,
528  r10, r11, r12, r13, r14, r15;
529  for (j = 0; j < 1024; j++) {
530  /* Handle 16 registers per iteration. */
531  r0 = r[0] & 63;
532  r1 = (r[0] >> 6 | r[1] << 2) & 63;
533  r2 = (r[1] >> 4 | r[2] << 4) & 63;
534  r3 = (r[2] >> 2) & 63;
535  r4 = r[3] & 63;
536  r5 = (r[3] >> 6 | r[4] << 2) & 63;
537  r6 = (r[4] >> 4 | r[5] << 4) & 63;
538  r7 = (r[5] >> 2) & 63;
539  r8 = r[6] & 63;
540  r9 = (r[6] >> 6 | r[7] << 2) & 63;
541  r10 = (r[7] >> 4 | r[8] << 4) & 63;
542  r11 = (r[8] >> 2) & 63;
543  r12 = r[9] & 63;
544  r13 = (r[9] >> 6 | r[10] << 2) & 63;
545  r14 = (r[10] >> 4 | r[11] << 4) & 63;

578 /* Convert the HLL with sparse representation given as input in its dense
579  * representation. Both representations are represented by SDS strings, and
580  * the input representation is freed as a side effect.
581  *
582  * The function returns C_OK if the sparse representation was valid,
583  * otherwise C_ERR is returned if the representation was corrupted. */
584 int hllSparseToDense(robj *o) {
585  sds sparse = o->ptr, dense;
586  struct hllhdr *hdr, *oldhdr = (struct hllhdr*)sparse;
587  int idx = 0, runlen, regval;
588  uint8_t *p = (uint8_t*)sparse, *end = p+sdslen(sparse);
589 
590  /* If the representation is already the right one return ASAP. */
591  hdr = (struct hllhdr*) sparse;
592  if (hdr->encoding == HLL_DENSE) return C_OK;
593 
594  /* Create a string of the right size filled with zero bytes.
595  * Note that the cached cardinality is set to 0 as a side effect
596  * that is exactly the cardinality of an empty HLL. */
597  dense = sdsnewlen(NULL,HLL_DENSE_SIZE);
598  hdr = (struct hllhdr*) dense;
599  *hdr = *oldhdr; /* This will copy the magic and cached cardinality. */
600  hdr->encoding = HLL_DENSE;
601 
602  /* Now read the sparse representation and set non-zero registers
603  * accordingly. */
604  p += HLL_HDR_SIZE;
605  while(p < end) {
606  if (HLL_SPARSE_IS_ZERO(p)) {
607  runlen = HLL_SPARSE_ZERO_LEN(p);
608  idx += runlen;
609  p++;
610  } else if (HLL_SPARSE_IS_XZERO(p)) {
611  runlen = HLL_SPARSE_XZERO_LEN(p);
612  idx += runlen;
613  p += 2;
614  } else {
615  runlen = HLL_SPARSE_VAL_LEN(p);
616  regval = HLL_SPARSE_VAL_VALUE(p);
617  while(runlen--) {
618  HLL_DENSE_SET_REGISTER(hdr->registers,idx,regval);
619  idx++;
620  }
621  p++;
622  }
623  }
624 
625  /* If the sparse representation was valid, we expect to find idx
626  * set to HLL_REGISTERS. */
627  if (idx != HLL_REGISTERS) {
628  sdsfree(dense);
629  return C_ERR;
630  }
631 
632  /* Free the old representation and set the new one. */
633  sdsfree(o->ptr);
634  o->ptr = dense;
635  return C_OK;
636 }

638 /* Low level function to set the sparse HLL register at 'index' to the
639  * specified value if the current value is smaller than 'count'.
640  *
641  * The object 'o' is the String object holding the HLL. The function requires
642  * a reference to the object in order to be able to enlarge the string if
643  * needed.
644  *
645  * On success, the function returns 1 if the cardinality changed, or 0
646  * if the register for this element was not updated.
647  * On error (if the representation is invalid) -1 is returned.
648  *
649  * As a side effect the function may promote the HLL representation from
650  * sparse to dense: this happens when a register requires to be set to a value
651  * not representable with the sparse representation, or when the resulting
652  * size would be greater than server.hll_sparse_max_bytes. */
653 int hllSparseSet(robj *o, long index, uint8_t count) {
654  struct hllhdr *hdr;
655  uint8_t oldcount, *sparse, *end, *p, *prev, *next;
656  long first, span;
657  long is_zero = 0, is_xzero = 0, is_val = 0, runlen = 0;
658 
659  /* If the count is too big to be representable by the sparse representation
660  * switch to dense representation. */
661  if (count > HLL_SPARSE_VAL_MAX_VALUE) goto promote;
662 
663  /* When updating a sparse representation, sometimes we may need to
664  * enlarge the buffer for up to 3 bytes in the worst case (XZERO split
665  * into XZERO-VAL-XZERO). Make sure there is enough space right now
666  * so that the pointers we take during the execution of the function
667  * will be valid all the time. */
668  o->ptr = sdsMakeRoomFor(o->ptr,3);
669 
670  /* Step 1: we need to locate the opcode we need to modify to check
671  * if a value update is actually needed. */
672  sparse = p = ((uint8_t*)o->ptr) + HLL_HDR_SIZE;
673  end = p + sdslen(o->ptr) - HLL_HDR_SIZE;
674 
675  first = 0;
676  prev = NULL; /* Points to previous opcode at the end of the loop. */
677  next = NULL; /* Points to the next opcode at the end of the loop. */
678  span = 0;
679  while(p < end) {
680  long oplen;
681 
682  /* Set span to the number of registers covered by this opcode.
683  *
684  * This is the most performance critical loop of the sparse
685  * representation. Sorting the conditionals from the most to the
686  * least frequent opcode in many-bytes sparse HLLs is faster. */
687  oplen = 1;
688  if (HLL_SPARSE_IS_ZERO(p)) {
689  span = HLL_SPARSE_ZERO_LEN(p);
690  } else if (HLL_SPARSE_IS_VAL(p)) {
691  span = HLL_SPARSE_VAL_LEN(p);
692  } else { /* XZERO. */
693  span = HLL_SPARSE_XZERO_LEN(p);
694  oplen = 2;
695  }
696  /* Break if this opcode covers the register as 'index'. */
697  if (index <= first+span-1) break;
698  prev = p;
699  p += oplen;
700  first += span;
701  }
702  if (span == 0) return -1; /* Invalid format. */
703 
704  next = HLL_SPARSE_IS_XZERO(p) ? p+2 : p+1;
705  if (next >= end) next = NULL;
706 
707  /* Cache current opcode type to avoid using the macro again and
708  * again for something that will not change.
709  * Also cache the run-length of the opcode. */
710  if (HLL_SPARSE_IS_ZERO(p)) {
711  is_zero = 1;
712  runlen = HLL_SPARSE_ZERO_LEN(p);
713  } else if (HLL_SPARSE_IS_XZERO(p)) {
714  is_xzero = 1;
715  runlen = HLL_SPARSE_XZERO_LEN(p);
716  } else {
717  is_val = 1;
718  runlen = HLL_SPARSE_VAL_LEN(p);
719  }
720 
721  /* Step 2: After the loop:
722  *
723  * 'first' stores to the index of the first register covered
724  * by the current opcode, which is pointed by 'p'.
725  *
726  * 'next' ad 'prev' store respectively the next and previous opcode,
727  * or NULL if the opcode at 'p' is respectively the last or first.
728  *
729  * 'span' is set to the number of registers covered by the current
730  * opcode.
731  *
732  * There are different cases in order to update the data structure
733  * in place without generating it from scratch:
734  *
735  * A) If it is a VAL opcode already set to a value >= our 'count'
736  * no update is needed, regardless of the VAL run-length field.
737  * In this case PFADD returns 0 since no changes are performed.
738  *
739  * B) If it is a VAL opcode with len = 1 (representing only our
740  * register) and the value is less than 'count', we just update it
741  * since this is a trivial case. */
742  if (is_val) {
743  oldcount = HLL_SPARSE_VAL_VALUE(p);
744  /* Case A. */
745  if (oldcount >= count) return 0;
746 
747  /* Case B. */
748  if (runlen == 1) {
749  HLL_SPARSE_VAL_SET(p,count,1);
750  goto updated;
751  }
752  }
753 
754  /* C) Another trivial to handle case is a ZERO opcode with a len of 1.
755  * We can just replace it with a VAL opcode with our value and len of 1. */
756  if (is_zero && runlen == 1) {
757  HLL_SPARSE_VAL_SET(p,count,1);
758  goto updated;
759  }
760 
761  /* D) General case.
762  *
763  * The other cases are more complex: our register requires to be updated
764  * and is either currently represented by a VAL opcode with len > 1,
765  * by a ZERO opcode with len > 1, or by an XZERO opcode.
766  *
767  * In those cases the original opcode must be split into multiple
768  * opcodes. The worst case is an XZERO split in the middle resuling into
769  * XZERO - VAL - XZERO, so the resulting sequence max length is
770  * 5 bytes.
771  *
772  * We perform the split writing the new sequence into the 'new' buffer
773  * with 'newlen' as length. Later the new sequence is inserted in place
774  * of the old one, possibly moving what is on the right a few bytes
775  * if the new sequence is longer than the older one. */
776  uint8_t seq[5], *n = seq;
777  int last = first+span-1; /* Last register covered by the sequence. */
778  int len;
779 
780  if (is_zero || is_xzero) {
781  /* Handle splitting of ZERO / XZERO. */
782  if (index != first) {
783  len = index-first;
784  if (len > HLL_SPARSE_ZERO_MAX_LEN) {
785  HLL_SPARSE_XZERO_SET(n,len);
786  n += 2;
787  } else {
788  HLL_SPARSE_ZERO_SET(n,len);
789  n++;
790  }
791  }
792  HLL_SPARSE_VAL_SET(n,count,1);
793  n++;
794  if (index != last) {
795  len = last-index;
796  if (len > HLL_SPARSE_ZERO_MAX_LEN) {
797  HLL_SPARSE_XZERO_SET(n,len);
798  n += 2;
799  } else {
800  HLL_SPARSE_ZERO_SET(n,len);
801  n++;
802  }
803  }
804  } else {
805  /* Handle splitting of VAL. */
806  int curval = HLL_SPARSE_VAL_VALUE(p);
807 
808  if (index != first) {
809  len = index-first;
810  HLL_SPARSE_VAL_SET(n,curval,len);
811  n++;
812  }
813  HLL_SPARSE_VAL_SET(n,count,1);
814  n++;
815  if (index != last) {
816  len = last-index;
817  HLL_SPARSE_VAL_SET(n,curval,len);
818  n++;
819  }
820  }
821 
822  /* Step 3: substitute the new sequence with the old one.
823  *
824  * Note that we already allocated space on the sds string
825  * calling sdsMakeRoomFor(). */
826  int seqlen = n-seq;
827  int oldlen = is_xzero ? 2 : 1;
828  int deltalen = seqlen-oldlen;
829 
830  if (deltalen > 0 &&
831  sdslen(o->ptr)+deltalen > server.hll_sparse_max_bytes) goto promote;
832  if (deltalen && next) memmove(next+deltalen,next,end-next);
833  sdsIncrLen(o->ptr,deltalen);
834  memcpy(p,seq,seqlen);
835  end += deltalen;
836 
837 updated:
838  /* Step 4: Merge adjacent values if possible.
839  *
840  * The representation was updated, however the resulting representation
841  * may not be optimal: adjacent VAL opcodes can sometimes be merged into
842  * a single one. */
843  p = prev ? prev : sparse;
844  int scanlen = 5; /* Scan up to 5 upcodes starting from prev. */
845  while (p < end && scanlen--) {
846  if (HLL_SPARSE_IS_XZERO(p)) {
847  p += 2;
848  continue;
849  } else if (HLL_SPARSE_IS_ZERO(p)) {
850  p++;
851  continue;
852  }
853  /* We need two adjacent VAL opcodes to try a merge, having
854  * the same value, and a len that fits the VAL opcode max len. */
855  if (p+1 < end && HLL_SPARSE_IS_VAL(p+1)) {
856  int v1 = HLL_SPARSE_VAL_VALUE(p);
857  int v2 = HLL_SPARSE_VAL_VALUE(p+1);
858  if (v1 == v2) {
859  int len = HLL_SPARSE_VAL_LEN(p)+HLL_SPARSE_VAL_LEN(p+1);
860  if (len <= HLL_SPARSE_VAL_MAX_LEN) {
861  HLL_SPARSE_VAL_SET(p+1,v1,len);
862  memmove(p,p+1,end-p);
863  sdsIncrLen(o->ptr,-1);
864  end--;
865  /* After a merge we reiterate without incrementing 'p'
866  * in order to try to merge the just merged value with
867  * a value on its right. */
868  continue;
869  }
870  }
871  }
872  p++;
873  }
874 
875  /* Invalidate the cached cardinality. */
876  hdr = o->ptr;
877  HLL_INVALIDATE_CACHE(hdr);
878  return 1;
879 
880 promote: /* Promote to dense representation. */
881  if (hllSparseToDense(o) == C_ERR) return -1; /* Corrupted HLL. */
882  hdr = o->ptr;
883 
884  /* We need to call hllDenseAdd() to perform the operation after the
885  * conversion. However the result must be 1, since if we need to
886  * convert from sparse to dense a register requires to be updated.
887  *
888  * Note that this in turn means that PFADD will make sure the command
889  * is propagated to slaves / AOF, so if there is a sparse -> dense
890  * conversion, it will be performed in all the slaves as well. */
891  int dense_retval = hllDenseSet(hdr->registers,index,count);
892  serverAssert(dense_retval == 1);
893  return dense_retval;
894 }

967 /* Helper function sigma as defined in
968  * "New cardinality estimation algorithms for HyperLogLog sketches"
969  * Otmar Ertl, arXiv:1702.01284 */
970 double hllSigma(double x) {
971  if (x == 1.) return INFINITY;
972  double zPrime;
973  double y = 1;
974  double z = x;
975  do {
976  x *= x;
977  zPrime = z;
978  z += x * y;
979  y += y;
980  } while(zPrime != z);
981  return z;
982 }

984 /* Helper function tau as defined in
985  * "New cardinality estimation algorithms for HyperLogLog sketches"
986  * Otmar Ertl, arXiv:1702.01284 */
987 double hllTau(double x) {
988  if (x == 0. || x == 1.) return 0.;
989  double zPrime;
990  double y = 1.0;
991  double z = 1 - x;
992  do {
993  x = sqrt(x);
994  zPrime = z;
995  y *= 0.5;
996  z -= pow(1 - x, 2)*y;
997  } while(zPrime != z);
998  return z / 3;
999 }

1001 /* Return the approximated cardinality of the set based on the harmonic
1002  * mean of the registers values. 'hdr' points to the start of the SDS
1003  * representing the String object holding the HLL representation.
1004  *
1005  * If the sparse representation of the HLL object is not valid, the integer
1006  * pointed by 'invalid' is set to non-zero, otherwise it is left untouched.
1007  *
1008  * hllCount() supports a special internal-only encoding of HLL_RAW, that
1009  * is, hdr->registers will point to an uint8_t array of HLL_REGISTERS element.
1010  * This is useful in order to speedup PFCOUNT when called against multiple
1011  * keys (no need to work with 6-bit integers encoding). */
1012 uint64_t hllCount(struct hllhdr *hdr, int *invalid) {
1013  double m = HLL_REGISTERS;
1014  double E;
1015  int j;
1016  int reghisto[HLL_Q+2] = {0};
1017 
1018  /* Compute register histogram */
1019  if (hdr->encoding == HLL_DENSE) {
1020  hllDenseRegHisto(hdr->registers,reghisto);
1021  } else if (hdr->encoding == HLL_SPARSE) {
1022  hllSparseRegHisto(hdr->registers,
1023  sdslen((sds)hdr)-HLL_HDR_SIZE,invalid,reghisto);
1024  } else if (hdr->encoding == HLL_RAW) {
1025  hllRawRegHisto(hdr->registers,reghisto);
1026  } else {
1027  serverPanic("Unknown HyperLogLog encoding in hllCount()");
1028  }
1029 
1030  /* Estimate cardinality form register histogram. See:
1031  * "New cardinality estimation algorithms for HyperLogLog sketches"
1032  * Otmar Ertl, arXiv:1702.01284 */
1033  double z = m * hllTau((m-reghisto[HLL_Q+1])/(double)m);
1034  for (j = HLL_Q; j >= 1; --j) {
1035  z += reghisto[j];
1036  z *= 0.5;
1037  }
1038  z += m * hllSigma(reghisto[0]/(double)m);
1039  E = llroundl(HLL_ALPHA_INF*m*m/z);
1040 
1041  return (uint64_t) E;
1042 }

1056  *
1057  * The hll object must be already validated via isHLLObjectOrReply()

1105 /* Create an HLL object. We always create the HLL using sparse encoding.
1106  * This will be upgraded to the dense representation as needed. */
1107 robj *createHLLObject(void) {
1108  robj *o;
1109  struct hllhdr *hdr;
1110  sds s;
1111  uint8_t *p;
1112  int sparselen = HLL_HDR_SIZE +
1113  (((HLL_REGISTERS+(HLL_SPARSE_XZERO_MAX_LEN-1)) /
1114  HLL_SPARSE_XZERO_MAX_LEN)*2);
1115  int aux;
1116 
1117  /* Populate the sparse representation with as many XZERO opcodes as
1118  * needed to represent all the registers. */
1119  aux = HLL_REGISTERS;
1120  s = sdsnewlen(NULL,sparselen);
1121  p = (uint8_t*)s + HLL_HDR_SIZE;
1122  while(aux) {
1123  int xzero = HLL_SPARSE_XZERO_MAX_LEN;
1124  if (xzero > aux) xzero = aux;
1125  HLL_SPARSE_XZERO_SET(p,xzero);
1126  p += 2;
1127  aux -= xzero;
1128  }
1129  serverAssert((p-(uint8_t*)s) == sparselen);
1130 
1131  /* Create the actual object. */
1132  o = createObject(OBJ_STRING,s);
1133  hdr = o->ptr;
1134  memcpy(hdr->magic,"HYLL",4);

1379 /* PFSELFTEST
1380  * This command performs a self-test of the HLL registers implementation.
1381  * Something that is not easy to test from within the outside. */
1382 #define HLL_TEST_CYCLES 1000
1383 void pfselftestCommand(client *c) {
1384  unsigned int j, i;
1385  sds bitcounters = sdsnewlen(NULL,HLL_DENSE_SIZE);
1386  struct hllhdr *hdr = (struct hllhdr*) bitcounters, *hdr2;
1387  robj *o = NULL;
1388  uint8_t bytecounters[HLL_REGISTERS];
1389 
1390  /* Test 1: access registers.
1391  * The test is conceived to test that the different counters of our data
1392  * structure are accessible and that setting their values both result in
1393  * the correct value to be retained and not affect adjacent values. */
1394  for (j = 0; j < HLL_TEST_CYCLES; j++) {
1395  /* Set the HLL counters and an array of unsigned byes of the
1396  * same size to the same set of random values. */
1397  for (i = 0; i < HLL_REGISTERS; i++) {
1398  unsigned int r = rand() & HLL_REGISTER_MAX;
1399 
1400  bytecounters[i] = r;
1401  HLL_DENSE_SET_REGISTER(hdr->registers,i,r);
1402  }
1403  /* Check that we are able to retrieve the same values. */
1404  for (i = 0; i < HLL_REGISTERS; i++) {
1405  unsigned int val;
1406 
1407  HLL_DENSE_GET_REGISTER(val,hdr->registers,i);
1408  if (val != bytecounters[i]) {
1409  addReplyErrorFormat(c,
1410  "TESTFAILED Register %d should be %d but is %d",
1411  i, (int) bytecounters[i], (int) val);
1412  goto cleanup;
1413  }
1414  }
1415  }
1416 
1417  /* Test 2: approximation error.
1418  * The test adds unique elements and check that the estimated value
1419  * is always reasonable bounds.
1420  *
1421  * We check that the error is smaller than a few times than the expected
1422  * standard error, to make it very unlikely for the test to fail because
1423  * of a "bad" run.
1424  *
1425  * The test is performed with both dense and sparse HLLs at the same
1426  * time also verifying that the computed cardinality is the same. */
1427  memset(hdr->registers,0,HLL_DENSE_SIZE-HLL_HDR_SIZE);
1428  o = createHLLObject();
1429  double relerr = 1.04/sqrt(HLL_REGISTERS);

1431  uint64_t seed = (uint64_t)rand() | (uint64_t)rand() << 32;
1432  uint64_t ele;
1433  for (j = 1; j <= 10000000; j++) {
1434  ele = j ^ seed;

1436  hllAdd(o,(unsigned char*)&ele,sizeof(ele));

1439  * encoding. */

1441  hdr2 = o->ptr;
1442  if (hdr2->encoding != HLL_SPARSE) {
1443  addReplyError(c, "TESTFAILED sparse encoding not used");
1444  goto cleanup;
1445  }
1446  }
1447 
1448  /* Check that dense and sparse representations agree. */
1449  if (j == checkpoint && hllCount(hdr,NULL) != hllCount(o->ptr,NULL)) {
1450  addReplyError(c, "TESTFAILED dense/sparse disagree");
1451  goto cleanup;
1452  }
1453 
1454  /* Check error. */
1455  if (j == checkpoint) {
1456  int64_t abserr = checkpoint - (int64_t)hllCount(hdr,NULL);
1457  uint64_t maxerr = ceil(relerr*6*checkpoint);

1460  * since from time to time it is statistically likely to get
1461  * much higher error due to collision, resulting into a false
1462  * positive. */

1464 
1465  if (abserr < 0) abserr = -abserr;
1466  if (abserr > (int64_t)maxerr) {
1467  addReplyErrorFormat(c,
1468  "TESTFAILED Too big error. card:%llu abserr:%llu",
1469  (unsigned long long) checkpoint,
1470  (unsigned long long) abserr);
1471  goto cleanup;
1472  }
1473  checkpoint *= 10;
1474  }
1475  }
1476

1478  addReply(c,shared.ok);
1479

1481  sdsfree(bitcounters);
1482  if (o) decrRefCount(o);
