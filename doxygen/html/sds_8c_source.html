44 static inline int sdsHdrSize(char type) {
45  switch(type&SDS_TYPE_MASK) {
46  case SDS_TYPE_5:
47  return sizeof(struct sdshdr5);
48  case SDS_TYPE_8:
49  return sizeof(struct sdshdr8);

52  case SDS_TYPE_32:
53  return sizeof(struct sdshdr32);
54  case SDS_TYPE_64:
55  return sizeof(struct sdshdr64);
56  }
57  return 0;

60 static inline char sdsReqType(size_t string_size) {
61  if (string_size < 1<<5)
62  return SDS_TYPE_5;
63  if (string_size < 1<<8)
64  return SDS_TYPE_8;
65  if (string_size < 1<<16)
66  return SDS_TYPE_16;
67 #if (LONG_MAX == LLONG_MAX)
68  if (string_size < 1ll<<32)
69  return SDS_TYPE_32;
70  return SDS_TYPE_64;
71 #else
72  return SDS_TYPE_32;
73 #endif
74 }

76 /* Create a new sds string with the content specified by the 'init' pointer
77  * and 'initlen'.
78  * If NULL is used for 'init' the string is initialized with zero bytes.
79  * If SDS_NOINIT is used, the buffer is left uninitialized;
80  *
81  * The string is always null-termined (all the sds strings are, always) so
82  * even if you create an sds string with:
83  *
84  * mystring = sdsnewlen("abc",3);
85  *
86  * You can print the string with printf() as there is an implicit \0 at the
87  * end of the string. However the string is binary safe and can contain
88  * \0 characters in the middle, as the length is stored in the sds header. */
89 sds sdsnewlen(const void *init, size_t initlen) {
90  void *sh;
91  sds s;
92  char type = sdsReqType(initlen);
93  /* Empty strings are usually created in order to append. Use type 8
94  * since type 5 is not good at this. */
95  if (type == SDS_TYPE_5 && initlen == 0) type = SDS_TYPE_8;
96  int hdrlen = sdsHdrSize(type);
97  unsigned char *fp; /* flags pointer. */
98 
99  sh = s_malloc(hdrlen+initlen+1);
100  if (init==SDS_NOINIT)
101  init = NULL;
102  else if (!init)
103  memset(sh, 0, hdrlen+initlen+1);
104  if (sh == NULL) return NULL;
105  s = (char*)sh+hdrlen;
106  fp = ((unsigned char*)s)-1;
107  switch(type) {
108  case SDS_TYPE_5: {
109  *fp = type | (initlen << SDS_TYPE_BITS);
110  break;
111  }
112  case SDS_TYPE_8: {
113  SDS_HDR_VAR(8,s);
114  sh->len = initlen;
115  sh->alloc = initlen;
116  *fp = type;
117  break;
118  }
119  case SDS_TYPE_16: {
120  SDS_HDR_VAR(16,s);
121  sh->len = initlen;
122  sh->alloc = initlen;
123  *fp = type;
124  break;
125  }
126  case SDS_TYPE_32: {
127  SDS_HDR_VAR(32,s);
128  sh->len = initlen;
129  sh->alloc = initlen;
130  *fp = type;
131  break;
132  }
133  case SDS_TYPE_64: {
134  SDS_HDR_VAR(64,s);
135  sh->len = initlen;
136  sh->alloc = initlen;
137  *fp = type;
138  break;
139  }
140  }
141  if (initlen && init)
142  memcpy(s, init, initlen);
143  s[initlen] = '\0';
144  return s;
145 }

198 /* Enlarge the free space at the end of the sds string so that the caller
199  * is sure that after calling this function can overwrite up to addlen
200  * bytes after the end of the string, plus one more byte for nul term.
201  *
202  * Note: this does not change the *length* of the sds string as returned
203  * by sdslen(), but only the free buffer space we have. */
204 sds sdsMakeRoomFor(sds s, size_t addlen) {
205  void *sh, *newsh;
206  size_t avail = sdsavail(s);
207  size_t len, newlen;
208  char type, oldtype = s[-1] & SDS_TYPE_MASK;
209  int hdrlen;
210 
211  /* Return ASAP if there is enough space left. */
212  if (avail >= addlen) return s;
213 
214  len = sdslen(s);
215  sh = (char*)s-sdsHdrSize(oldtype);
216  newlen = (len+addlen);
217  if (newlen < SDS_MAX_PREALLOC)
218  newlen *= 2;
219  else
220  newlen += SDS_MAX_PREALLOC;
221 
222  type = sdsReqType(newlen);
223 
224  /* Don't use type 5: the user is appending to the string and type 5 is
225  * not able to remember empty space, so sdsMakeRoomFor() must be called
226  * at every appending operation. */
227  if (type == SDS_TYPE_5) type = SDS_TYPE_8;
228 
229  hdrlen = sdsHdrSize(type);
230  if (oldtype==type) {
231  newsh = s_realloc(sh, hdrlen+newlen+1);
232  if (newsh == NULL) return NULL;
233  s = (char*)newsh+hdrlen;
234  } else {
235  /* Since the header size changes, need to move the string forward,
236  * and can't use realloc */
237  newsh = s_malloc(hdrlen+newlen+1);
238  if (newsh == NULL) return NULL;
239  memcpy((char*)newsh+hdrlen, s, len+1);
240  s_free(sh);
241  s = (char*)newsh+hdrlen;
242  s[-1] = type;
243  sdssetlen(s, len);
244  }
245  sdssetalloc(s, newlen);
246  return s;
247 }

249 /* Reallocate the sds string so that it has no free space at the end. The
250  * contained string remains not altered, but next concatenation operations
251  * will require a reallocation.
252  *
253  * After the call, the passed sds string is no longer valid and all the
254  * references must be substituted with the new pointer returned by the call. */
255 sds sdsRemoveFreeSpace(sds s) {
256  void *sh, *newsh;
257  char type, oldtype = s[-1] & SDS_TYPE_MASK;
258  int hdrlen, oldhdrlen = sdsHdrSize(oldtype);
259  size_t len = sdslen(s);
260  sh = (char*)s-oldhdrlen;
261 
262  /* Check what would be the minimum SDS header that is just good enough to
263  * fit this string. */
264  type = sdsReqType(len);
265  hdrlen = sdsHdrSize(type);
266 
267  /* If the type is the same, or at least a large enough type is still
268  * required, we just realloc(), letting the allocator to do the copy
269  * only if really needed. Otherwise if the change is huge, we manually
270  * reallocate the string to use the different header type. */
271  if (oldtype==type || type > SDS_TYPE_8) {
272  newsh = s_realloc(sh, oldhdrlen+len+1);
273  if (newsh == NULL) return NULL;
274  s = (char*)newsh+oldhdrlen;
275  } else {
276  newsh = s_malloc(hdrlen+len+1);
277  if (newsh == NULL) return NULL;
278  memcpy((char*)newsh+hdrlen, s, len+1);
279  s_free(sh);
280  s = (char*)newsh+hdrlen;
281  s[-1] = type;
282  sdssetlen(s, len);
283  }
284  sdssetalloc(s, len);
285  return s;
286 }

306 /* Increment the sds length and decrements the left free space at the
307  * end of the string according to 'incr'. Also set the null term
308  * in the new end of the string.
309  *
310  * This function is used in order to fix the string length after the
311  * user calls sdsMakeRoomFor(), writes something after the end of
312  * the current string, and finally needs to set the new length.
313  *
314  * Note: it is possible to use a negative increment in order to
315  * right-trim the string.
316  *
317  * Usage example:
318  *
319  * Using sdsIncrLen() and sdsMakeRoomFor() it is possible to mount the
320  * following schema, to cat bytes coming from the kernel to the end of an
321  * sds string without copying into an intermediate buffer:
322  *
323  * oldlen = sdslen(s);
324  * s = sdsMakeRoomFor(s, BUFFER_SIZE);
325  * nread = read(fd, s+oldlen, BUFFER_SIZE);
326  * ... check for nread <= 0 and handle it ...
327  * sdsIncrLen(s, nread);
328  */
329 void sdsIncrLen(sds s, ssize_t incr) {
330  unsigned char flags = s[-1];
331  size_t len;
332  switch(flags&SDS_TYPE_MASK) {
333  case SDS_TYPE_5: {
334  unsigned char *fp = ((unsigned char*)s)-1;
335  unsigned char oldlen = SDS_TYPE_5_LEN(flags);
336  assert((incr > 0 && oldlen+incr < 32) || (incr < 0 && oldlen >= (unsigned int)(-incr)));
337  *fp = SDS_TYPE_5 | ((oldlen+incr) << SDS_TYPE_BITS);
338  len = oldlen+incr;
339  break;
340  }
341  case SDS_TYPE_8: {
342  SDS_HDR_VAR(8,s);
343  assert((incr >= 0 && sh->alloc-sh->len >= incr) || (incr < 0 && sh->len >= (unsigned int)(-incr)));
344  len = (sh->len += incr);
345  break;
346  }
347  case SDS_TYPE_16: {
348  SDS_HDR_VAR(16,s);
349  assert((incr >= 0 && sh->alloc-sh->len >= incr) || (incr < 0 && sh->len >= (unsigned int)(-incr)));
350  len = (sh->len += incr);
351  break;
352  }
353  case SDS_TYPE_32: {
354  SDS_HDR_VAR(32,s);
355  assert((incr >= 0 && sh->alloc-sh->len >= (unsigned int)incr) || (incr < 0 && sh->len >= (unsigned int)(-incr)));
356  len = (sh->len += incr);
357  break;
358  }
359  case SDS_TYPE_64: {
360  SDS_HDR_VAR(64,s);
361  assert((incr >= 0 && sh->alloc-sh->len >= (uint64_t)incr) || (incr < 0 && sh->len >= (uint64_t)(-incr)));
362  len = (sh->len += incr);
363  break;
364  }
365  default: len = 0; /* Just to avoid compilation warnings. */
366  }
367  s[len] = '\0';
368 }

370 /* Grow the sds to have the specified length. Bytes that were not part of
371  * the original length of the sds will be set to zero.
372  *
373  * if the specified length is smaller than the current length, no operation
374  * is performed. */
375 sds sdsgrowzero(sds s, size_t len) {
376  size_t curlen = sdslen(s);
377 
378  if (len <= curlen) return s;
379  s = sdsMakeRoomFor(s,len-curlen);
380  if (s == NULL) return NULL;
381 
382  /* Make sure added region doesn't contain garbage */
383  memset(s+curlen,0,(len-curlen+1)); /* also set trailing \0 byte */
384  sdssetlen(s, len);
385  return s;
386 }

439 /* Helper for sdscatlonglong() doing the actual number -> string
440  * conversion. 's' must point to a string with room for at least
441  * SDS_LLSTR_SIZE bytes.
442  *
443  * The function returns the length of the null-terminated string
444  * representation stored at 's'. */
445 #define SDS_LLSTR_SIZE 21
446 int sdsll2str(char *s, long long value) {
447  char *p, aux;
448  unsigned long long v;
449  size_t l;
450 
451  /* Generate the string representation, this method produces
452  * an reversed string. */
453  v = (value < 0) ? -value : value;
454  p = s;
455  do {
456  *p++ = '0'+(v%10);
457  v /= 10;
458  } while(v);
459  if (value < 0) *p++ = '-';
460 
461  /* Compute length and add null term. */
462  l = p-s;
463  *p = '\0';
464 
465  /* Reverse the string. */
466  p--;
467  while(s < p) {
468  aux = *s;
469  *s = *p;
470  *p = aux;
471  s++;
472  p--;
473  }
474  return l;
475 }

517 /* Like sdscatprintf() but gets va_list instead of being variadic. */
518 sds sdscatvprintf(sds s, const char *fmt, va_list ap) {
519  va_list cpy;
520  char staticbuf[1024], *buf = staticbuf, *t;
521  size_t buflen = strlen(fmt)*2;
522 
523  /* We try to start using a static buffer for speed.
524  * If not possible we revert to heap allocation. */
525  if (buflen > sizeof(staticbuf)) {
526  buf = s_malloc(buflen);
527  if (buf == NULL) return NULL;
528  } else {
529  buflen = sizeof(staticbuf);
530  }
531 
532  /* Try with buffers two times bigger every time we fail to
533  * fit the string in the current buffer size. */
534  while(1) {
535  buf[buflen-2] = '\0';
536  va_copy(cpy,ap);
537  vsnprintf(buf, buflen, fmt, cpy);
538  va_end(cpy);
539  if (buf[buflen-2] != '\0') {
540  if (buf != staticbuf) s_free(buf);
541  buflen *= 2;
542  buf = s_malloc(buflen);
543  if (buf == NULL) return NULL;
544  continue;
545  }
546  break;
547  }
548 
549  /* Finally concat the obtained string to the SDS string and return it. */
550  t = sdscat(s, buf);
551  if (buf != staticbuf) s_free(buf);
552  return t;
553 }

580 /* This function is similar to sdscatprintf, but much faster as it does
581  * not rely on sprintf() family functions implemented by the libc that
582  * are often very slow. Moreover directly handling the sds string as
583  * new data is concatenated provides a performance improvement.
584  *
585  * However this function only handles an incompatible subset of printf-alike
586  * format specifiers:
587  *
588  * %s - C String
589  * %S - SDS string
590  * %i - signed int
591  * %I - 64 bit signed integer (long long, int64_t)
592  * %u - unsigned int
593  * %U - 64 bit unsigned integer (unsigned long long, uint64_t)
594  * %% - Verbatim "%" character.
595  */
596 sds sdscatfmt(sds s, char const *fmt, ...) {
597  size_t initlen = sdslen(s);
598  const char *f = fmt;
599  long i;
600  va_list ap;
601 
602  va_start(ap,fmt);
603  f = fmt; /* Next format specifier byte to process. */
604  i = initlen; /* Position of the next byte to write to dest str. */
605  while(*f) {
606  char next, *str;
607  size_t l;
608  long long num;
609  unsigned long long unum;
610 
611  /* Make sure there is always space for at least 1 char. */
612  if (sdsavail(s)==0) {
613  s = sdsMakeRoomFor(s,1);
614  }
615 
616  switch(*f) {
617  case '%':
618  next = *(f+1);
619  f++;
620  switch(next) {
621  case 's':
622  case 'S':
623  str = va_arg(ap,char*);
624  l = (next == 's') ? strlen(str) : sdslen(str);
625  if (sdsavail(s) < l) {
626  s = sdsMakeRoomFor(s,l);
627  }
628  memcpy(s+i,str,l);
629  sdsinclen(s,l);
630  i += l;
631  break;
632  case 'i':
633  case 'I':
634  if (next == 'i')
635  num = va_arg(ap,int);
636  else
637  num = va_arg(ap,long long);
638  {
639  char buf[SDS_LLSTR_SIZE];
640  l = sdsll2str(buf,num);
641  if (sdsavail(s) < l) {
642  s = sdsMakeRoomFor(s,l);
643  }
644  memcpy(s+i,buf,l);
645  sdsinclen(s,l);
646  i += l;
647  }
648  break;
649  case 'u':
650  case 'U':
651  if (next == 'u')
652  unum = va_arg(ap,unsigned int);
653  else
654  unum = va_arg(ap,unsigned long long);
655  {
656  char buf[SDS_LLSTR_SIZE];
657  l = sdsull2str(buf,unum);
658  if (sdsavail(s) < l) {
659  s = sdsMakeRoomFor(s,l);
660  }
661  memcpy(s+i,buf,l);
662  sdsinclen(s,l);
663  i += l;
664  }
665  break;
666  default: /* Handle %% and generally %<unknown>. */
667  s[i++] = next;
668  sdsinclen(s,1);
669  break;
670  }
671  break;
672  default:
673  s[i++] = *f;
674  sdsinclen(s,1);
675  break;
676  }
677  f++;
678  }
679  va_end(ap);
680 
681  /* Add null-term */
682  s[i] = '\0';
683  return s;
684 }

686 /* Remove the part of the string from left and from right composed just of
687  * contiguous characters found in 'cset', that is a null terminted C string.
688  *
689  * After the call, the modified sds string is no longer valid and all the
690  * references must be substituted with the new pointer returned by the call.
691  *
692  * Example:
693  *
694  * s = sdsnew("AA...AA.a.aa.aHelloWorld :::");
695  * s = sdstrim(s,"Aa. :");
696  * printf("%s\n", s);
697  *
698  * Output will be just "Hello World".
699  */
700 sds sdstrim(sds s, const char *cset) {
701  char *start, *end, *sp, *ep;
702  size_t len;
703 
704  sp = start = s;
705  ep = end = s+sdslen(s)-1;
706  while(sp <= end && strchr(cset, *sp)) sp++;
707  while(ep > sp && strchr(cset, *ep)) ep--;
708  len = (sp > ep) ? 0 : ((ep-sp)+1);
709  if (s != sp) memmove(s, sp, len);
710  s[len] = '\0';
711  sdssetlen(s,len);
712  return s;
713 }

715 /* Turn the string into a smaller (or equal) string containing only the
716  * substring specified by the 'start' and 'end' indexes.
717  *
718  * start and end can be negative, where -1 means the last character of the
719  * string, -2 the penultimate character, and so forth.
720  *
721  * The interval is inclusive, so the start and end characters will be part
722  * of the resulting string.
723  *
724  * The string is modified in-place.
725  *
726  * Example:
727  *
728  * s = sdsnew("Hello World");
729  * sdsrange(s,1,-1); => "ello World"
730  */
731 void sdsrange(sds s, ssize_t start, ssize_t end) {
732  size_t newlen, len = sdslen(s);
733 
734  if (len == 0) return;
735  if (start < 0) {
736  start = len+start;
737  if (start < 0) start = 0;
738  }
739  if (end < 0) {
740  end = len+end;
741  if (end < 0) end = 0;
742  }
743  newlen = (start > end) ? 0 : (end-start)+1;
744  if (newlen != 0) {
745  if (start >= (ssize_t)len) {
746  newlen = 0;
747  } else if (end >= (ssize_t)len) {
748  end = len-1;
749  newlen = (start > end) ? 0 : (end-start)+1;
750  }
751  } else {
752  start = 0;
753  }
754  if (start && newlen) memmove(s, s+start, newlen);
755  s[newlen] = 0;
756  sdssetlen(s,newlen);
757 }

773 /* Compare two sds strings s1 and s2 with memcmp().
774  *
775  * Return value:
776  *
777  * positive if s1 > s2.
778  * negative if s1 < s2.
779  * 0 if s1 and s2 are exactly the same binary string.
780  *
781  * If two strings share exactly the same prefix, but one of the two has
782  * additional characters, the longer string is considered to be greater than
783  * the smaller one. */
784 int sdscmp(const sds s1, const sds s2) {
785  size_t l1, l2, minlen;
786  int cmp;
787 
788  l1 = sdslen(s1);
789  l2 = sdslen(s2);
790  minlen = (l1 < l2) ? l1 : l2;
791  cmp = memcmp(s1,s2,minlen);
792  if (cmp == 0) return l1>l2? 1: (l1<l2? -1: 0);
793  return cmp;
794 }

796 /* Split 's' with separator in 'sep'. An array
797  * of sds strings is returned. *count will be set
798  * by reference to the number of tokens returned.
799  *
800  * On out of memory, zero length string, zero length
801  * separator, NULL is returned.
802  *
803  * Note that 'sep' is able to split a string using
804  * a multi-character separator. For example
805  * sdssplit("foo_-_bar","_-_"); will return two
806  * elements "foo" and "bar".
807  *
808  * This version of the function is binary-safe but
809  * requires length arguments. sdssplit() is just the
810  * same function but for zero-terminated strings.
811  */
812 sds *sdssplitlen(const char *s, ssize_t len, const char *sep, int seplen, int *count) {
813  int elements = 0, slots = 5;
814  long start = 0, j;
815  sds *tokens;
816 
817  if (seplen < 1 || len < 0) return NULL;
818 
819  tokens = s_malloc(sizeof(sds)*slots);
820  if (tokens == NULL) return NULL;
821 
822  if (len == 0) {
823  *count = 0;
824  return tokens;
825  }
826  for (j = 0; j < (len-(seplen-1)); j++) {
827  /* make sure there is room for the next element and the final one */
828  if (slots < elements+2) {
829  sds *newtokens;
830 
831  slots *= 2;
832  newtokens = s_realloc(tokens,sizeof(sds)*slots);
833  if (newtokens == NULL) goto cleanup;
834  tokens = newtokens;
835  }
836  /* search the separator */
837  if ((seplen == 1 && *(s+j) == sep[0]) || (memcmp(s+j,sep,seplen) == 0)) {
838  tokens[elements] = sdsnewlen(s+start,j-start);
839  if (tokens[elements] == NULL) goto cleanup;
840  elements++;
841  start = j+seplen;
842  j = j+seplen-1; /* skip the separator */
843  }
844  }
845  /* Add the final element. We are sure there is room in the tokens array. */
846  tokens[elements] = sdsnewlen(s+start,len-start);
847  if (tokens[elements] == NULL) goto cleanup;
848  elements++;
849  *count = elements;
850  return tokens;
851 
852 cleanup:
853  {
854  int i;
855  for (i = 0; i < elements; i++) sdsfree(tokens[i]);
856  s_free(tokens);
857  *count = 0;
858  return NULL;
859  }
860 }

864  if (!tokens) return;

872  * escapes in the form "\n\r\a...." or "\x<hex-number>".
873  *
874  * After the call, the modified sds string is no longer valid and all the

932 /* Split a line into arguments, where every argument can be in the
933  * following programming-language REPL-alike form:
934  *
935  * foo bar "newline are supported\n" and "\xff\x00otherstuff"
936  *
937  * The number of arguments is stored into *argc, and an array
938  * of sds is returned.
939  *
940  * The caller should free the resulting array of sds strings with
941  * sdsfreesplitres().
942  *
943  * Note that sdscatrepr() is able to convert back a string into
944  * a quoted string in the same format sdssplitargs() is able to parse.
945  *
946  * The function returns the allocated tokens on success, even when the
947  * input string is empty, or NULL if the input contains unbalanced
948  * quotes or closed quotes followed by non space characters
949  * as in: "foo"bar or "foo'
950  */
951 sds *sdssplitargs(const char *line, int *argc) {
952  const char *p = line;
953  char *current = NULL;
954  char **vector = NULL;
955 
956  *argc = 0;
957  while(1) {
958  /* skip blanks */
959  while(*p && isspace(*p)) p++;
960  if (*p) {
961  /* get a token */
962  int inq=0; /* set to 1 if we are in "quotes" */
963  int insq=0; /* set to 1 if we are in 'single quotes' */
964  int done=0;
965 
966  if (current == NULL) current = sdsempty();
967  while(!done) {
968  if (inq) {
969  if (*p == '\\' && *(p+1) == 'x' &&
970  is_hex_digit(*(p+2)) &&
971  is_hex_digit(*(p+3)))
972  {
973  unsigned char byte;
974 
975  byte = (hex_digit_to_int(*(p+2))*16)+
976  hex_digit_to_int(*(p+3));
977  current = sdscatlen(current,(char*)&byte,1);
978  p += 3;
979  } else if (*p == '\\' && *(p+1)) {
980  char c;
981 
982  p++;
983  switch(*p) {
984  case 'n': c = '\n'; break;
985  case 'r': c = '\r'; break;
986  case 't': c = '\t'; break;
987  case 'b': c = '\b'; break;
988  case 'a': c = '\a'; break;
989  default: c = *p; break;
990  }
991  current = sdscatlen(current,&c,1);
992  } else if (*p == '"') {
993  /* closing quote must be followed by a space or
994  * nothing at all. */
995  if (*(p+1) && !isspace(*(p+1))) goto err;
996  done=1;
997  } else if (!*p) {
998  /* unterminated quotes */
999  goto err;
1000  } else {
1001  current = sdscatlen(current,p,1);
1002  }
1003  } else if (insq) {
1004  if (*p == '\\' && *(p+1) == '\'') {
1005  p++;
1006  current = sdscatlen(current,"'",1);
1007  } else if (*p == '\'') {
1008  /* closing quote must be followed by a space or
1009  * nothing at all. */
1010  if (*(p+1) && !isspace(*(p+1))) goto err;
1011  done=1;
1012  } else if (!*p) {
1013  /* unterminated quotes */
1014  goto err;
1015  } else {
1016  current = sdscatlen(current,p,1);
1017  }
1018  } else {
1019  switch(*p) {
1020  case ' ':
1021  case '\n':
1022  case '\r':
1023  case '\t':
1024  case '\0':
1025  done=1;
1026  break;
1027  case '"':
1028  inq=1;
1029  break;
1030  case '\'':
1031  insq=1;
1032  break;
1033  default:
1034  current = sdscatlen(current,p,1);
1035  break;
1036  }
1037  }
1038  if (*p) p++;
1039  }
1040  /* add the token to the vector */
1041  vector = s_realloc(vector,((*argc)+1)*sizeof(char*));
1042  vector[*argc] = current;
1043  (*argc)++;
1044  current = NULL;
1045  } else {
1046  /* Even on empty input string return something not NULL. */
1047  if (vector == NULL) vector = s_malloc(sizeof(void*));
1048  return vector;
1049  }
1050  }
1051 
1052 err:
1053  while((*argc)--)
1054  sdsfree(vector[*argc]);
1055  s_free(vector);
1056  if (current) sdsfree(current);
1057  *argc = 0;
1058  return NULL;
1059 }
