173 uint64_t crc64(uint64_t crc, const unsigned char *s, uint64_t l) {
174  uint64_t j;
175 
176  for (j = 0; j < l; j++) {
177  uint8_t byte = s[j];
178  crc = crc64_tab[(uint8_t)crc ^ byte] ^ (crc >> 8);
179  }
180  return crc;
181 }
