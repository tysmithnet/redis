35 /* ZSETs are ordered sets using two data structures to hold the same elements
36  * in order to get O(log(N)) INSERT and REMOVE operations into a sorted
37  * data structure.
38  *
39  * The elements are added to a hash table mapping Redis objects to scores.
40  * At the same time the elements are added to a skip list mapping scores
41  * to Redis objects (so objects are sorted by scores in this "view").
42  *
43  * Note that the SDS string representing the element is the same in both
44  * the hash table and skiplist in order to save memory. What we do in order
45  * to manage the shared SDS string more easily is to free the SDS string
46  * only in zslFreeNode(). The dictionary has no value free method set.
47  * So we should always remove an element from the dictionary, and later from
48  * the skiplist.
49  *
50  * This skiplist implementation is almost a C translation of the original
51  * algorithm described by William Pugh in "Skip Lists: A Probabilistic
52  * Alternative to Balanced Trees", modified in three ways:
53  * a) this implementation allows for repeated scores.
54  * b) the comparison is not just by key (our 'score') but by satellite data.
55  * c) there is a back pointer, so it's a doubly linked list with the back
56  * pointers being only at "level 1". This allows to traverse the list
57  * from tail to head, useful for ZREVRANGE. */

118 /* Returns a random level for the new skiplist node we are going to create.
119  * The return value of this function is between 1 and ZSKIPLIST_MAXLEVEL
120  * (both inclusive), with a powerlaw-alike distribution where higher
121  * levels are less likely to be returned. */
122 int zslRandomLevel(void) {
123  int level = 1;

129 /* Insert a new node in the skiplist. Assumes the element does not already
130  * exist (up to the caller to enforce that). The skiplist takes ownership
131  * of the passed SDS string 'ele'. */
132 zskiplistNode *zslInsert(zskiplist *zsl, double score, sds ele) {
133  zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
134  unsigned int rank[ZSKIPLIST_MAXLEVEL];
135  int i, level;
136 
137  serverAssert(!isnan(score));
138  x = zsl->header;
139  for (i = zsl->level-1; i >= 0; i--) {
140  /* store rank that is crossed to reach the insert position */
141  rank[i] = i == (zsl->level-1) ? 0 : rank[i+1];
142  while (x->level[i].forward &&
143  (x->level[i].forward->score < score ||
144  (x->level[i].forward->score == score &&
145  sdscmp(x->level[i].forward->ele,ele) < 0)))
146  {
147  rank[i] += x->level[i].span;
148  x = x->level[i].forward;
149  }
150  update[i] = x;
151  }
152  /* we assume the element is not already inside, since we allow duplicated
153  * scores, reinserting the same element should never happen since the
154  * caller of zslInsert() should test in the hash table if the element is
155  * already inside or not. */
156  level = zslRandomLevel();
157  if (level > zsl->level) {
158  for (i = zsl->level; i < level; i++) {
159  rank[i] = 0;
160  update[i] = zsl->header;
161  update[i]->level[i].span = zsl->length;
162  }
163  zsl->level = level;
164  }
165  x = zslCreateNode(level,score,ele);
166  for (i = 0; i < level; i++) {
167  x->level[i].forward = update[i]->level[i].forward;
168  update[i]->level[i].forward = x;
169 
170  /* update span covered by update[i] as x is inserted here */
171  x->level[i].span = update[i]->level[i].span - (rank[0] - rank[i]);
172  update[i]->level[i].span = (rank[0] - rank[i]) + 1;
173  }
174 
175  /* increment span for untouched levels */
176  for (i = level; i < zsl->level; i++) {
177  update[i]->level[i].span++;
178  }
179 
180  x->backward = (update[0] == zsl->header) ? NULL : update[0];
181  if (x->level[0].forward)
182  x->level[0].forward->backward = x;
183  else
184  zsl->tail = x;
185  zsl->length++;
186  return x;
187 }

189 /* Internal function used by zslDelete, zslDeleteByScore and zslDeleteByRank */
190 void zslDeleteNode(zskiplist *zsl, zskiplistNode *x, zskiplistNode **update) {
191  int i;
192  for (i = 0; i < zsl->level; i++) {
193  if (update[i]->level[i].forward == x) {
194  update[i]->level[i].span += x->level[i].span - 1;
195  update[i]->level[i].forward = x->level[i].forward;
196  } else {
197  update[i]->level[i].span -= 1;
198  }
199  }
200  if (x->level[0].forward) {
201  x->level[0].forward->backward = x->backward;
202  } else {
203  zsl->tail = x->backward;
204  }
205  while(zsl->level > 1 && zsl->header->level[zsl->level-1].forward == NULL)
206  zsl->level--;
207  zsl->length--;
208 }

210 /* Delete an element with matching score/element from the skiplist.
211  * The function returns 1 if the node was found and deleted, otherwise
212  * 0 is returned.
213  *
214  * If 'node' is NULL the deleted node is freed by zslFreeNode(), otherwise
215  * it is not freed (but just unlinked) and *node is set to the node pointer,
216  * so that it is possible for the caller to reuse the node (including the
217  * referenced SDS string at node->ele). */
218 int zslDelete(zskiplist *zsl, double score, sds ele, zskiplistNode **node) {
219  zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
220  int i;
221 
222  x = zsl->header;
223  for (i = zsl->level-1; i >= 0; i--) {
224  while (x->level[i].forward &&
225  (x->level[i].forward->score < score ||
226  (x->level[i].forward->score == score &&
227  sdscmp(x->level[i].forward->ele,ele) < 0)))
228  {
229  x = x->level[i].forward;
230  }
231  update[i] = x;
232  }
233  /* We may have multiple elements with the same score, what we need
234  * is to find the element with both the right score and object. */
235  x = x->level[0].forward;
236  if (x && score == x->score && sdscmp(x->ele,ele) == 0) {
237  zslDeleteNode(zsl, x, update);
238  if (!node)
239  zslFreeNode(x);
240  else
241  *node = x;
242  return 1;
243  }
244  return 0; /* not found */
245 }

247 /* Update the score of an elmenent inside the sorted set skiplist.
248  * Note that the element must exist and must match 'score'.
249  * This function does not update the score in the hash table side, the
250  * caller should take care of it.
251  *
252  * Note that this function attempts to just update the node, in case after
253  * the score update, the node would be exactly at the same position.
254  * Otherwise the skiplist is modified by removing and re-adding a new
255  * element, which is more costly.
256  *
257  * The function returns the updated element skiplist node pointer. */
258 zskiplistNode *zslUpdateScore(zskiplist *zsl, double curscore, sds ele, double newscore) {
259  zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
260  int i;
261 
262  /* We need to seek to element to update to start: this is useful anyway,
263  * we'll have to update or remove it. */
264  x = zsl->header;
265  for (i = zsl->level-1; i >= 0; i--) {
266  while (x->level[i].forward &&
267  (x->level[i].forward->score < curscore ||
268  (x->level[i].forward->score == curscore &&
269  sdscmp(x->level[i].forward->ele,ele) < 0)))
270  {
271  x = x->level[i].forward;
272  }
273  update[i] = x;
274  }
275 
276  /* Jump to our element: note that this function assumes that the
277  * element with the matching score exists. */
278  x = x->level[0].forward;
279  serverAssert(x && curscore == x->score && sdscmp(x->ele,ele) == 0);
280 
281  /* If the node, after the score update, would be still exactly
282  * at the same position, we can just update the score without
283  * actually removing and re-inserting the element in the skiplist. */
284  if ((x->backward == NULL || x->backward->score < newscore) &&
285  (x->level[0].forward == NULL || x->level[0].forward->score > newscore))
286  {
287  x->score = newscore;
288  return x;
289  }
290 
291  /* No way to reuse the old node: we need to remove and insert a new
292  * one at a different place. */
293  zslDeleteNode(zsl, x, update);
294  zskiplistNode *newnode = zslInsert(zsl,newscore,x->ele);
295  /* We reused the old node x->ele SDS string, free the node now
296  * since zslInsert created a new one. */
297  x->ele = NULL;
298  zslFreeNode(x);
299  return newnode;
300 }

310 /* Returns if there is a part of the zset is in range. */
311 int zslIsInRange(zskiplist *zsl, zrangespec *range) {
312  zskiplistNode *x;
313 
314  /* Test for ranges that will always be empty. */
315  if (range->min > range->max ||
316  (range->min == range->max && (range->minex || range->maxex)))
317  return 0;
318  x = zsl->tail;
319  if (x == NULL || !zslValueGteMin(x->score,range))
320  return 0;
321  x = zsl->header->level[0].forward;
322  if (x == NULL || !zslValueLteMax(x->score,range))
323  return 0;
324  return 1;
325 }

327 /* Find the first node that is contained in the specified range.
328  * Returns NULL when no element is contained in the range. */
329 zskiplistNode *zslFirstInRange(zskiplist *zsl, zrangespec *range) {
330  zskiplistNode *x;
331  int i;
332 
333  /* If everything is out of range, return early. */
334  if (!zslIsInRange(zsl,range)) return NULL;
335 
336  x = zsl->header;
337  for (i = zsl->level-1; i >= 0; i--) {
338  /* Go forward while *OUT* of range. */
339  while (x->level[i].forward &&
340  !zslValueGteMin(x->level[i].forward->score,range))
341  x = x->level[i].forward;
342  }
343 
344  /* This is an inner range, so the next node cannot be NULL. */
345  x = x->level[0].forward;
346  serverAssert(x != NULL);
347 
348  /* Check if score <= max. */
349  if (!zslValueLteMax(x->score,range)) return NULL;
350  return x;
351 }

378 /* Delete all the elements with score between min and max from the skiplist.
379  * Min and max are inclusive, so a score >= min || score <= max is deleted.
380  * Note that this function takes the reference to the hash table view of the
381  * sorted set, in order to remove the elements from the hash table too. */
382 unsigned long zslDeleteRangeByScore(zskiplist *zsl, zrangespec *range, dict *dict) {
383  zskiplistNode *update[ZSKIPLIST_MAXLEVEL], *x;
384  unsigned long removed = 0;
385  int i;
386 
387  x = zsl->header;
388  for (i = zsl->level-1; i >= 0; i--) {
389  while (x->level[i].forward && (range->minex ?
390  x->level[i].forward->score <= range->min :
391  x->level[i].forward->score < range->min))
392  x = x->level[i].forward;
393  update[i] = x;
394  }
395 
396  /* Current node is the last with score < or <= min. */
397  x = x->level[0].forward;
398 
399  /* Delete nodes while in range. */
400  while (x &&
401  (range->maxex ? x->score < range->max : x->score <= range->max))
402  {
403  zskiplistNode *next = x->level[0].forward;
404  zslDeleteNode(zsl,x,update);
405  dictDelete(dict,x->ele);
406  zslFreeNode(x); /* Here is where x->ele is actually released. */
407  removed++;
408  x = next;
409  }
410  return removed;
411 }

519 /* Populate the rangespec according to the objects min and max. */
520 static int zslParseRange(robj *min, robj *max, zrangespec *spec) {
521  char *eptr;
522  spec->minex = spec->maxex = 0;
523 
524  /* Parse the min-max interval. If one of the values is prefixed
525  * by the "(" character, it's considered "open". For instance
526  * ZRANGEBYSCORE zset (1.5 (2.5 will match min < x < max
527  * ZRANGEBYSCORE zset 1.5 2.5 will instead match min <= x <= max */
528  if (min->encoding == OBJ_ENCODING_INT) {
529  spec->min = (long)min->ptr;
530  } else {
531  if (((char*)min->ptr)[0] == '(') {
532  spec->min = strtod((char*)min->ptr+1,&eptr);
533  if (eptr[0] != '\0' || isnan(spec->min)) return C_ERR;
534  spec->minex = 1;
535  } else {
536  spec->min = strtod((char*)min->ptr,&eptr);
537  if (eptr[0] != '\0' || isnan(spec->min)) return C_ERR;
538  }
539  }
540  if (max->encoding == OBJ_ENCODING_INT) {
541  spec->max = (long)max->ptr;
542  } else {
543  if (((char*)max->ptr)[0] == '(') {
544  spec->max = strtod((char*)max->ptr+1,&eptr);
545  if (eptr[0] != '\0' || isnan(spec->max)) return C_ERR;
546  spec->maxex = 1;
547  } else {
548  spec->max = strtod((char*)max->ptr,&eptr);
549  if (eptr[0] != '\0' || isnan(spec->max)) return C_ERR;
550  }
551  }
552 
553  return C_OK;
554 }

558 /* Parse max or min argument of ZRANGEBYLEX.
559  * (foo means foo (open interval)
560  * [foo means foo (closed interval)
561  * - means the min string possible
562  * + means the max string possible
563  *
564  * If the string is valid the *dest pointer is set to the redis object
565  * that will be used for the comparison, and ex will be set to 0 or 1
566  * respectively if the item is exclusive or inclusive. C_OK will be
567  * returned.
568  *
569  * If the string is not a valid range C_ERR is returned, and the value
570  * of *dest and *ex is undefined. */
571 int zslParseLexRangeItem(robj *item, sds *dest, int *ex) {
572  char *c = item->ptr;
573 
574  switch(c[0]) {
575  case '+':
576  if (c[1] != '\0') return C_ERR;
577  *ex = 0;
578  *dest = shared.maxstring;
579  return C_OK;
580  case '-':
581  if (c[1] != '\0') return C_ERR;
582  *ex = 0;
583  *dest = shared.minstring;
584  return C_OK;
585  case '(':
586  *ex = 1;
587  *dest = sdsnewlen(c+1,sdslen(c)-1);
588  return C_OK;
589  case '[':
590  *ex = 0;
591  *dest = sdsnewlen(c+1,sdslen(c)-1);
592  return C_OK;
593  default:
594  return C_ERR;
595  }
596 }

1273 /* Add a new element or update the score of an existing element in a sorted
1274  * set, regardless of its encoding.
1275  *
1276  * The set of flags change the command behavior. They are passed with an integer
1277  * pointer since the function will clear the flags and populate them with
1278  * other flags to indicate different conditions.
1279  *
1280  * The input flags are the following:
1281  *
1282  * ZADD_INCR: Increment the current element score by 'score' instead of updating
1283  * the current element score. If the element does not exist, we
1284  * assume 0 as previous score.
1285  * ZADD_NX: Perform the operation only if the element does not exist.
1286  * ZADD_XX: Perform the operation only if the element already exist.
1287  *
1288  * When ZADD_INCR is used, the new score of the element is stored in
1289  * '*newscore' if 'newscore' is not NULL.
1290  *
1291  * The returned flags are the following:
1292  *
1293  * ZADD_NAN: The resulting score is not a number.
1294  * ZADD_ADDED: The element was added (not present before the call).
1295  * ZADD_UPDATED: The element score was updated.
1296  * ZADD_NOP: No operation was performed because of NX or XX.
1297  *
1298  * Return value:
1299  *
1300  * The function returns 1 on success, and sets the appropriate flags
1301  * ADDED or UPDATED to signal what happened during the operation (note that
1302  * none could be set if we re-added an element using the same score it used
1303  * to have, or in the case a zero increment is used).
1304  *
1305  * The function returns 0 on erorr, currently only when the increment
1306  * produces a NAN condition, or when the 'score' value is NAN since the
1307  * start.
1308  *
1309  * The commad as a side effect of adding a new element may convert the sorted
1310  * set internal encoding from ziplist to hashtable+skiplist.
1311  *
1312  * Memory managemnet of 'ele':
1313  *
1314  * The function does not take ownership of the 'ele' SDS string, but copies
1315  * it if needed. */
1316 int zsetAdd(robj *zobj, double score, sds ele, int *flags, double *newscore) {
1317  /* Turn options into simple to check vars. */
1318  int incr = (*flags & ZADD_INCR) != 0;
1319  int nx = (*flags & ZADD_NX) != 0;
1320  int xx = (*flags & ZADD_XX) != 0;
1321  *flags = 0; /* We'll return our response flags. */
1322  double curscore;
1323 
1324  /* NaN as input is an error regardless of all the other parameters. */
1325  if (isnan(score)) {
1326  *flags = ZADD_NAN;
1327  return 0;
1328  }
1329 
1330  /* Update the sorted set according to its encoding. */
1331  if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1332  unsigned char *eptr;
1333 
1334  if ((eptr = zzlFind(zobj->ptr,ele,&curscore)) != NULL) {
1335  /* NX? Return, same element already exists. */
1336  if (nx) {
1337  *flags |= ZADD_NOP;
1338  return 1;
1339  }
1340 
1341  /* Prepare the score for the increment if needed. */
1342  if (incr) {
1343  score += curscore;
1344  if (isnan(score)) {
1345  *flags |= ZADD_NAN;
1346  return 0;
1347  }
1348  if (newscore) *newscore = score;
1349  }
1350 
1351  /* Remove and re-insert when score changed. */
1352  if (score != curscore) {
1353  zobj->ptr = zzlDelete(zobj->ptr,eptr);
1354  zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1355  *flags |= ZADD_UPDATED;
1356  }
1357  return 1;
1358  } else if (!xx) {
1359  /* Optimize: check if the element is too large or the list
1360  * becomes too long *before* executing zzlInsert. */
1361  zobj->ptr = zzlInsert(zobj->ptr,ele,score);
1362  if (zzlLength(zobj->ptr) > server.zset_max_ziplist_entries)
1363  zsetConvert(zobj,OBJ_ENCODING_SKIPLIST);
1364  if (sdslen(ele) > server.zset_max_ziplist_value)
1365  zsetConvert(zobj,OBJ_ENCODING_SKIPLIST);
1366  if (newscore) *newscore = score;
1367  *flags |= ZADD_ADDED;
1368  return 1;
1369  } else {
1370  *flags |= ZADD_NOP;
1371  return 1;
1372  }
1373  } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1374  zset *zs = zobj->ptr;
1375  zskiplistNode *znode;
1376  dictEntry *de;
1377 
1378  de = dictFind(zs->dict,ele);
1379  if (de != NULL) {
1380  /* NX? Return, same element already exists. */
1381  if (nx) {
1382  *flags |= ZADD_NOP;
1383  return 1;
1384  }
1385  curscore = *(double*)dictGetVal(de);
1386 
1387  /* Prepare the score for the increment if needed. */
1388  if (incr) {
1389  score += curscore;
1390  if (isnan(score)) {
1391  *flags |= ZADD_NAN;
1392  return 0;
1393  }
1394  if (newscore) *newscore = score;
1395  }
1396 
1397  /* Remove and re-insert when score changes. */
1398  if (score != curscore) {
1399  znode = zslUpdateScore(zs->zsl,curscore,ele,score);
1400  /* Note that we did not removed the original element from
1401  * the hash table representing the sorted set, so we just
1402  * update the score. */
1403  dictGetVal(de) = &znode->score; /* Update score ptr. */
1404  *flags |= ZADD_UPDATED;
1405  }
1406  return 1;
1407  } else if (!xx) {
1408  ele = sdsdup(ele);
1409  znode = zslInsert(zs->zsl,score,ele);
1410  serverAssert(dictAdd(zs->dict,ele,&znode->score) == DICT_OK);
1411  *flags |= ZADD_ADDED;
1412  if (newscore) *newscore = score;
1413  return 1;
1414  } else {
1415  *flags |= ZADD_NOP;
1416  return 1;
1417  }
1418  } else {
1419  serverPanic("Unknown sorted set encoding");
1420  }
1421  return 0; /* Never reached. */
1422 }

1424 /* Delete the element 'ele' from the sorted set, returning 1 if the element
1425  * existed and was deleted, 0 otherwise (the element was not there). */
1426 int zsetDel(robj *zobj, sds ele) {
1427  if (zobj->encoding == OBJ_ENCODING_ZIPLIST) {
1428  unsigned char *eptr;
1429 
1430  if ((eptr = zzlFind(zobj->ptr,ele,NULL)) != NULL) {
1431  zobj->ptr = zzlDelete(zobj->ptr,eptr);
1432  return 1;
1433  }
1434  } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1435  zset *zs = zobj->ptr;
1436  dictEntry *de;
1437  double score;
1438 
1439  de = dictUnlink(zs->dict,ele);
1440  if (de != NULL) {
1441  /* Get the score in order to delete from the skiplist later. */
1442  score = *(double*)dictGetVal(de);
1443 
1444  /* Delete from the hash table and later from the skiplist.
1445  * Note that the order is important: deleting from the skiplist
1446  * actually releases the SDS string representing the element,
1447  * which is shared between the skiplist and the hash table, so
1448  * we need to delete from the skiplist as the final step. */
1449  dictFreeUnlinkedEntry(zs->dict,de);

1453  serverAssert(retval);

1456  return 1;
1457  }
1458  } else {
1459  serverPanic("Unknown sorted set encoding");
1460  }

1462 }

1464 /* Given a sorted set object returns the 0-based rank of the object or
1465  * -1 if the object does not exist.
1466  *
1467  * For rank we mean the position of the element in the sorted collection
1468  * of elements. So the first element has rank 0, the second rank 1, and so
1469  * forth up to length-1 elements.
1470  *
1471  * If 'reverse' is false, the rank is returned considering as first element
1472  * the one with the lowest score. Otherwise if 'reverse' is non-zero
1473  * the rank is computed considering as element with rank 0 the one with
1474  * the highest score. */
1475 long zsetRank(robj *zobj, sds ele, int reverse) {

1477  unsigned long rank;
1478 
1479  llen = zsetLength(zobj);
1480

1482  unsigned char *zl = zobj->ptr;
1483  unsigned char *eptr, *sptr;
1484 
1485  eptr = ziplistIndex(zl,0);
1486  serverAssert(eptr != NULL);
1487  sptr = ziplistNext(zl,eptr);
1488  serverAssert(sptr != NULL);
1489 
1490  rank = 1;
1491  while(eptr != NULL) {
1492  if (ziplistCompare(eptr,(unsigned char*)ele,sdslen(ele)))
1493  break;
1494  rank++;
1495  zzlNext(zl,&eptr,&sptr);
1496  }
1497 
1498  if (eptr != NULL) {
1499  if (reverse)
1500  return llen-rank;
1501  else
1502  return rank-1;
1503  } else {
1504  return -1;
1505  }
1506  } else if (zobj->encoding == OBJ_ENCODING_SKIPLIST) {
1507  zset *zs = zobj->ptr;
1508  zskiplist *zsl = zs->zsl;
1509  dictEntry *de;
1510  double score;
1511 
1512  de = dictFind(zs->dict,ele);
1513  if (de != NULL) {
1514  score = *(double*)dictGetVal(de);
1515  rank = zslGetRank(zsl,score,ele);
1516  /* Existing elements always have a rank. */
1517  serverAssert(rank != 0);
1518  if (reverse)
1519  return llen-rank;
1520  else
1521  return rank-1;
1522  } else {
1523  return -1;
1524  }
1525  } else {
1526  serverPanic("Unknown sorted set encoding");
1527  }
1528 }

1534 /* This generic command implements both ZADD and ZINCRBY. */
1535 void zaddGenericCommand(client *c, int flags) {
1536  static char *nanerr = "resulting score is not a number (NaN)";
1537  robj *key = c->argv[1];
1538  robj *zobj;
1539  sds ele;
1540  double score = 0, *scores = NULL;
1541  int j, elements;
1542  int scoreidx = 0;
1543  /* The following vars are used in order to track what the command actually
1544  * did during the execution, to reply to the client and to trigger the
1545  * notification of keyspace change. */
1546  int added = 0; /* Number of new elements added. */
1547  int updated = 0; /* Number of elements with updated score. */
1548  int processed = 0; /* Number of elements processed, may remain zero with
1549  options like XX. */
1550 
1551  /* Parse options. At the end 'scoreidx' is set to the argument position
1552  * of the score of the first score-element pair. */
1553  scoreidx = 2;
1554  while(scoreidx < c->argc) {
1555  char *opt = c->argv[scoreidx]->ptr;
1556  if (!strcasecmp(opt,"nx")) flags |= ZADD_NX;
1557  else if (!strcasecmp(opt,"xx")) flags |= ZADD_XX;
1558  else if (!strcasecmp(opt,"ch")) flags |= ZADD_CH;
1559  else if (!strcasecmp(opt,"incr")) flags |= ZADD_INCR;
1560  else break;
1561  scoreidx++;
1562  }
1563 
1564  /* Turn options into simple to check vars. */
1565  int incr = (flags & ZADD_INCR) != 0;
1566  int nx = (flags & ZADD_NX) != 0;
1567  int xx = (flags & ZADD_XX) != 0;
1568  int ch = (flags & ZADD_CH) != 0;
1569 
1570  /* After the options, we expect to have an even number of args, since
1571  * we expect any number of score-element pairs. */
1572  elements = c->argc-scoreidx;
1573  if (elements % 2 || !elements) {
1574  addReply(c,shared.syntaxerr);
1575  return;
1576  }
1577  elements /= 2; /* Now this holds the number of score-element pairs. */
1578 
1579  /* Check for incompatible options. */
1580  if (nx && xx) {
1581  addReplyError(c,
1582  "XX and NX options at the same time are not compatible");
1583  return;
1584  }
1585 
1586  if (incr && elements > 1) {
1587  addReplyError(c,
1588  "INCR option supports a single increment-element pair");
1589  return;
1590  }
1591 
1592  /* Start parsing all the scores, we need to emit any syntax error
1593  * before executing additions to the sorted set, as the command should
1594  * either execute fully or nothing at all. */
1595  scores = zmalloc(sizeof(double)*elements);
1596  for (j = 0; j < elements; j++) {
1597  if (getDoubleFromObjectOrReply(c,c->argv[scoreidx+j*2],&scores[j],NULL)
1598  != C_OK) goto cleanup;
1599  }
1600 
1601  /* Lookup the key and create the sorted set if does not exist. */
1602  zobj = lookupKeyWrite(c->db,key);
1603  if (zobj == NULL) {
1604  if (xx) goto reply_to_client; /* No key + XX option: nothing to do. */
1605  if (server.zset_max_ziplist_entries == 0 ||
1606  server.zset_max_ziplist_value < sdslen(c->argv[scoreidx+1]->ptr))
1607  {
1608  zobj = createZsetObject();
1609  } else {
1610  zobj = createZsetZiplistObject();
1611  }
1612  dbAdd(c->db,key,zobj);
1613  } else {
1614  if (zobj->type != OBJ_ZSET) {
1615  addReply(c,shared.wrongtypeerr);
1616  goto cleanup;
1617  }
1618  }
1619 
1620  for (j = 0; j < elements; j++) {
1621  double newscore;
1622  score = scores[j];
1623  int retflags = flags;
1624 
1625  ele = c->argv[scoreidx+1+j*2]->ptr;
1626  int retval = zsetAdd(zobj, score, ele, &retflags, &newscore);
1627  if (retval == 0) {
1628  addReplyError(c,nanerr);
1629  goto cleanup;
1630  }
1631  if (retflags & ZADD_ADDED) added++;
1632  if (retflags & ZADD_UPDATED) updated++;
1633  if (!(retflags & ZADD_NOP)) processed++;
1634  score = newscore;
1635  }
1636  server.dirty += (added+updated);
1637 
1638 reply_to_client:
1639  if (incr) { /* ZINCRBY or INCR option. */
1640  if (processed)
1641  addReplyDouble(c,score);
1642  else
1643  addReply(c,shared.nullbulk);
1644  } else { /* ZADD. */
1645  addReplyLongLong(c,ch ? added+updated : added);
1646  }
1647 
1648 cleanup:
1649  zfree(scores);
1650  if (added || updated) {
1651  signalModifiedKey(c->db,key);
1652  notifyKeyspaceEvent(NOTIFY_ZSET,
1653  incr ? "zincr" : "zadd", key, c->db->id);
1654  }
1655 }
