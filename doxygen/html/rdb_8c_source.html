118 /* This function loads a time from the RDB file. It gets the version of the
119  * RDB because, unfortunately, before Redis 5 (RDB version 9), the function
120  * failed to convert data to/from little endian, so RDB files with keys having
121  * expires could not be shared between big endian and little endian systems
122  * (because the expire time will be totally wrong). The fix for this is just
123  * to call memrev64ifbe(), however if we fix this for all the RDB versions,
124  * this call will introduce an incompatibility for big endian systems:
125  * after upgrading to Redis version 5 they will no longer be able to load their
126  * own old RDB files. Because of that, we instead fix the function only for new
127  * RDB versions, and load older RDB versions as we used to do in the past,
128  * allowing big endian systems to load their own old RDB files. */
129 long long rdbLoadMillisecondTime(rio *rdb, int rdbver) {
130  int64_t t64;
131  rdbLoadRaw(rdb,&t64,8);
132  if (rdbver >= 9) /* Check the top comment of this function. */
133  memrev64ifbe(&t64); /* Convert in big endian if the system is BE. */
134  return (long long)t64;

137 /* Saves an encoded length. The first two bits in the first byte are used to
138  * hold the encoding type. See the RDB_* definitions for more information
139  * on the types of encoding. */
140 int rdbSaveLen(rio *rdb, uint64_t len) {
141  unsigned char buf[2];
142  size_t nwritten;
143 
144  if (len < (1<<6)) {
145  /* Save a 6 bit len */
146  buf[0] = (len&0xFF)|(RDB_6BITLEN<<6);
147  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
148  nwritten = 1;
149  } else if (len < (1<<14)) {
150  /* Save a 14 bit len */
151  buf[0] = ((len>>8)&0xFF)|(RDB_14BITLEN<<6);
152  buf[1] = len&0xFF;
153  if (rdbWriteRaw(rdb,buf,2) == -1) return -1;
154  nwritten = 2;
155  } else if (len <= UINT32_MAX) {
156  /* Save a 32 bit len */
157  buf[0] = RDB_32BITLEN;
158  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
159  uint32_t len32 = htonl(len);
160  if (rdbWriteRaw(rdb,&len32,4) == -1) return -1;
161  nwritten = 1+4;
162  } else {
163  /* Save a 64 bit len */
164  buf[0] = RDB_64BITLEN;
165  if (rdbWriteRaw(rdb,buf,1) == -1) return -1;
166  len = htonu64(len);
167  if (rdbWriteRaw(rdb,&len,8) == -1) return -1;
168  nwritten = 1+8;
169  }
170  return nwritten;
171 }

174 /* Load an encoded length. If the loaded length is a normal length as stored
175  * with rdbSaveLen(), the read length is set to '*lenptr'. If instead the
176  * loaded length describes a special encoding that follows, then '*isencoded'
177  * is set to 1 and the encoding format is stored at '*lenptr'.
178  *
179  * See the RDB_ENC_* definitions in rdb.h for more information on special
180  * encodings.
181  *
182  * The function returns -1 on error, 0 on success. */
183 int rdbLoadLenByRef(rio *rdb, int *isencoded, uint64_t *lenptr) {
184  unsigned char buf[2];
185  int type;
186 
187  if (isencoded) *isencoded = 0;
188  if (rioRead(rdb,buf,1) == 0) return -1;
189  type = (buf[0]&0xC0)>>6;
190  if (type == RDB_ENCVAL) {
191  /* Read a 6 bit encoding type. */
192  if (isencoded) *isencoded = 1;
193  *lenptr = buf[0]&0x3F;
194  } else if (type == RDB_6BITLEN) {
195  /* Read a 6 bit len. */
196  *lenptr = buf[0]&0x3F;
197  } else if (type == RDB_14BITLEN) {
198  /* Read a 14 bit len. */
199  if (rioRead(rdb,buf+1,1) == 0) return -1;
200  *lenptr = ((buf[0]&0x3F)<<8)|buf[1];
201  } else if (buf[0] == RDB_32BITLEN) {
202  /* Read a 32 bit len. */
203  uint32_t len;
204  if (rioRead(rdb,&len,4) == 0) return -1;
205  *lenptr = ntohl(len);
206  } else if (buf[0] == RDB_64BITLEN) {
207  /* Read a 64 bit len. */
208  uint64_t len;
209  if (rioRead(rdb,&len,8) == 0) return -1;
210  *lenptr = ntohu64(len);
211  } else {
212  rdbExitReportCorruptRDB(
213  "Unknown length encoding %d in rdbLoadLen()",type);
214  return -1; /* Never reached. */
215  }
216  return 0;
217 }

230 /* Encodes the "value" argument as integer when it fits in the supported ranges
231  * for encoded types. If the function successfully encodes the integer, the
232  * representation is stored in the buffer pointer to by "enc" and the string
233  * length is returned. Otherwise 0 is returned. */
234 int rdbEncodeInteger(long long value, unsigned char *enc) {
235  if (value >= -(1<<7) && value <= (1<<7)-1) {
236  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT8;
237  enc[1] = value&0xFF;
238  return 2;
239  } else if (value >= -(1<<15) && value <= (1<<15)-1) {
240  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT16;
241  enc[1] = value&0xFF;
242  enc[2] = (value>>8)&0xFF;
243  return 3;
244  } else if (value >= -((long long)1<<31) && value <= ((long long)1<<31)-1) {
245  enc[0] = (RDB_ENCVAL<<6)|RDB_ENC_INT32;
246  enc[1] = value&0xFF;
247  enc[2] = (value>>8)&0xFF;
248  enc[3] = (value>>16)&0xFF;
249  enc[4] = (value>>24)&0xFF;
250  return 5;
251  } else {
252  return 0;
253  }
254 }

256 /* Loads an integer-encoded object with the specified encoding type "enctype".
257  * The returned value changes according to the flags, see
258  * rdbGenerincLoadStringObject() for more info. */
259 void *rdbLoadIntegerObject(rio *rdb, int enctype, int flags, size_t *lenptr) {
260  int plain = flags & RDB_LOAD_PLAIN;
261  int sds = flags & RDB_LOAD_SDS;
262  int encode = flags & RDB_LOAD_ENC;
263  unsigned char enc[4];
264  long long val;
265 
266  if (enctype == RDB_ENC_INT8) {
267  if (rioRead(rdb,enc,1) == 0) return NULL;
268  val = (signed char)enc[0];
269  } else if (enctype == RDB_ENC_INT16) {
270  uint16_t v;
271  if (rioRead(rdb,enc,2) == 0) return NULL;
272  v = enc[0]|(enc[1]<<8);
273  val = (int16_t)v;
274  } else if (enctype == RDB_ENC_INT32) {
275  uint32_t v;
276  if (rioRead(rdb,enc,4) == 0) return NULL;
277  v = enc[0]|(enc[1]<<8)|(enc[2]<<16)|(enc[3]<<24);
278  val = (int32_t)v;
279  } else {
280  val = 0; /* anti-warning */
281  rdbExitReportCorruptRDB("Unknown RDB integer encoding type %d",enctype);
282  }
283  if (plain || sds) {
284  char buf[LONG_STR_SIZE], *p;
285  int len = ll2string(buf,sizeof(buf),val);
286  if (lenptr) *lenptr = len;
287  p = plain ? zmalloc(len) : sdsnewlen(SDS_NOINIT,len);
288  memcpy(p,buf,len);
289  return p;
290  } else if (encode) {
291  return createStringObjectFromLongLongForValue(val);
292  } else {
293  return createObject(OBJ_STRING,sdsfromlonglong(val));
294  }
295 }

297 /* String objects in the form "2391" "-100" without any space and with a
298  * range of values that can fit in an 8, 16 or 32 bit signed value can be
299  * encoded as integers to save space */
300 int rdbTryIntegerEncoding(char *s, size_t len, unsigned char *enc) {
301  long long value;
302  char *endptr, buf[32];
303 
304  /* Check if it's possible to encode this value as a number */
305  value = strtoll(s, &endptr, 10);
306  if (endptr[0] != '\0') return 0;
307  ll2string(buf,32,value);
308 
309  /* If the number converted back into a string is not identical
310  * then it's not possible to encode the string as integer */
311  if (strlen(buf) != len || memcmp(buf,s,len)) return 0;
312 
313  return rdbEncodeInteger(value,enc);
314 }

317  size_t original_len) {
318  unsigned char byte;
319  ssize_t n, nwritten = 0;
320 
321  /* Data compressed! Let's save it on disk */
322  byte = (RDB_ENCVAL<<6)|RDB_ENC_LZF;
323  if ((n = rdbWriteRaw(rdb,&byte,1)) == -1) goto writeerr;
324  nwritten += n;
325 
326  if ((n = rdbSaveLen(rdb,compress_len)) == -1) goto writeerr;
327  nwritten += n;
328

330  nwritten += n;
331 
332  if ((n = rdbWriteRaw(rdb,data,compress_len)) == -1) goto writeerr;
333  nwritten += n;
334

336 
337 writeerr:

359 /* Load an LZF compressed string in RDB format. The returned value
360  * changes according to 'flags'. For more info check the
361  * rdbGenericLoadStringObject() function. */
362 void *rdbLoadLzfStringObject(rio *rdb, int flags, size_t *lenptr) {
363  int plain = flags & RDB_LOAD_PLAIN;
364  int sds = flags & RDB_LOAD_SDS;
365  uint64_t len, clen;
366  unsigned char *c = NULL;
367  char *val = NULL;
368 
369  if ((clen = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
370  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
371  if ((c = zmalloc(clen)) == NULL) goto err;
372 
373  /* Allocate our target according to the uncompressed size. */
374  if (plain) {
375  val = zmalloc(len);
376  } else {
377  val = sdsnewlen(SDS_NOINIT,len);
378  }
379  if (lenptr) *lenptr = len;
380 
381  /* Load the compressed representation and uncompress it to target. */
382  if (rioRead(rdb,c,clen) == 0) goto err;
383  if (lzf_decompress(c,clen,val,len) == 0) {
384  if (rdbCheckMode) rdbCheckSetError("Invalid LZF compressed string");
385  goto err;
386  }
387  zfree(c);
388 
389  if (plain || sds) {
390  return val;
391  } else {
392  return createObject(OBJ_STRING,val);
393  }
394 err:
395  zfree(c);
396  if (plain)
397  zfree(val);
398  else
399  sdsfree(val);
400  return NULL;
401 }

403 /* Save a string object as [len][data] on disk. If the object is a string
404  * representation of an integer value we try to save it in a special form */
405 ssize_t rdbSaveRawString(rio *rdb, unsigned char *s, size_t len) {
406  int enclen;
407  ssize_t n, nwritten = 0;
408 
409  /* Try integer encoding */
410  if (len <= 11) {
411  unsigned char buf[5];
412  if ((enclen = rdbTryIntegerEncoding((char*)s,len,buf)) > 0) {
413  if (rdbWriteRaw(rdb,buf,enclen) == -1) return -1;
414  return enclen;
415  }
416  }
417 
418  /* Try LZF compression - under 20 bytes it's unable to compress even
419  * aaaaaaaaaaaaaaaaaa so skip it */
420  if (server.rdb_compression && len > 20) {
421  n = rdbSaveLzfStringObject(rdb,s,len);
422  if (n == -1) return -1;
423  if (n > 0) return n;
424  /* Return value of 0 means data can't be compressed, save the old way */
425  }
426 
427  /* Store verbatim */
428  if ((n = rdbSaveLen(rdb,len)) == -1) return -1;
429  nwritten += n;
430  if (len > 0) {
431  if (rdbWriteRaw(rdb,s,len) == -1) return -1;
432  nwritten += len;
433  }
434  return nwritten;
435 }

468 /* Load a string object from an RDB file according to flags:
469  *
470  * RDB_LOAD_NONE (no flags): load an RDB object, unencoded.
471  * RDB_LOAD_ENC: If the returned type is a Redis object, try to
472  * encode it in a special way to be more memory
473  * efficient. When this flag is passed the function
474  * no longer guarantees that obj->ptr is an SDS string.
475  * RDB_LOAD_PLAIN: Return a plain string allocated with zmalloc()
476  * instead of a Redis object with an sds in it.
477  * RDB_LOAD_SDS: Return an SDS string instead of a Redis object.
478  *
479  * On I/O error NULL is returned.
480  */
481 void *rdbGenericLoadStringObject(rio *rdb, int flags, size_t *lenptr) {
482  int encode = flags & RDB_LOAD_ENC;
483  int plain = flags & RDB_LOAD_PLAIN;
484  int sds = flags & RDB_LOAD_SDS;
485  int isencoded;
486  uint64_t len;
487 
488  len = rdbLoadLen(rdb,&isencoded);
489  if (isencoded) {
490  switch(len) {
491  case RDB_ENC_INT8:
492  case RDB_ENC_INT16:
493  case RDB_ENC_INT32:
494  return rdbLoadIntegerObject(rdb,len,flags,lenptr);
495  case RDB_ENC_LZF:
496  return rdbLoadLzfStringObject(rdb,flags,lenptr);
497  default:
498  rdbExitReportCorruptRDB("Unknown RDB string encoding type %d",len);
499  }
500  }
501 
502  if (len == RDB_LENERR) return NULL;
503  if (plain || sds) {
504  void *buf = plain ? zmalloc(len) : sdsnewlen(SDS_NOINIT,len);
505  if (lenptr) *lenptr = len;
506  if (len && rioRead(rdb,buf,len) == 0) {
507  if (plain)
508  zfree(buf);
509  else
510  sdsfree(buf);
511  return NULL;
512  }
513  return buf;
514  } else {
515  robj *o = encode ? createStringObject(SDS_NOINIT,len) :
516  createRawStringObject(SDS_NOINIT,len);
517  if (len && rioRead(rdb,o->ptr,len) == 0) {
518  decrRefCount(o);
519  return NULL;
520  }
521  return o;
522  }
523 }

533 /* Save a double value. Doubles are saved as strings prefixed by an unsigned
534  * 8 bit integer specifying the length of the representation.
535  * This 8 bit integer has special values in order to specify the following
536  * conditions:
537  * 253: not a number
538  * 254: + inf
539  * 255: - inf
540  */
541 int rdbSaveDoubleValue(rio *rdb, double val) {
542  unsigned char buf[128];
543  int len;
544 
545  if (isnan(val)) {
546  buf[0] = 253;
547  len = 1;
548  } else if (!isfinite(val)) {
549  len = 1;
550  buf[0] = (val < 0) ? 255 : 254;
551  } else {
552 #if (DBL_MANT_DIG >= 52) && (LLONG_MAX == 0x7fffffffffffffffLL)
553  /* Check if the float is in a safe range to be casted into a
554  * long long. We are assuming that long long is 64 bit here.
555  * Also we are assuming that there are no implementations around where
556  * double has precision < 52 bit.
557  *
558  * Under this assumptions we test if a double is inside an interval
559  * where casting to long long is safe. Then using two castings we
560  * make sure the decimal part is zero. If all this is true we use
561  * integer printing function that is much faster. */
562  double min = -4503599627370495; /* (2^52)-1 */
563  double max = 4503599627370496; /* -(2^52) */
564  if (val > min && val < max && val == ((double)((long long)val)))
565  ll2string((char*)buf+1,sizeof(buf)-1,(long long)val);
566  else
567 #endif
568  snprintf((char*)buf+1,sizeof(buf)-1,"%.17g",val);
569  buf[0] = strlen((char*)buf+1);
570  len = buf[0]+1;
571  }
572  return rdbWriteRaw(rdb,buf,len);
573 }

576 int rdbLoadDoubleValue(rio *rdb, double *val) {
577  char buf[256];
578  unsigned char len;
579 
580  if (rioRead(rdb,&len,1) == 0) return -1;
581  switch(len) {
582  case 255: *val = R_NegInf; return 0;

674 /* This helper function serializes a consumer group Pending Entries List (PEL)
675  * into the RDB file. The 'nacks' argument tells the function if also persist
676  * the informations about the not acknowledged message, or if to persist
677  * just the IDs: this is useful because for the global consumer group PEL
678  * we serialized the NACKs as well, but when serializing the local consumer
679  * PELs we just add the ID, that will be resolved inside the global PEL to
680  * put a reference to the same structure. */
681 ssize_t rdbSaveStreamPEL(rio *rdb, rax *pel, int nacks) {
682  ssize_t n, nwritten = 0;
683 
684  /* Number of entries in the PEL. */
685  if ((n = rdbSaveLen(rdb,raxSize(pel))) == -1) return -1;
686  nwritten += n;
687 
688  /* Save each entry. */
689  raxIterator ri;
690  raxStart(&ri,pel);
691  raxSeek(&ri,"^",NULL,0);
692  while(raxNext(&ri)) {
693  /* We store IDs in raw form as 128 big big endian numbers, like
694  * they are inside the radix tree key. */
695  if ((n = rdbWriteRaw(rdb,ri.key,sizeof(streamID))) == -1) return -1;
696  nwritten += n;
697 
698  if (nacks) {
699  streamNACK *nack = ri.data;
700  if ((n = rdbSaveMillisecondTime(rdb,nack->delivery_time)) == -1)
701  return -1;
702  nwritten += n;
703  if ((n = rdbSaveLen(rdb,nack->delivery_count)) == -1) return -1;
704  nwritten += n;
705  /* We don't save the consumer name: we'll save the pending IDs
706  * for each consumer in the consumer PEL, and resolve the consumer
707  * at loading time. */
708  }
709  }
710  raxStop(&ri);
711  return nwritten;
712 }

715  * function for the stream data type serialization. What we do here is to
716  * persist the consumer metadata, and it's PEL, for each consumer. */

752 /* Save a Redis object.
753  * Returns -1 on error, number of bytes written on success. */
754 ssize_t rdbSaveObject(rio *rdb, robj *o) {
755  ssize_t n = 0, nwritten = 0;
756 
757  if (o->type == OBJ_STRING) {
758  /* Save a string value */
759  if ((n = rdbSaveStringObject(rdb,o)) == -1) return -1;
760  nwritten += n;
761  } else if (o->type == OBJ_LIST) {
762  /* Save a list value */
763  if (o->encoding == OBJ_ENCODING_QUICKLIST) {
764  quicklist *ql = o->ptr;
765  quicklistNode *node = ql->head;
766 
767  if ((n = rdbSaveLen(rdb,ql->len)) == -1) return -1;
768  nwritten += n;
769 
770  while(node) {
771  if (quicklistNodeIsCompressed(node)) {
772  void *data;
773  size_t compress_len = quicklistGetLzf(node, &data);
774  if ((n = rdbSaveLzfBlob(rdb,data,compress_len,node->sz)) == -1) return -1;
775  nwritten += n;
776  } else {
777  if ((n = rdbSaveRawString(rdb,node->zl,node->sz)) == -1) return -1;
778  nwritten += n;
779  }
780  node = node->next;
781  }
782  } else {
783  serverPanic("Unknown list encoding");
784  }
785  } else if (o->type == OBJ_SET) {
786  /* Save a set value */
787  if (o->encoding == OBJ_ENCODING_HT) {
788  dict *set = o->ptr;
789  dictIterator *di = dictGetIterator(set);
790  dictEntry *de;
791 
792  if ((n = rdbSaveLen(rdb,dictSize(set))) == -1) {
793  dictReleaseIterator(di);
794  return -1;
795  }
796  nwritten += n;
797 
798  while((de = dictNext(di)) != NULL) {
799  sds ele = dictGetKey(de);
800  if ((n = rdbSaveRawString(rdb,(unsigned char*)ele,sdslen(ele)))
801  == -1)
802  {
803  dictReleaseIterator(di);
804  return -1;
805  }
806  nwritten += n;
807  }
808  dictReleaseIterator(di);
809  } else if (o->encoding == OBJ_ENCODING_INTSET) {
810  size_t l = intsetBlobLen((intset*)o->ptr);
811 
812  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
813  nwritten += n;
814  } else {
815  serverPanic("Unknown set encoding");
816  }
817  } else if (o->type == OBJ_ZSET) {
818  /* Save a sorted set value */
819  if (o->encoding == OBJ_ENCODING_ZIPLIST) {
820  size_t l = ziplistBlobLen((unsigned char*)o->ptr);
821 
822  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
823  nwritten += n;
824  } else if (o->encoding == OBJ_ENCODING_SKIPLIST) {
825  zset *zs = o->ptr;
826  zskiplist *zsl = zs->zsl;
827 
828  if ((n = rdbSaveLen(rdb,zsl->length)) == -1) return -1;
829  nwritten += n;
830 
831  /* We save the skiplist elements from the greatest to the smallest
832  * (that's trivial since the elements are already ordered in the
833  * skiplist): this improves the load process, since the next loaded
834  * element will always be the smaller, so adding to the skiplist
835  * will always immediately stop at the head, making the insertion
836  * O(1) instead of O(log(N)). */
837  zskiplistNode *zn = zsl->tail;
838  while (zn != NULL) {
839  if ((n = rdbSaveRawString(rdb,
840  (unsigned char*)zn->ele,sdslen(zn->ele))) == -1)
841  {
842  return -1;
843  }
844  nwritten += n;
845  if ((n = rdbSaveBinaryDoubleValue(rdb,zn->score)) == -1)
846  return -1;
847  nwritten += n;
848  zn = zn->backward;
849  }
850  } else {
851  serverPanic("Unknown sorted set encoding");
852  }
853  } else if (o->type == OBJ_HASH) {
854  /* Save a hash value */
855  if (o->encoding == OBJ_ENCODING_ZIPLIST) {
856  size_t l = ziplistBlobLen((unsigned char*)o->ptr);
857 
858  if ((n = rdbSaveRawString(rdb,o->ptr,l)) == -1) return -1;
859  nwritten += n;
860 
861  } else if (o->encoding == OBJ_ENCODING_HT) {
862  dictIterator *di = dictGetIterator(o->ptr);
863  dictEntry *de;
864 
865  if ((n = rdbSaveLen(rdb,dictSize((dict*)o->ptr))) == -1) {
866  dictReleaseIterator(di);
867  return -1;
868  }
869  nwritten += n;
870 
871  while((de = dictNext(di)) != NULL) {
872  sds field = dictGetKey(de);
873  sds value = dictGetVal(de);
874 
875  if ((n = rdbSaveRawString(rdb,(unsigned char*)field,
876  sdslen(field))) == -1)
877  {
878  dictReleaseIterator(di);
879  return -1;
880  }
881  nwritten += n;
882  if ((n = rdbSaveRawString(rdb,(unsigned char*)value,
883  sdslen(value))) == -1)
884  {
885  dictReleaseIterator(di);
886  return -1;
887  }
888  nwritten += n;
889  }
890  dictReleaseIterator(di);
891  } else {
892  serverPanic("Unknown hash encoding");
893  }
894  } else if (o->type == OBJ_STREAM) {
895  /* Store how many listpacks we have inside the radix tree. */
896  stream *s = o->ptr;
897  rax *rax = s->rax;
898  if ((n = rdbSaveLen(rdb,raxSize(rax))) == -1) return -1;
899  nwritten += n;
900 
901  /* Serialize all the listpacks inside the radix tree as they are,
902  * when loading back, we'll use the first entry of each listpack
903  * to insert it back into the radix tree. */
904  raxIterator ri;
905  raxStart(&ri,rax);
906  raxSeek(&ri,"^",NULL,0);
907  while (raxNext(&ri)) {
908  unsigned char *lp = ri.data;
909  size_t lp_bytes = lpBytes(lp);
910  if ((n = rdbSaveRawString(rdb,ri.key,ri.key_len)) == -1) return -1;
911  nwritten += n;
912  if ((n = rdbSaveRawString(rdb,lp,lp_bytes)) == -1) return -1;
913  nwritten += n;
914  }
915  raxStop(&ri);
916 
917  /* Save the number of elements inside the stream. We cannot obtain
918  * this easily later, since our macro nodes should be checked for
919  * number of items: not a great CPU / space tradeoff. */
920  if ((n = rdbSaveLen(rdb,s->length)) == -1) return -1;
921  nwritten += n;
922  /* Save the last entry ID. */
923  if ((n = rdbSaveLen(rdb,s->last_id.ms)) == -1) return -1;
924  nwritten += n;
925  if ((n = rdbSaveLen(rdb,s->last_id.seq)) == -1) return -1;
926  nwritten += n;
927 
928  /* The consumer groups and their clients are part of the stream
929  * type, so serialize every consumer group. */
930 
931  /* Save the number of groups. */
932  size_t num_cgroups = s->cgroups ? raxSize(s->cgroups) : 0;
933  if ((n = rdbSaveLen(rdb,num_cgroups)) == -1) return -1;
934  nwritten += n;
935 
936  if (num_cgroups) {
937  /* Serialize each consumer group. */
938  raxStart(&ri,s->cgroups);
939  raxSeek(&ri,"^",NULL,0);
940  while(raxNext(&ri)) {
941  streamCG *cg = ri.data;
942 
943  /* Save the group name. */
944  if ((n = rdbSaveRawString(rdb,ri.key,ri.key_len)) == -1)
945  return -1;
946  nwritten += n;
947 
948  /* Last ID. */
949  if ((n = rdbSaveLen(rdb,cg->last_id.ms)) == -1) return -1;
950  nwritten += n;
951  if ((n = rdbSaveLen(rdb,cg->last_id.seq)) == -1) return -1;
952  nwritten += n;
953 
954  /* Save the global PEL. */
955  if ((n = rdbSaveStreamPEL(rdb,cg->pel,1)) == -1) return -1;
956  nwritten += n;
957 
958  /* Save the consumers of this group. */
959  if ((n = rdbSaveStreamConsumers(rdb,cg)) == -1) return -1;
960  nwritten += n;
961  }
962  raxStop(&ri);
963  }
964  } else if (o->type == OBJ_MODULE) {
965  /* Save a module-specific value. */
966  RedisModuleIO io;
967  moduleValue *mv = o->ptr;
968  moduleType *mt = mv->type;
969  moduleInitIOContext(io,mt,rdb);
970 
971  /* Write the "module" identifier as prefix, so that we'll be able
972  * to call the right module during loading. */
973  int retval = rdbSaveLen(rdb,mt->id);
974  if (retval == -1) return -1;
975  io.bytes += retval;
976 
977  /* Then write the module-specific representation + EOF marker. */
978  mt->rdb_save(&io,mv->value);
979  retval = rdbSaveLen(rdb,RDB_MODULE_OPCODE_EOF);
980  if (retval == -1) return -1;
981  io.bytes += retval;
982 
983  if (io.ctx) {
984  moduleFreeContext(io.ctx);
985  zfree(io.ctx);
986  }
987  return io.error ? -1 : (ssize_t)io.bytes;
988  } else {
989  serverPanic("Unknown object type");
990  }
991  return nwritten;
992 }

1005  * On error -1 is returned.
1006  * On success if the key was actually saved 1 is returned, otherwise 0
1007  * is returned (the key was already expired). */
1008 int rdbSaveKeyValuePair(rio *rdb, robj *key, robj *val, long long expiretime) {
1009  int savelru = server.maxmemory_policy & MAXMEMORY_FLAG_LRU;
1010  int savelfu = server.maxmemory_policy & MAXMEMORY_FLAG_LFU;
1011 
1012  /* Save the expire time */
1013  if (expiretime != -1) {
1014  if (rdbSaveType(rdb,RDB_OPCODE_EXPIRETIME_MS) == -1) return -1;
1015  if (rdbSaveMillisecondTime(rdb,expiretime) == -1) return -1;

1023  if (rdbSaveLen(rdb,idletime) == -1) return -1;

1040  if (rdbSaveStringObject(rdb,key) == -1) return -1;
1041  if (rdbSaveObject(rdb,val) == -1) return -1;

1094 /* Produces a dump of the database in RDB format sending it to the specified
1095  * Redis I/O channel. On success C_OK is returned, otherwise C_ERR
1096  * is returned and part of the output, or all the output, can be
1097  * missing because of I/O errors.
1098  *
1099  * When the function returns C_ERR and if 'error' is not NULL, the
1100  * integer pointed by 'error' is set to the value of errno just after the I/O
1101  * error. */
1102 int rdbSaveRio(rio *rdb, int *error, int flags, rdbSaveInfo *rsi) {
1103  dictIterator *di = NULL;
1104  dictEntry *de;
1105  char magic[10];
1106  int j;
1107  uint64_t cksum;
1108  size_t processed = 0;
1109 
1110  if (server.rdb_checksum)
1111  rdb->update_cksum = rioGenericUpdateChecksum;
1112  snprintf(magic,sizeof(magic),"REDIS%04d",RDB_VERSION);
1113  if (rdbWriteRaw(rdb,magic,9) == -1) goto werr;
1114  if (rdbSaveInfoAuxFields(rdb,flags,rsi) == -1) goto werr;
1115 
1116  for (j = 0; j < server.dbnum; j++) {
1117  redisDb *db = server.db+j;
1118  dict *d = db->dict;
1119  if (dictSize(d) == 0) continue;
1120  di = dictGetSafeIterator(d);
1121 
1122  /* Write the SELECT DB opcode */
1123  if (rdbSaveType(rdb,RDB_OPCODE_SELECTDB) == -1) goto werr;
1124  if (rdbSaveLen(rdb,j) == -1) goto werr;
1125 
1126  /* Write the RESIZE DB opcode. We trim the size to UINT32_MAX, which
1127  * is currently the largest type we are able to represent in RDB sizes.
1128  * However this does not limit the actual size of the DB to load since
1129  * these sizes are just hints to resize the hash tables. */
1130  uint64_t db_size, expires_size;
1131  db_size = dictSize(db->dict);
1132  expires_size = dictSize(db->expires);
1133  if (rdbSaveType(rdb,RDB_OPCODE_RESIZEDB) == -1) goto werr;
1134  if (rdbSaveLen(rdb,db_size) == -1) goto werr;

1381 /* Load a Redis object of the specified type from the specified file.
1382  * On success a newly allocated object is returned, otherwise NULL. */
1383 robj *rdbLoadObject(int rdbtype, rio *rdb) {
1384  robj *o = NULL, *ele, *dec;
1385  uint64_t len;
1386  unsigned int i;
1387 
1388  if (rdbtype == RDB_TYPE_STRING) {
1389  /* Read string value */
1390  if ((o = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
1391  o = tryObjectEncoding(o);
1392  } else if (rdbtype == RDB_TYPE_LIST) {
1393  /* Read list value */
1394  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1395 
1396  o = createQuicklistObject();
1397  quicklistSetOptions(o->ptr, server.list_max_ziplist_size,
1398  server.list_compress_depth);
1399 
1400  /* Load every single element of the list */
1401  while(len--) {
1402  if ((ele = rdbLoadEncodedStringObject(rdb)) == NULL) return NULL;
1403  dec = getDecodedObject(ele);
1404  size_t len = sdslen(dec->ptr);
1405  quicklistPushTail(o->ptr, dec->ptr, len);
1406  decrRefCount(dec);
1407  decrRefCount(ele);
1408  }
1409  } else if (rdbtype == RDB_TYPE_SET) {
1410  /* Read Set value */
1411  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1412 
1413  /* Use a regular set when there are too many entries. */
1414  if (len > server.set_max_intset_entries) {
1415  o = createSetObject();
1416  /* It's faster to expand the dict to the right size asap in order
1417  * to avoid rehashing */
1418  if (len > DICT_HT_INITIAL_SIZE)
1419  dictExpand(o->ptr,len);
1420  } else {
1421  o = createIntsetObject();
1422  }
1423 
1424  /* Load every single element of the set */
1425  for (i = 0; i < len; i++) {
1426  long long llval;
1427  sds sdsele;
1428 
1429  if ((sdsele = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1430  == NULL) return NULL;
1431 
1432  if (o->encoding == OBJ_ENCODING_INTSET) {
1433  /* Fetch integer value from element. */
1434  if (isSdsRepresentableAsLongLong(sdsele,&llval) == C_OK) {
1435  o->ptr = intsetAdd(o->ptr,llval,NULL);
1436  } else {
1437  setTypeConvert(o,OBJ_ENCODING_HT);
1438  dictExpand(o->ptr,len);
1439  }
1440  }
1441 
1442  /* This will also be called when the set was just converted
1443  * to a regular hash table encoded set. */
1444  if (o->encoding == OBJ_ENCODING_HT) {
1445  dictAdd((dict*)o->ptr,sdsele,NULL);
1446  } else {
1447  sdsfree(sdsele);
1448  }
1449  }
1450  } else if (rdbtype == RDB_TYPE_ZSET_2 || rdbtype == RDB_TYPE_ZSET) {
1451  /* Read list/set value. */
1452  uint64_t zsetlen;
1453  size_t maxelelen = 0;
1454  zset *zs;
1455 
1456  if ((zsetlen = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1457  o = createZsetObject();
1458  zs = o->ptr;
1459 
1460  if (zsetlen > DICT_HT_INITIAL_SIZE)
1461  dictExpand(zs->dict,zsetlen);
1462 
1463  /* Load every single element of the sorted set. */
1464  while(zsetlen--) {
1465  sds sdsele;
1466  double score;
1467  zskiplistNode *znode;
1468 
1469  if ((sdsele = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1470  == NULL) return NULL;
1471 
1472  if (rdbtype == RDB_TYPE_ZSET_2) {
1473  if (rdbLoadBinaryDoubleValue(rdb,&score) == -1) return NULL;
1474  } else {
1475  if (rdbLoadDoubleValue(rdb,&score) == -1) return NULL;
1476  }
1477 
1478  /* Don't care about integer-encoded strings. */
1479  if (sdslen(sdsele) > maxelelen) maxelelen = sdslen(sdsele);
1480 
1481  znode = zslInsert(zs->zsl,score,sdsele);
1482  dictAdd(zs->dict,sdsele,&znode->score);
1483  }
1484 
1485  /* Convert *after* loading, since sorted sets are not stored ordered. */
1486  if (zsetLength(o) <= server.zset_max_ziplist_entries &&
1487  maxelelen <= server.zset_max_ziplist_value)
1488  zsetConvert(o,OBJ_ENCODING_ZIPLIST);
1489  } else if (rdbtype == RDB_TYPE_HASH) {
1490  uint64_t len;
1491  int ret;
1492  sds field, value;
1493 
1494  len = rdbLoadLen(rdb, NULL);
1495  if (len == RDB_LENERR) return NULL;
1496 
1497  o = createHashObject();
1498 
1499  /* Too many entries? Use a hash table. */
1500  if (len > server.hash_max_ziplist_entries)
1501  hashTypeConvert(o, OBJ_ENCODING_HT);
1502 
1503  /* Load every field and value into the ziplist */
1504  while (o->encoding == OBJ_ENCODING_ZIPLIST && len > 0) {
1505  len--;
1506  /* Load raw strings */
1507  if ((field = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1508  == NULL) return NULL;
1509  if ((value = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1510  == NULL) return NULL;
1511 
1512  /* Add pair to ziplist */
1513  o->ptr = ziplistPush(o->ptr, (unsigned char*)field,
1514  sdslen(field), ZIPLIST_TAIL);
1515  o->ptr = ziplistPush(o->ptr, (unsigned char*)value,
1516  sdslen(value), ZIPLIST_TAIL);
1517 
1518  /* Convert to hash table if size threshold is exceeded */
1519  if (sdslen(field) > server.hash_max_ziplist_value ||
1520  sdslen(value) > server.hash_max_ziplist_value)
1521  {
1522  sdsfree(field);
1523  sdsfree(value);
1524  hashTypeConvert(o, OBJ_ENCODING_HT);
1525  break;
1526  }
1527  sdsfree(field);
1528  sdsfree(value);
1529  }
1530 
1531  if (o->encoding == OBJ_ENCODING_HT && len > DICT_HT_INITIAL_SIZE)
1532  dictExpand(o->ptr,len);
1533 
1534  /* Load remaining fields and values into the hash table */
1535  while (o->encoding == OBJ_ENCODING_HT && len > 0) {
1536  len--;
1537  /* Load encoded strings */
1538  if ((field = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1539  == NULL) return NULL;
1540  if ((value = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL))
1541  == NULL) return NULL;
1542 
1543  /* Add pair to hash table */
1544  ret = dictAdd((dict*)o->ptr, field, value);
1545  if (ret == DICT_ERR) {
1546  rdbExitReportCorruptRDB("Duplicate keys detected");
1547  }
1548  }
1549 
1550  /* All pairs should be read by now */
1551  serverAssert(len == 0);
1552  } else if (rdbtype == RDB_TYPE_LIST_QUICKLIST) {
1553  if ((len = rdbLoadLen(rdb,NULL)) == RDB_LENERR) return NULL;
1554  o = createQuicklistObject();
1555  quicklistSetOptions(o->ptr, server.list_max_ziplist_size,
1556  server.list_compress_depth);
1557 
1558  while (len--) {
1559  unsigned char *zl =
1560  rdbGenericLoadStringObject(rdb,RDB_LOAD_PLAIN,NULL);
1561  if (zl == NULL) return NULL;
1562  quicklistAppendZiplist(o->ptr, zl);
1563  }
1564  } else if (rdbtype == RDB_TYPE_HASH_ZIPMAP ||
1565  rdbtype == RDB_TYPE_LIST_ZIPLIST ||
1566  rdbtype == RDB_TYPE_SET_INTSET ||
1567  rdbtype == RDB_TYPE_ZSET_ZIPLIST ||
1568  rdbtype == RDB_TYPE_HASH_ZIPLIST)
1569  {
1570  unsigned char *encoded =
1571  rdbGenericLoadStringObject(rdb,RDB_LOAD_PLAIN,NULL);
1572  if (encoded == NULL) return NULL;
1573  o = createObject(OBJ_STRING,encoded); /* Obj type fixed below. */
1574 
1575  /* Fix the object encoding, and make sure to convert the encoded
1576  * data type into the base type if accordingly to the current
1577  * configuration there are too many elements in the encoded data
1578  * type. Note that we only check the length and not max element
1579  * size as this is an O(N) scan. Eventually everything will get
1580  * converted. */
1581  switch(rdbtype) {
1582  case RDB_TYPE_HASH_ZIPMAP:
1583  /* Convert to ziplist encoded hash. This must be deprecated
1584  * when loading dumps created by Redis 2.4 gets deprecated. */
1585  {
1586  unsigned char *zl = ziplistNew();
1587  unsigned char *zi = zipmapRewind(o->ptr);
1588  unsigned char *fstr, *vstr;
1589  unsigned int flen, vlen;
1590  unsigned int maxlen = 0;
1591 
1592  while ((zi = zipmapNext(zi, &fstr, &flen, &vstr, &vlen)) != NULL) {
1593  if (flen > maxlen) maxlen = flen;
1594  if (vlen > maxlen) maxlen = vlen;
1595  zl = ziplistPush(zl, fstr, flen, ZIPLIST_TAIL);
1596  zl = ziplistPush(zl, vstr, vlen, ZIPLIST_TAIL);
1597  }
1598 
1599  zfree(o->ptr);
1600  o->ptr = zl;
1601  o->type = OBJ_HASH;
1602  o->encoding = OBJ_ENCODING_ZIPLIST;
1603 
1604  if (hashTypeLength(o) > server.hash_max_ziplist_entries ||
1605  maxlen > server.hash_max_ziplist_value)
1606  {
1607  hashTypeConvert(o, OBJ_ENCODING_HT);
1608  }
1609  }
1610  break;
1611  case RDB_TYPE_LIST_ZIPLIST:
1612  o->type = OBJ_LIST;
1613  o->encoding = OBJ_ENCODING_ZIPLIST;
1614  listTypeConvert(o,OBJ_ENCODING_QUICKLIST);
1615  break;
1616  case RDB_TYPE_SET_INTSET:
1617  o->type = OBJ_SET;
1618  o->encoding = OBJ_ENCODING_INTSET;
1619  if (intsetLen(o->ptr) > server.set_max_intset_entries)
1620  setTypeConvert(o,OBJ_ENCODING_HT);
1621  break;
1622  case RDB_TYPE_ZSET_ZIPLIST:
1623  o->type = OBJ_ZSET;
1624  o->encoding = OBJ_ENCODING_ZIPLIST;
1625  if (zsetLength(o) > server.zset_max_ziplist_entries)
1626  zsetConvert(o,OBJ_ENCODING_SKIPLIST);
1627  break;
1628  case RDB_TYPE_HASH_ZIPLIST:
1629  o->type = OBJ_HASH;
1630  o->encoding = OBJ_ENCODING_ZIPLIST;
1631  if (hashTypeLength(o) > server.hash_max_ziplist_entries)
1632  hashTypeConvert(o, OBJ_ENCODING_HT);
1633  break;
1634  default:
1635  rdbExitReportCorruptRDB("Unknown RDB encoding type %d",rdbtype);
1636  break;
1637  }
1638  } else if (rdbtype == RDB_TYPE_STREAM_LISTPACKS) {
1639  o = createStreamObject();
1640  stream *s = o->ptr;
1641  uint64_t listpacks = rdbLoadLen(rdb,NULL);
1642 
1643  while(listpacks--) {
1644  /* Get the master ID, the one we'll use as key of the radix tree
1645  * node: the entries inside the listpack itself are delta-encoded
1646  * relatively to this ID. */
1647  sds nodekey = rdbGenericLoadStringObject(rdb,RDB_LOAD_SDS,NULL);
1648  if (sdslen(nodekey) != sizeof(streamID)) {
1649  rdbExitReportCorruptRDB("Stream node key entry is not the "
1650  "size of a stream ID");
1651  }
1652

1654  unsigned char *lp =
1655  rdbGenericLoadStringObject(rdb,RDB_LOAD_PLAIN,NULL);
1656  if (lp == NULL) return NULL;
1657  unsigned char *first = lpFirst(lp);
1658  if (first == NULL) {

1660  * deletion we should remove the radix tree key if the
1661  * resulting listpack is empty. */
1662  rdbExitReportCorruptRDB("Empty listpack inside stream");
1663  }
1664 
1665  /* Insert the key in the radix tree. */

1667  (unsigned char*)nodekey,sizeof(streamID),lp,NULL);
1668  sdsfree(nodekey);
1669  if (!retval)
1670  rdbExitReportCorruptRDB("Listpack re-added with existing key");
1671  }
1672  /* Load total number of items inside the stream. */
1673  s->length = rdbLoadLen(rdb,NULL);
1674  /* Load the last entry ID. */
1675  s->last_id.ms = rdbLoadLen(rdb,NULL);
1676  s->last_id.seq = rdbLoadLen(rdb,NULL);
1677 
1678  /* Consumer groups loading */
1679  size_t cgroups_count = rdbLoadLen(rdb,NULL);
1680  while(cgroups_count--) {
1681  /* Get the consumer group name and ID. We can then create the
1682  * consumer group ASAP and populate its structure as
1683  * we read more data. */

1699  * not yet populate the NACK structures with the message
1700  * owner, since consumers for this group and their messages will
1701  * be read as a next step. So for now leave them not resolved
1702  * and later populate it. */

1704  while(pel_size--) {

1709  nack->delivery_count = rdbLoadLen(rdb,NULL);

1711  rdbExitReportCorruptRDB("Duplicated gobal PEL entry "
1712  "loading stream consumer group");

1716  * consumers and their local PELs. */

1725  1);
1726  sdsfree(cname);

1728

1730  * consumer. */

1732  while(pel_size--) {

1738  "group global PEL");

1741  * loading the global PEL. Then set the same shared
1742  * NACK structure also in the consumer-specific PEL. */

1744  if (!raxInsert(consumer->pel,rawid,sizeof(rawid),nack,NULL))
1745  rdbExitReportCorruptRDB("Duplicated consumer PEL entry "
1746  " loading a stream consumer "
1747  "group");
1748  }
1749  }
1750  }
1751  } else if (rdbtype == RDB_TYPE_MODULE || rdbtype == RDB_TYPE_MODULE_2) {
1752  uint64_t moduleid = rdbLoadLen(rdb,NULL);
1753  moduleType *mt = moduleTypeLookupModuleByID(moduleid);

1845 /* Load an RDB file from the rio stream 'rdb'. On success C_OK is returned,
1846  * otherwise C_ERR is returned and 'errno' is set accordingly. */
1847 int rdbLoadRio(rio *rdb, rdbSaveInfo *rsi, int loading_aof) {
1848  uint64_t dbid;
1849  int type, rdbver;
1850  redisDb *db = server.db+0;
1851  char buf[1024];
1852 
1853  rdb->update_cksum = rdbLoadProgressCallback;
1854  rdb->max_processing_chunk = server.loading_process_events_interval_bytes;
1855  if (rioRead(rdb,buf,9) == 0) goto eoferr;
1856  buf[9] = '\0';
1857  if (memcmp(buf,"REDIS",5) != 0) {
1858  serverLog(LL_WARNING,"Wrong signature trying to load DB from file");
1859  errno = EINVAL;
1860  return C_ERR;
1861  }
1862  rdbver = atoi(buf+5);
1863  if (rdbver < 1 || rdbver > RDB_VERSION) {
1864  serverLog(LL_WARNING,"Can't handle RDB format version %d",rdbver);
1865  errno = EINVAL;
1866  return C_ERR;
1867  }
1868 
1869  /* Key-specific attributes, set by opcodes before the key type. */
1870  long long lru_idle = -1, lfu_freq = -1, expiretime = -1, now = mstime();
1871  long long lru_clock = LRU_CLOCK();
1872 
1873  while(1) {
1874  robj *key, *val;
1875 
1876  /* Read type. */
1877  if ((type = rdbLoadType(rdb)) == -1) goto eoferr;
1878 
1879  /* Handle special types. */
1880  if (type == RDB_OPCODE_EXPIRETIME) {
1881  /* EXPIRETIME: load an expire associated with the next key
1882  * to load. Note that after loading an expire we need to
1883  * load the actual type, and continue. */
1884  expiretime = rdbLoadTime(rdb);
1885  expiretime *= 1000;
1886  continue; /* Read next opcode. */
1887  } else if (type == RDB_OPCODE_EXPIRETIME_MS) {
1888  /* EXPIRETIME_MS: milliseconds precision expire times introduced
1889  * with RDB v3. Like EXPIRETIME but no with more precision. */
1890  expiretime = rdbLoadMillisecondTime(rdb,rdbver);
1891  continue; /* Read next opcode. */
1892  } else if (type == RDB_OPCODE_FREQ) {
1893  /* FREQ: LFU frequency. */
1894  uint8_t byte;
1895  if (rioRead(rdb,&byte,1) == 0) goto eoferr;
1896  lfu_freq = byte;
1897  continue; /* Read next opcode. */
1898  } else if (type == RDB_OPCODE_IDLE) {
1899  /* IDLE: LRU idle time. */
1900  uint64_t qword;
1901  if ((qword = rdbLoadLen(rdb,NULL)) == RDB_LENERR) goto eoferr;
1902  lru_idle = qword;
1903  continue; /* Read next opcode. */
1904  } else if (type == RDB_OPCODE_EOF) {
1905  /* EOF: End of file, exit the main loop. */
1906  break;
1907  } else if (type == RDB_OPCODE_SELECTDB) {
1908  /* SELECTDB: Select the specified database. */
1909  if ((dbid = rdbLoadLen(rdb,NULL)) == RDB_LENERR) goto eoferr;
1910  if (dbid >= (unsigned)server.dbnum) {
1911  serverLog(LL_WARNING,
1912  "FATAL: Data file was created with a Redis "
1913  "server configured to handle more than %d "
1914  "databases. Exiting\n", server.dbnum);
1915  exit(1);
1916  }
1917  db = server.db+dbid;
1918  continue; /* Read next opcode. */
1919  } else if (type == RDB_OPCODE_RESIZEDB) {
1920  /* RESIZEDB: Hint about the size of the keys in the currently
1921  * selected data base, in order to avoid useless rehashing. */
1922  uint64_t db_size, expires_size;
1923  if ((db_size = rdbLoadLen(rdb,NULL)) == RDB_LENERR)
1924  goto eoferr;
1925  if ((expires_size = rdbLoadLen(rdb,NULL)) == RDB_LENERR)
1926  goto eoferr;
1927  dictExpand(db->dict,db_size);
1928  dictExpand(db->expires,expires_size);
1929  continue; /* Read next opcode. */
1930  } else if (type == RDB_OPCODE_AUX) {
1931  /* AUX: generic string-string fields. Use to add state to RDB
1932  * which is backward compatible. Implementations of RDB loading
1933  * are requierd to skip AUX fields they don't understand.
1934  *
1935  * An AUX field is composed of two strings: key and value. */
1936  robj *auxkey, *auxval;
1937  if ((auxkey = rdbLoadStringObject(rdb)) == NULL) goto eoferr;
1938  if ((auxval = rdbLoadStringObject(rdb)) == NULL) goto eoferr;
1939 
1940  if (((char*)auxkey->ptr)[0] == '%') {
1941  /* All the fields with a name staring with '%' are considered
1942  * information fields and are logged at startup with a log
1943  * level of NOTICE. */
1944  serverLog(LL_NOTICE,"RDB '%s': %s",
1945  (char*)auxkey->ptr,
1946  (char*)auxval->ptr);
1947  } else if (!strcasecmp(auxkey->ptr,"repl-stream-db")) {
1948  if (rsi) rsi->repl_stream_db = atoi(auxval->ptr);
1949  } else if (!strcasecmp(auxkey->ptr,"repl-id")) {
1950  if (rsi && sdslen(auxval->ptr) == CONFIG_RUN_ID_SIZE) {
1951  memcpy(rsi->repl_id,auxval->ptr,CONFIG_RUN_ID_SIZE+1);
1952  rsi->repl_id_is_set = 1;
1953  }
1954  } else if (!strcasecmp(auxkey->ptr,"repl-offset")) {
1955  if (rsi) rsi->repl_offset = strtoll(auxval->ptr,NULL,10);
1956  } else if (!strcasecmp(auxkey->ptr,"lua")) {
1957  /* Load the script back in memory. */
1958  if (luaCreateFunction(NULL,server.lua,auxval) == NULL) {
1959  rdbExitReportCorruptRDB(
1960  "Can't load Lua script from RDB file! "
1961  "BODY: %s", auxval->ptr);
1962  }
1963  } else {
1964  /* We ignore fields we don't understand, as by AUX field
1965  * contract. */
1966  serverLog(LL_DEBUG,"Unrecognized RDB AUX field: '%s'",
1967  (char*)auxkey->ptr);
1968  }
1969 
1970  decrRefCount(auxkey);
1971  decrRefCount(auxval);
1972  continue; /* Read type again. */
1973  } else if (type == RDB_OPCODE_MODULE_AUX) {
1974  /* This is just for compatibility with the future: we have plans
1975  * to add the ability for modules to store anything in the RDB
1976  * file, like data that is not related to the Redis key space.
1977  * Such data will potentially be stored both before and after the
1978  * RDB keys-values section. For this reason since RDB version 9,
1979  * we have the ability to read a MODULE_AUX opcode followed by an
1980  * identifier of the module, and a serialized value in "MODULE V2"
1981  * format. */
1982  uint64_t moduleid = rdbLoadLen(rdb,NULL);
1983  moduleType *mt = moduleTypeLookupModuleByID(moduleid);
1984  char name[10];
1985  moduleTypeNameByID(name,moduleid);
1986 
1987  if (!rdbCheckMode && mt == NULL) {
1988  /* Unknown module. */
1989  serverLog(LL_WARNING,"The RDB file contains AUX module data I can't load: no matching module '%s'", name);
1990  exit(1);
1991  } else if (!rdbCheckMode && mt != NULL) {
1992  /* This version of Redis actually does not know what to do
1993  * with modules AUX data... */
1994  serverLog(LL_WARNING,"The RDB file contains AUX module data I can't load for the module '%s'. Probably you want to use a newer version of Redis which implements aux data callbacks", name);
1995  exit(1);
1996  } else {
1997  /* RDB check mode. */
1998  robj *aux = rdbLoadCheckModuleValue(rdb,name);
1999  decrRefCount(aux);
2000  }
2001  }
2002 
2003  /* Read key */
2004  if ((key = rdbLoadStringObject(rdb)) == NULL) goto eoferr;
2005  /* Read value */
2006  if ((val = rdbLoadObject(type,rdb)) == NULL) goto eoferr;
2007  /* Check if the key already expired. This function is used when loading
2008  * an RDB file from disk, either at startup, or when an RDB was
2009  * received from the master. In the latter case, the master is
2010  * responsible for key expiry. If we would expire keys here, the
2011  * snapshot taken by the master may not be reflected on the slave. */
2012  if (server.masterhost == NULL && !loading_aof && expiretime != -1 && expiretime < now) {
2013  decrRefCount(key);
2014  decrRefCount(val);
2015  } else {
2016  /* Add the new object in the hash table */
2017  dbAdd(db,key,val);
2018 
2019  /* Set the expire time if needed */
2020  if (expiretime != -1) setExpire(NULL,db,key,expiretime);
2021 
2022  /* Set usage information (for eviction). */
2023  objectSetLRUOrLFU(val,lfu_freq,lru_idle,lru_clock);
2024 
2025  /* Decrement the key refcount since dbAdd() will take its
2026  * own reference. */
2027  decrRefCount(key);
2028  }
2029 
2030  /* Reset the state that is key-specified and is populated by
2031  * opcodes before the key, so that we start from scratch again. */
2032  expiretime = -1;
2033  lfu_freq = -1;
2034  lru_idle = -1;
2035  }
2036  /* Verify the checksum if RDB version is >= 5 */
2037  if (rdbver >= 5) {
2038  uint64_t cksum, expected = rdb->cksum;
2039 
2040  if (rioRead(rdb,&cksum,8) == 0) goto eoferr;
2041  if (server.rdb_checksum) {
2042  memrev64ifbe(&cksum);
2043  if (cksum == 0) {
2044  serverLog(LL_WARNING,"RDB file was saved with checksum disabled: no check performed.");
2045  } else if (cksum != expected) {
2046  serverLog(LL_WARNING,"Wrong RDB checksum. Aborting now.");
2047  rdbExitReportCorruptRDB("RDB CRC error");
2048  }
2049  }
2050  }
2051  return C_OK;
2052 
2053 eoferr: /* unexpected end of file is handled here with a fatal exit */
2054  serverLog(LL_WARNING,"Short read or OOM loading DB. Unrecoverable error, aborting now.");
2055  rdbExitReportCorruptRDB("Unexpected EOF reading RDB file");
2056  return C_ERR; /* Just to avoid warning */
2057 }
