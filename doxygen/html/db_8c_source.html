41 /* Update LFU when an object is accessed.
42  * Firstly, decrement the counter if the decrement time is reached.
43  * Then logarithmically increment the counter, and update the access time. */
44 void updateLFU(robj *val) {
45  unsigned long counter = LFUDecrAndReturn(val);
46  counter = LFULogIncr(counter);
47  val->lru = (LFUGetTimeInMinutes()<<8) | counter;
48 }

51  * implementations that should instead rely on lookupKeyRead(),
52  * lookupKeyWrite() and lookupKeyReadWithFlags(). */
53 robj *lookupKey(redisDb *db, robj *key, int flags) {
54  dictEntry *de = dictFind(db->dict,key->ptr);
55  if (de) {
56  robj *val = dictGetVal(de);
57 
58  /* Update the access time for the ageing algorithm.
59  * Don't do it if we have a saving child, as this will trigger
60  * a copy on write madness. */
61  if (server.rdb_child_pid == -1 &&
62  server.aof_child_pid == -1 &&
63  !(flags & LOOKUP_NOTOUCH))
64  {
65  if (server.maxmemory_policy & MAXMEMORY_FLAG_LFU) {
66  updateLFU(val);
67  } else {
68  val->lru = LRU_CLOCK();
69  }
70  }
71  return val;
72  } else {
73  return NULL;
74  }

77 /* Lookup a key for read operations, or return NULL if the key is not found
78  * in the specified DB.
79  *
80  * As a side effect of calling this function:
81  * 1. A key gets expired if it reached it's TTL.
82  * 2. The key last access time is updated.
83  * 3. The global keys hits/misses stats are updated (reported in INFO).
84  *
85  * This API should not be used when we write to the key after obtaining
86  * the object linked to the key, but only for read only operations.
87  *
88  * Flags change the behavior of this command:
89  *
90  * LOOKUP_NONE (or zero): no special flags are passed.
91  * LOOKUP_NOTOUCH: don't alter the last access time of the key.
92  *
93  * Note: this function also returns NULL if the key is logically expired
94  * but still existing, in case this is a slave, since this API is called only
95  * for read operations. Even if the key expiry is master-driven, we can
96  * correctly report a key is expired on slaves even if the master is lagging
97  * expiring our key via DELs in the replication link. */
98 robj *lookupKeyReadWithFlags(redisDb *db, robj *key, int flags) {
99  robj *val;
100 
101  if (expireIfNeeded(db,key) == 1) {
102  /* Key expired. If we are in the context of a master, expireIfNeeded()
103  * returns 0 only when the key does not exist at all, so it's safe
104  * to return NULL ASAP. */
105  if (server.masterhost == NULL) return NULL;
106 
107  /* However if we are in the context of a slave, expireIfNeeded() will
108  * not really try to expire the key, it only returns information
109  * about the "logical" status of the key: key expiring is up to the
110  * master in order to have a consistent view of master's data set.
111  *
112  * However, if the command caller is not the master, and as additional
113  * safety measure, the command invoked is a read-only command, we can
114  * safely return NULL here, and provide a more consistent behavior
115  * to clients accessign expired values in a read-only fashion, that
116  * will say the key as non existing.
117  *
118  * Notably this covers GETs when slaves are used to scale reads. */
119  if (server.current_client &&
120  server.current_client != server.master &&
121  server.current_client->cmd &&
122  server.current_client->cmd->flags & CMD_READONLY)
123  {
124  return NULL;
125  }
126  }
127  val = lookupKey(db,key,flags);
128  if (val == NULL)
129  server.stat_keyspace_misses++;
130  else
131  server.stat_keyspace_hits++;
132  return val;
133 }

225 /* Return a random key, in form of a Redis object.
226  * If there are no keys, NULL is returned.
227  *
228  * The function makes sure to return keys not already expired. */
229 robj *dbRandomKey(redisDb *db) {
230  dictEntry *de;
231  int maxtries = 100;
232  int allvolatile = dictSize(db->dict) == dictSize(db->expires);
233 
234  while(1) {
235  sds key;
236  robj *keyobj;
237 
238  de = dictGetRandomKey(db->dict);
239  if (de == NULL) return NULL;
240 
241  key = dictGetKey(de);
242  keyobj = createStringObject(key,sdslen(key));
243  if (dictFind(db->expires,key)) {
244  if (allvolatile && server.masterhost && --maxtries == 0) {
245  /* If the DB is composed only of keys with an expire set,
246  * it could happen that all the keys are already logically
247  * expired in the slave, so the function cannot stop because
248  * expireIfNeeded() is false, nor it can stop because
249  * dictGetRandomKey() returns NULL (there are keys to return).
250  * To prevent the infinite loop we do some tries, but if there
251  * are the conditions for an infinite loop, eventually we
252  * return a key name that may be already expired. */
253  return keyobj;
254  }
255  if (expireIfNeeded(db,keyobj)) {
256  decrRefCount(keyobj);
257  continue; /* search for another key. This expired. */
258  }
259  }
260  return keyobj;
261  }
262 }

284 /* Prepare the string object stored at 'key' to be modified destructively
285  * to implement commands like SETBIT or APPEND.
286  *
287  * An object is usually ready to be modified unless one of the two conditions
288  * are true:
289  *
290  * 1) The object 'o' is shared (refcount > 1), we don't want to affect
291  * other users.
292  * 2) The object encoding is not "RAW".
293  *
294  * If the object is found in one of the above conditions (or both) by the
295  * function, an unshared / not-encoded copy of the string object is stored
296  * at 'key' in the specified 'db'. Otherwise the object 'o' itself is
297  * returned.
298  *
299  * USAGE:
300  *
301  * The object 'o' is what the caller already obtained by looking up 'key'
302  * in 'db', the usage pattern looks like this:
303  *
304  * o = lookupKeyWrite(db,key);
305  * if (checkType(c,o,OBJ_STRING)) return;
306  * o = dbUnshareStringValue(db,key,o);
307  *
308  * At this point the caller is ready to modify the object, for example
309  * using an sdscat() call to append some data, or anything else.
310  */
311 robj *dbUnshareStringValue(redisDb *db, robj *key, robj *o) {
312  serverAssert(o->type == OBJ_STRING);
313  if (o->refcount != 1 || o->encoding != OBJ_ENCODING_RAW) {
314  robj *decoded = getDecodedObject(o);
315  o = createRawStringObject(decoded->ptr, sdslen(decoded->ptr));
316  decrRefCount(decoded);
317  dbOverwrite(db,key,o);
318  }
319  return o;

322 /* Remove all keys from all the databases in a Redis server.
323  * If callback is given the function is called from time to time to
324  * signal that work is in progress.
325  *
326  * The dbnum can be -1 if all the DBs should be flushed, or the specified
327  * DB number if we want to flush only a single Redis database number.
328  *
329  * Flags are be EMPTYDB_NO_FLAGS if no special flags are specified or
330  * EMPTYDB_ASYNC if we want the memory to be freed in a different thread
331  * and the function to return ASAP.
332  *
333  * On success the fuction returns the number of keys removed from the
334  * database(s). Otherwise -1 is returned in the specific case the
335  * DB number is out of range, and errno is set to EINVAL. */
336 long long emptyDb(int dbnum, int flags, void(callback)(void*)) {
337  int async = (flags & EMPTYDB_ASYNC);
338  long long removed = 0;
339 
340  if (dbnum < -1 || dbnum >= server.dbnum) {
341  errno = EINVAL;
342  return -1;
343  }
344 
345  int startdb, enddb;
346  if (dbnum == -1) {
347  startdb = 0;
348  enddb = server.dbnum-1;
349  } else {
350  startdb = enddb = dbnum;
351  }
352 
353  for (int j = startdb; j <= enddb; j++) {
354  removed += dictSize(server.db[j].dict);
355  if (async) {
356  emptyDbAsync(&server.db[j]);
357  } else {
358  dictEmpty(server.db[j].dict,callback);
359  dictEmpty(server.db[j].expires,callback);
360  }
361  }
362  if (server.cluster_enabled) {
363  if (async) {
364  slotToKeyFlushAsync();
365  } else {
366  slotToKeyFlush();
367  }
368  }
369  if (dbnum == -1) flushSlaveKeysWithExpireList();
370  return removed;
371 }

607 /* This command implements SCAN, HSCAN and SSCAN commands.
608  * If object 'o' is passed, then it must be a Hash or Set object, otherwise
609  * if 'o' is NULL the command will operate on the dictionary associated with
610  * the current database.
611  *
612  * When 'o' is not NULL the function assumes that the first argument in
613  * the client arguments vector is a key so it skips it before iterating
614  * in order to parse options.
615  *
616  * In the case of a Hash object the function returns both the field and value
617  * of every element on the Hash. */
618 void scanGenericCommand(client *c, robj *o, unsigned long cursor) {
619  int i, j;
620  list *keys = listCreate();
621  listNode *node, *nextnode;
622  long count = 10;
623  sds pat = NULL;
624  int patlen = 0, use_pattern = 0;
625  dict *ht;
626 
627  /* Object must be NULL (to iterate keys names), or the type of the object
628  * must be Set, Sorted Set, or Hash. */
629  serverAssert(o == NULL || o->type == OBJ_SET || o->type == OBJ_HASH ||
630  o->type == OBJ_ZSET);
631 
632  /* Set i to the first option argument. The previous one is the cursor. */
633  i = (o == NULL) ? 2 : 3; /* Skip the key argument if needed. */
634 
635  /* Step 1: Parse options. */
636  while (i < c->argc) {
637  j = c->argc - i;
638  if (!strcasecmp(c->argv[i]->ptr, "count") && j >= 2) {
639  if (getLongFromObjectOrReply(c, c->argv[i+1], &count, NULL)
640  != C_OK)
641  {
642  goto cleanup;
643  }
644 
645  if (count < 1) {
646  addReply(c,shared.syntaxerr);
647  goto cleanup;
648  }
649 
650  i += 2;
651  } else if (!strcasecmp(c->argv[i]->ptr, "match") && j >= 2) {
652  pat = c->argv[i+1]->ptr;
653  patlen = sdslen(pat);
654 
655  /* The pattern always matches if it is exactly "*", so it is
656  * equivalent to disabling it. */
657  use_pattern = !(pat[0] == '*' && patlen == 1);
658 
659  i += 2;
660  } else {
661  addReply(c,shared.syntaxerr);
662  goto cleanup;
663  }
664  }
665 
666  /* Step 2: Iterate the collection.
667  *
668  * Note that if the object is encoded with a ziplist, intset, or any other
669  * representation that is not a hash table, we are sure that it is also
670  * composed of a small number of elements. So to avoid taking state we
671  * just return everything inside the object in a single call, setting the
672  * cursor to zero to signal the end of the iteration. */
673 
674  /* Handle the case of a hash table. */
675  ht = NULL;
676  if (o == NULL) {
677  ht = c->db->dict;
678  } else if (o->type == OBJ_SET && o->encoding == OBJ_ENCODING_HT) {
679  ht = o->ptr;
680  } else if (o->type == OBJ_HASH && o->encoding == OBJ_ENCODING_HT) {
681  ht = o->ptr;
682  count *= 2; /* We return key / value for this type. */
683  } else if (o->type == OBJ_ZSET && o->encoding == OBJ_ENCODING_SKIPLIST) {
684  zset *zs = o->ptr;
685  ht = zs->dict;
686  count *= 2; /* We return key / value for this type. */
687  }
688 
689  if (ht) {
690  void *privdata[2];
691  /* We set the max number of iterations to ten times the specified
692  * COUNT, so if the hash table is in a pathological state (very
693  * sparsely populated) we avoid to block too much time at the cost
694  * of returning no or very few elements. */
695  long maxiterations = count*10;
696 
697  /* We pass two pointers to the callback: the list to which it will
698  * add new elements, and the object containing the dictionary so that
699  * it is possible to fetch more data in a type-dependent way. */
700  privdata[0] = keys;
701  privdata[1] = o;
702  do {
703  cursor = dictScan(ht, cursor, scanCallback, NULL, privdata);
704  } while (cursor &&
705  maxiterations-- &&
706  listLength(keys) < (unsigned long)count);
707  } else if (o->type == OBJ_SET) {
708  int pos = 0;
709  int64_t ll;
710 
711  while(intsetGet(o->ptr,pos++,&ll))
712  listAddNodeTail(keys,createStringObjectFromLongLong(ll));
713  cursor = 0;
714  } else if (o->type == OBJ_HASH || o->type == OBJ_ZSET) {
715  unsigned char *p = ziplistIndex(o->ptr,0);
716  unsigned char *vstr;
717  unsigned int vlen;
718  long long vll;
719 
720  while(p) {
721  ziplistGet(p,&vstr,&vlen,&vll);
722  listAddNodeTail(keys,
723  (vstr != NULL) ? createStringObject((char*)vstr,vlen) :
724  createStringObjectFromLongLong(vll));
725  p = ziplistNext(o->ptr,p);
726  }
727  cursor = 0;
728  } else {
729  serverPanic("Not handled encoding in SCAN.");
730  }
731 
732  /* Step 3: Filter elements. */
733  node = listFirst(keys);
734  while (node) {
735  robj *kobj = listNodeValue(node);
736  nextnode = listNextNode(node);
737  int filter = 0;
738 
739  /* Filter element if it does not match the pattern. */
740  if (!filter && use_pattern) {
741  if (sdsEncodedObject(kobj)) {
742  if (!stringmatchlen(pat, patlen, kobj->ptr, sdslen(kobj->ptr), 0))
743  filter = 1;
744  } else {
745  char buf[LONG_STR_SIZE];
746  int len;
747 
748  serverAssert(kobj->encoding == OBJ_ENCODING_INT);
749  len = ll2string(buf,sizeof(buf),(long)kobj->ptr);
750  if (!stringmatchlen(pat, patlen, buf, len, 0)) filter = 1;
751  }
752  }
753 
754  /* Filter element if it is an expired key. */
755  if (!filter && o == NULL && expireIfNeeded(c->db, kobj)) filter = 1;
756 
757  /* Remove the element and its associted value if needed. */
758  if (filter) {
759  decrRefCount(kobj);
760  listDelNode(keys, node);
761  }
762 
763  /* If this is a hash or a sorted set, we have a flat list of
764  * key-value elements, so if this element was filtered, remove the
765  * value, or skip it if it was not filtered: we only match keys. */
766  if (o && (o->type == OBJ_ZSET || o->type == OBJ_HASH)) {
767  node = nextnode;
768  nextnode = listNextNode(node);
769  if (filter) {
770  kobj = listNodeValue(node);
771  decrRefCount(kobj);
772  listDelNode(keys, node);
773  }
774  }
775  node = nextnode;
776  }
777 
778  /* Step 4: Reply to the client. */
779  addReplyMultiBulkLen(c, 2);
780  addReplyBulkLongLong(c,cursor);
781 
782  addReplyMultiBulkLen(c, listLength(keys));
783  while ((node = listFirst(keys)) != NULL) {
784  robj *kobj = listNodeValue(node);
785  addReplyBulk(c, kobj);
786  decrRefCount(kobj);
787  listDelNode(keys, node);
788  }
789 
790 cleanup:
791  listSetFreeMethod(keys,decrRefCountVoid);
792  listRelease(keys);
793 }

986 /* Swap two databases at runtime so that all clients will magically see
987  * the new database even if already connected. Note that the client
988  * structure c->db points to a given DB, so we need to be smarter and
989  * swap the underlying referenced structures, otherwise we would need
990  * to fix all the references to the Redis DB structure.
991  *
992  * Returns C_ERR if at least one of the DB ids are out of range, otherwise
993  * C_OK is returned. */
994 int dbSwapDatabases(int id1, int id2) {
995  if (id1 < 0 || id1 >= server.dbnum ||
996  id2 < 0 || id2 >= server.dbnum) return C_ERR;
997  if (id1 == id2) return C_OK;
998  redisDb aux = server.db[id1];
999  redisDb *db1 = &server.db[id1], *db2 = &server.db[id2];
1000 
1001  /* Swap hash tables. Note that we don't swap blocking_keys,
1002  * ready_keys and watched_keys, since we want clients to
1003  * remain in the same DB they were. */
1004  db1->dict = db2->dict;
1005  db1->expires = db2->expires;
1006  db1->avg_ttl = db2->avg_ttl;
1007 
1008  db2->dict = aux.dict;
1009  db2->expires = aux.expires;
1010  db2->avg_ttl = aux.avg_ttl;
1011

1013  * of swapping the two DBs, a client that was waiting for list
1014  * X in a given DB, may now actually be unblocked if X happens
1015  * to exist in the new version of the DB, after the swap.
1016  *
1017  * However normally we only do this check for efficiency reasons
1018  * in dbAdd() when a list is created. So here we need to rescan
1019  * the list of clients blocked on lists and signal lists as ready
1020  * if needed. */
1021  scanDatabaseForReadyLists(db1);
1022  scanDatabaseForReadyLists(db2);
1023  return C_OK;
1024 }

1123 /* This function is called when we are going to perform some operation
1124  * in a given key, but such key may be already logically expired even if
1125  * it still exists in the database. The main way this function is called
1126  * is via lookupKey*() family of functions.
1127  *
1128  * The behavior of the function depends on the replication role of the
1129  * instance, because slave instances do not expire keys, they wait
1130  * for DELs from the master for consistency matters. However even
1131  * slaves will try to have a coherent return value for the function,
1132  * so that read commands executed in the slave side will be able to
1133  * behave like if the key is expired even if still present (because the
1134  * master has yet to propagate the DEL).
1135  *
1136  * In masters as a side effect of finding a key which is expired, such
1137  * key will be evicted from the database. Also this may trigger the
1138  * propagation of a DEL/UNLINK command in AOF / replication stream.
1139  *
1140  * The return value of the function is 0 if the key is still valid,
1141  * otherwise the function returns 1 if the key is expired. */
1142 int expireIfNeeded(redisDb *db, robj *key) {
1143  mstime_t when = getExpire(db,key);
1144  mstime_t now;
1145 
1146  if (when < 0) return 0; /* No expire for this key */
1147 
1148  /* Don't expire anything while loading. It will be done later. */
1149  if (server.loading) return 0;
1150 
1151  /* If we are in the context of a Lua script, we pretend that time is
1152  * blocked to when the Lua script started. This way a key can expire
1153  * only the first time it is accessed and not in the middle of the
1154  * script execution, making propagation to slaves / AOF consistent.
1155  * See issue #1525 on Github for more information. */
1156  now = server.lua_caller ? server.lua_time_start : mstime();
1157 
1158  /* If we are running in the context of a slave, return ASAP:
1159  * the slave key expiration is controlled by the master that will
1160  * send us synthesized DEL operations for expired keys.
1161  *
1162  * Still we try to return the right information to the caller,
1163  * that is, 0 if we think the key should be still valid, 1 if
1164  * we think the key is expired at this time. */
1165  if (server.masterhost != NULL) return now > when;
1166 
1167  /* Return when this key has not expired */
1168  if (now <= when) return 0;
1169 
1170  /* Delete the key */
1171  server.stat_expiredkeys++;
1172  propagateExpire(db,key,server.lazyfree_lazy_expire);
1173  notifyKeyspaceEvent(NOTIFY_EXPIRED,
1174  "expired",key,db->id);
1175  return server.lazyfree_lazy_expire ? dbAsyncDelete(db,key) :
1176  dbSyncDelete(db,key);
1177 }

1184  * (firstkey, lastkey, step). */

1186  int j, i = 0, last, *keys;
1187  UNUSED(argv);
1188

1191  return NULL;
1192  }
1193

1195  if (last < 0) last = argc+last;

1197  for (j = cmd->firstkey; j <= last; j += cmd->keystep) {
1198  if (j >= argc) {

1200  * of arguments (negative arity parameter) do not have dispatch
1201  * time arity checks, so we need to handle the case where the user
1202  * passed an invalid number of arguments here. In this case we

1204  * an arity or syntax error. */

1208  return NULL;

1210  serverPanic("Redis built-in command declared keys positions not matching the arity requirements.");
1211  }
1212  }

1214  }
