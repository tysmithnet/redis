105 /* Get enum value from name. If there is no match INT_MIN is returned. */
106 int configEnumGetValue(configEnum *ce, char *name) {
107  while(ce->name != NULL) {
108  if (!strcasecmp(ce->name,name)) return ce->val;
109  ce++;
110  }
111  return INT_MIN;
112 }
113 
114 /* Get enum name from value. If no match is found NULL is returned. */
115 const char *configEnumGetName(configEnum *ce, int val) {
116  while(ce->name != NULL) {
117  if (ce->val == val) return ce->name;
118  ce++;
119  }
120  return NULL;
121 }

180  for (i = 0; i < totlines; i++) {
181  sds *argv;

183 
184  linenum = i+1;

186

1651  * config rewrite state, and return it to the caller.
1652  *
1653  * If it is impossible to read the old file, NULL is returned.
1654  * If the old file does not exist at all, an empty state is returned. */

1658  char buf[CONFIG_MAX_LINE+1];
1659  int linenum = -1;
1660

1662

1664  state->rewritten = dictCreate(&optionSetDictType,NULL);
1665  state->numlines = 0;
1666  state->lines = NULL;

1671  while(fgets(buf,CONFIG_MAX_LINE+1,fp) != NULL) {

1724 /* Rewrite the specified configuration option with the new "line".
1725  * It progressively uses lines of the file that were already used for the same
1726  * configuration option in the old version of the file, removing that line from
1727  * the map of options -> line numbers.
1728  *
1729  * If there are lines associated with a given configuration option and
1730  * "force" is non-zero, the line is appended to the configuration file.
1731  * Usually "force" is true when an option has not its default value, so it
1732  * must be rewritten even if not present previously.
1733  *
1734  * The first time a line is appended into a configuration file, a comment
1735  * is added to show that starting from that point the config file was generated
1736  * by CONFIG REWRITE.
1737  *
1738  * "line" is either used, or freed, so the caller does not need to free it
1739  * in any way. */
1740 void rewriteConfigRewriteLine(struct rewriteConfigState *state, const char *option, sds line, int force) {
1741  sds o = sdsnew(option);
1742  list *l = dictFetchValue(state->option_to_line,o);
1743 
1744  rewriteConfigMarkAsProcessed(state,option);
1745 
1746  if (!l && !force) {
1747  /* Option not used previously, and we are not forced to use it. */
1748  sdsfree(line);
1749  sdsfree(o);
1750  return;
1751  }
1752 
1753  if (l) {
1754  listNode *ln = listFirst(l);
1755  int linenum = (long) ln->value;
1756 
1757  /* There are still lines in the old configuration file we can reuse
1758  * for this option. Replace the line with the new one. */
1759  listDelNode(l,ln);
1760  if (listLength(l) == 0) dictDelete(state->option_to_line,o);
1761  sdsfree(state->lines[linenum]);
1762  state->lines[linenum] = line;
1763  } else {
1764  /* Append a new line. */
1765  if (!state->has_tail) {
1766  rewriteConfigAppendLine(state,
1767  sdsnew(REDIS_CONFIG_REWRITE_SIGNATURE));
1768  state->has_tail = 1;
1769  }
1770  rewriteConfigAppendLine(state,line);
1771  }
1772  sdsfree(o);
1773 }

2012 /* At the end of the rewrite process the state contains the remaining

2017  *

2019  * blanking all the lines still associated. */
2020 void rewriteConfigRemoveOrphaned(struct rewriteConfigState *state) {
2021  dictIterator *di = dictGetIterator(state->option_to_line);
2022  dictEntry *de;
2023 
2024  while((de = dictNext(di)) != NULL) {
2025  list *l = dictGetVal(de);
2026  sds option = dictGetKey(de);
2027 
2028  /* Don't blank lines about options the rewrite process
2029  * don't understand. */
2030  if (dictFind(state->rewritten,option) == NULL) {
2031  serverLog(LL_DEBUG,"Not rewritten option: %s", option);
2032  continue;
2033  }
2034
