83 /* Perform the SHA1 of the input string. We use this both for hashing script
84  * bodies in order to obtain the Lua function name, and in the implementation
85  * of redis.sha1().
86  *
87  * 'digest' should point to a 41 bytes buffer: 40 for SHA1 converted into an
88  * hexadecimal number, plus 1 byte for null term. */
89 void sha1hex(char *digest, char *script, size_t len) {
90  SHA1_CTX ctx;
91  unsigned char hash[20];
92  char *cset = "0123456789abcdef";
93  int j;
94 
95  SHA1Init(&ctx);
96  SHA1Update(&ctx,(unsigned char*)script,len);
97  SHA1Final(hash,&ctx);
98 
99  for (j = 0; j < 20; j++) {
100  digest[j*2] = cset[((hash[j]&0xF0)>>4)];
101  digest[j*2+1] = cset[(hash[j]&0xF)];
102  }
103  digest[40] = '\0';
104 }

110 /* Take a Redis reply in the Redis protocol format and convert it into a
111  * Lua type. Thanks to this function, and the introduction of not connected
112  * clients, it is trivial to implement the redis() lua function.
113  *
114  * Basically we take the arguments, execute the Redis command in the context
115  * of a non connected client, then take the generated reply and convert it
116  * into a suitable Lua type. With this trick the scripting feature does not
117  * need the introduction of a full Redis internals API. The script
118  * is like a normal client that bypasses all the slow I/O paths.
119  *
120  * Note: in this function we do not do any sanity check as the reply is
121  * generated by Redis directly. This allows us to go faster.
122  *
123  * Errors are returned as a table with a single 'err' field set to the
124  * error string.
125  */
126 
127 char *redisProtocolToLuaType(lua_State *lua, char* reply) {
128  char *p = reply;
129 
130  switch(*p) {
131  case ':': p = redisProtocolToLuaType_Int(lua,reply); break;
132  case '$': p = redisProtocolToLuaType_Bulk(lua,reply); break;
133  case '+': p = redisProtocolToLuaType_Status(lua,reply); break;
134  case '-': p = redisProtocolToLuaType_Error(lua,reply); break;
135  case '*': p = redisProtocolToLuaType_MultiBulk(lua,reply); break;
136  }
137  return p;
138 }

203 /* This function is used in order to push an error on the Lua stack in the
204  * format used by redis.pcall to return errors, which is a lua table
205  * with a single "err" field set to the error string. Note that this
206  * table is never a valid reply by proper commands, since the returned
207  * tables are otherwise always indexed by integers, never by strings. */
208 void luaPushError(lua_State *lua, char *error) {
209  lua_Debug dbg;
210 
211  /* If debugging is active and in step mode, log errors resulting from
212  * Redis commands. */
213  if (ldb.active && ldb.step) {
214  ldbLog(sdscatprintf(sdsempty(),"<error> %s",error));
215  }
216 
217  lua_newtable(lua);
218  lua_pushstring(lua,"err");
219 
220  /* Attempt to figure out where this function was called, if possible */
221  if(lua_getstack(lua, 1, &dbg) && lua_getinfo(lua, "nSl", &dbg)) {
222  sds msg = sdscatprintf(sdsempty(), "%s: %d: %s",
223  dbg.source, dbg.currentline, error);
224  lua_pushstring(lua, msg);
225  sdsfree(msg);
226  } else {
227  lua_pushstring(lua, error);
228  }
229  lua_settable(lua,-3);
230 }

242 /* Sort the array currently in the stack. We do this to make the output
243  * of commands like KEYS or SMEMBERS something deterministic when called
244  * from Lua (to play well with AOf/replication).
245  *
246  * The array is sorted using table.sort itself, and assuming all the
247  * list elements are strings. */
248 void luaSortArray(lua_State *lua) {
249  /* Initial Stack: array */
250  lua_getglobal(lua,"table");
251  lua_pushstring(lua,"sort");
252  lua_gettable(lua,-2); /* Stack: array, table, table.sort */
253  lua_pushvalue(lua,-3); /* Stack: array, table, table.sort, array */
254  if (lua_pcall(lua,1,0,0)) {
255  /* Stack: array, table, error */
256 
257  /* We are not interested in the error, we assume that the problem is
258  * that there are 'false' elements inside the array, so we try
259  * again with a slower function but able to handle this case, that
260  * is: table.sort(table, __redis__compare_helper) */
261  lua_pop(lua,1); /* Stack: array, table */
262  lua_pushstring(lua,"sort"); /* Stack: array, table, sort */
263  lua_gettable(lua,-2); /* Stack: array, table, table.sort */
264  lua_pushvalue(lua,-3); /* Stack: array, table, table.sort, array */
265  lua_getglobal(lua,"__redis__compare_helper");
266  /* Stack: array, table, table.sort, array, __redis__compare_helper */
267  lua_call(lua,2,0);
268  }
269  /* Stack: array (sorted), table */
270  lua_pop(lua,1); /* Stack: array (sorted) */
271 }

277 void luaReplyToRedisReply(client *c, lua_State *lua) {
278  int t = lua_type(lua,-1);
279 
280  switch(t) {
281  case LUA_TSTRING:
282  addReplyBulkCBuffer(c,(char*)lua_tostring(lua,-1),lua_strlen(lua,-1));
283  break;
284  case LUA_TBOOLEAN:
285  addReply(c,lua_toboolean(lua,-1) ? shared.cone : shared.nullbulk);
286  break;
287  case LUA_TNUMBER:
288  addReplyLongLong(c,(long long)lua_tonumber(lua,-1));
289  break;
290  case LUA_TTABLE:
291  /* We need to check if it is an array, an error, or a status reply.
292  * Error are returned as a single element table with 'err' field.
293  * Status replies are returned as single element table with 'ok'
294  * field. */
295  lua_pushstring(lua,"err");
296  lua_gettable(lua,-2);
297  t = lua_type(lua,-1);
298  if (t == LUA_TSTRING) {
299  sds err = sdsnew(lua_tostring(lua,-1));
300  sdsmapchars(err,"\r\n"," ",2);
301  addReplySds(c,sdscatprintf(sdsempty(),"-%s\r\n",err));
302  sdsfree(err);
303  lua_pop(lua,2);
304  return;
305  }
306 
307  lua_pop(lua,1);
308  lua_pushstring(lua,"ok");
309  lua_gettable(lua,-2);
310  t = lua_type(lua,-1);
311  if (t == LUA_TSTRING) {
312  sds ok = sdsnew(lua_tostring(lua,-1));
313  sdsmapchars(ok,"\r\n"," ",2);
314  addReplySds(c,sdscatprintf(sdsempty(),"+%s\r\n",ok));
315  sdsfree(ok);
316  lua_pop(lua,1);
317  } else {
318  void *replylen = addDeferredMultiBulkLength(c);
319  int j = 1, mbulklen = 0;
320 
321  lua_pop(lua,1); /* Discard the 'ok' field value we popped */
322  while(1) {
323  lua_pushnumber(lua,j++);
324  lua_gettable(lua,-2);
325  t = lua_type(lua,-1);
326  if (t == LUA_TNIL) {
327  lua_pop(lua,1);
328  break;
329  }
330  luaReplyToRedisReply(c, lua);
331  mbulklen++;
332  }
333  setDeferredMultiBulkLength(c,replylen,mbulklen);
334  }
335  break;
336  default:
337  addReply(c,shared.nullbulk);
338  }
339  lua_pop(lua,1);
340 }

346 #define LUA_CMD_OBJCACHE_SIZE 32
347 #define LUA_CMD_OBJCACHE_MAX_LEN 64
348 int luaRedisGenericCommand(lua_State *lua, int raise_error) {
349  int j, argc = lua_gettop(lua);
350  struct redisCommand *cmd;
351  client *c = server.lua_client;
352  sds reply;
353 
354  /* Cached across calls. */
355  static robj **argv = NULL;
356  static int argv_size = 0;
357  static robj *cached_objects[LUA_CMD_OBJCACHE_SIZE];
358  static size_t cached_objects_len[LUA_CMD_OBJCACHE_SIZE];
359  static int inuse = 0; /* Recursive calls detection. */
360 
361  /* Reflect MULTI state */
362  if (server.lua_multi_emitted || (server.lua_caller->flags & CLIENT_MULTI)) {
363  c->flags |= CLIENT_MULTI;
364  } else {
365  c->flags &= ~CLIENT_MULTI;
366  }
367 
368  /* By using Lua debug hooks it is possible to trigger a recursive call
369  * to luaRedisGenericCommand(), which normally should never happen.
370  * To make this function reentrant is futile and makes it slower, but
371  * we should at least detect such a misuse, and abort. */
372  if (inuse) {
373  char *recursion_warning =
374  "luaRedisGenericCommand() recursive call detected. "
375  "Are you doing funny stuff with Lua debug hooks?";
376  serverLog(LL_WARNING,"%s",recursion_warning);
377  luaPushError(lua,recursion_warning);
378  return 1;
379  }
380  inuse++;
381 
382  /* Require at least one argument */
383  if (argc == 0) {
384  luaPushError(lua,
385  "Please specify at least one argument for redis.call()");
386  inuse--;
387  return raise_error ? luaRaiseError(lua) : 1;
388  }
389 
390  /* Build the arguments vector */
391  if (argv_size < argc) {
392  argv = zrealloc(argv,sizeof(robj*)*argc);
393  argv_size = argc;
394  }
395 
396  for (j = 0; j < argc; j++) {
397  char *obj_s;
398  size_t obj_len;
399  char dbuf[64];
400 
401  if (lua_type(lua,j+1) == LUA_TNUMBER) {
402  /* We can't use lua_tolstring() for number -> string conversion
403  * since Lua uses a format specifier that loses precision. */
404  lua_Number num = lua_tonumber(lua,j+1);
405 
406  obj_len = snprintf(dbuf,sizeof(dbuf),"%.17g",(double)num);
407  obj_s = dbuf;
408  } else {
409  obj_s = (char*)lua_tolstring(lua,j+1,&obj_len);
410  if (obj_s == NULL) break; /* Not a string. */
411  }
412 
413  /* Try to use a cached object. */
414  if (j < LUA_CMD_OBJCACHE_SIZE && cached_objects[j] &&
415  cached_objects_len[j] >= obj_len)
416  {
417  sds s = cached_objects[j]->ptr;
418  argv[j] = cached_objects[j];
419  cached_objects[j] = NULL;
420  memcpy(s,obj_s,obj_len+1);
421  sdssetlen(s, obj_len);
422  } else {
423  argv[j] = createStringObject(obj_s, obj_len);
424  }
425  }
426 
427  /* Check if one of the arguments passed by the Lua script
428  * is not a string or an integer (lua_isstring() return true for
429  * integers as well). */
430  if (j != argc) {
431  j--;
432  while (j >= 0) {
433  decrRefCount(argv[j]);
434  j--;
435  }
436  luaPushError(lua,
437  "Lua redis() command arguments must be strings or integers");
438  inuse--;
439  return raise_error ? luaRaiseError(lua) : 1;
440  }
441 
442  /* Setup our fake client for command execution */
443  c->argv = argv;
444  c->argc = argc;
445 
446  /* Log the command if debugging is active. */
447  if (ldb.active && ldb.step) {
448  sds cmdlog = sdsnew("<redis>");
449  for (j = 0; j < c->argc; j++) {
450  if (j == 10) {
451  cmdlog = sdscatprintf(cmdlog," ... (%d more)",
452  c->argc-j-1);
453  break;
454  } else {
455  cmdlog = sdscatlen(cmdlog," ",1);
456  cmdlog = sdscatsds(cmdlog,c->argv[j]->ptr);
457  }
458  }
459  ldbLog(cmdlog);
460  }
461 
462  /* Command lookup */
463  cmd = lookupCommand(argv[0]->ptr);
464  if (!cmd || ((cmd->arity > 0 && cmd->arity != argc) ||
465  (argc < -cmd->arity)))
466  {
467  if (cmd)
468  luaPushError(lua,
469  "Wrong number of args calling Redis command From Lua script");
470  else
471  luaPushError(lua,"Unknown Redis command called from Lua script");
472  goto cleanup;
473  }
474  c->cmd = c->lastcmd = cmd;
475 
476  /* There are commands that are not allowed inside scripts. */
477  if (cmd->flags & CMD_NOSCRIPT) {
478  luaPushError(lua, "This Redis command is not allowed from scripts");
479  goto cleanup;
480  }
481 
482  /* Write commands are forbidden against read-only slaves, or if a
483  * command marked as non-deterministic was already called in the context
484  * of this script. */
485  if (cmd->flags & CMD_WRITE) {
486  int deny_write_type = writeCommandsDeniedByDiskError();
487  if (server.lua_random_dirty && !server.lua_replicate_commands) {
488  luaPushError(lua,
489  "Write commands not allowed after non deterministic commands. Call redis.replicate_commands() at the start of your script in order to switch to single commands replication mode.");
490  goto cleanup;
491  } else if (server.masterhost && server.repl_slave_ro &&
492  !server.loading &&
493  !(server.lua_caller->flags & CLIENT_MASTER))
494  {
495  luaPushError(lua, shared.roslaveerr->ptr);
496  goto cleanup;
497  } else if (deny_write_type != DISK_ERROR_TYPE_NONE) {
498  if (deny_write_type == DISK_ERROR_TYPE_RDB) {
499  luaPushError(lua, shared.bgsaveerr->ptr);
500  } else {
501  sds aof_write_err = sdscatfmt(sdsempty(),
502  "-MISCONF Errors writing to the AOF file: %s\r\n",
503  strerror(server.aof_last_write_errno));
504  luaPushError(lua, aof_write_err);
505  sdsfree(aof_write_err);
506  }
507  goto cleanup;
508  }
509  }
510 
511  /* If we reached the memory limit configured via maxmemory, commands that
512  * could enlarge the memory usage are not allowed, but only if this is the
513  * first write in the context of this script, otherwise we can't stop
514  * in the middle. */
515  if (server.maxmemory && /* Maxmemory is actually enabled. */
516  !server.loading && /* Don't care about mem if loading. */
517  !server.masterhost && /* Slave must execute the script. */
518  server.lua_write_dirty == 0 && /* Script had no side effects so far. */
519  (cmd->flags & CMD_DENYOOM))
520  {
521  if (getMaxmemoryState(NULL,NULL,NULL,NULL) != C_OK) {
522  luaPushError(lua, shared.oomerr->ptr);
523  goto cleanup;
524  }
525  }
526 
527  if (cmd->flags & CMD_RANDOM) server.lua_random_dirty = 1;
528  if (cmd->flags & CMD_WRITE) server.lua_write_dirty = 1;
529 
530  /* If this is a Redis Cluster node, we need to make sure Lua is not
531  * trying to access non-local keys, with the exception of commands
532  * received from our master or when loading the AOF back in memory. */
533  if (server.cluster_enabled && !server.loading &&
534  !(server.lua_caller->flags & CLIENT_MASTER))
535  {
536  /* Duplicate relevant flags in the lua client. */
537  c->flags &= ~(CLIENT_READONLY|CLIENT_ASKING);
538  c->flags |= server.lua_caller->flags & (CLIENT_READONLY|CLIENT_ASKING);
539  if (getNodeByQuery(c,c->cmd,c->argv,c->argc,NULL,NULL) !=
540  server.cluster->myself)
541  {
542  luaPushError(lua,
543  "Lua script attempted to access a non local key in a "
544  "cluster node");
545  goto cleanup;
546  }
547  }
548 
549  /* If we are using single commands replication, we need to wrap what
550  * we propagate into a MULTI/EXEC block, so that it will be atomic like
551  * a Lua script in the context of AOF and slaves. */
552  if (server.lua_replicate_commands &&
553  !server.lua_multi_emitted &&
554  !(server.lua_caller->flags & CLIENT_MULTI) &&
555  server.lua_write_dirty &&
556  server.lua_repl != PROPAGATE_NONE)
557  {
558  execCommandPropagateMulti(server.lua_caller);
559  server.lua_multi_emitted = 1;
560  }
561 
562  /* Run the command */
563  int call_flags = CMD_CALL_SLOWLOG | CMD_CALL_STATS;
564  if (server.lua_replicate_commands) {
565  /* Set flags according to redis.set_repl() settings. */
566  if (server.lua_repl & PROPAGATE_AOF)
567  call_flags |= CMD_CALL_PROPAGATE_AOF;
568  if (server.lua_repl & PROPAGATE_REPL)
569  call_flags |= CMD_CALL_PROPAGATE_REPL;
570  }
571  call(c,call_flags);
572 
573  /* Convert the result of the Redis command into a suitable Lua type.
574  * The first thing we need is to create a single string from the client
575  * output buffers. */
576  if (listLength(c->reply) == 0 && c->bufpos < PROTO_REPLY_CHUNK_BYTES) {
577  /* This is a fast path for the common case of a reply inside the
578  * client static buffer. Don't create an SDS string but just use
579  * the client buffer directly. */
580  c->buf[c->bufpos] = '\0';
581  reply = c->buf;
582  c->bufpos = 0;
583  } else {
584  reply = sdsnewlen(c->buf,c->bufpos);
585  c->bufpos = 0;
586  while(listLength(c->reply)) {
587  clientReplyBlock *o = listNodeValue(listFirst(c->reply));
588 
589  reply = sdscatlen(reply,o->buf,o->used);
590  listDelNode(c->reply,listFirst(c->reply));
591  }
592  }
593  if (raise_error && reply[0] != '-') raise_error = 0;
594  redisProtocolToLuaType(lua,reply);

1116  * by specific libc random() implementations and will output the same sequence
1117  * (for the same seed) in every arch. */
1118 
1119 /* The following implementation is the one shipped with Lua itself but with
1120  * rand() replaced by redisLrand48(). */
1121 int redis_math_random (lua_State *L) {
1122  /* the `%' avoids the (rare) case of r==1, and is needed also because on
1123  some systems (SunOS!) `rand()' may return a value larger than RAND_MAX */
1124  lua_Number r = (lua_Number)(redisLrand48()%REDIS_LRAND48_MAX) /
1125  (lua_Number)REDIS_LRAND48_MAX;
1126  switch (lua_gettop(L)) { /* check number of arguments */
1127  case 0: { /* no arguments */
1128  lua_pushnumber(L, r); /* Number between 0 and 1 */
1129  break;
1130  }
1131  case 1: { /* only upper limit */
1132  int u = luaL_checkint(L, 1);
1133  luaL_argcheck(L, 1<=u, 1, "interval is empty");
1134  lua_pushnumber(L, floor(r*u)+1); /* int between 1 and `u' */
1135  break;
1136  }
1137  case 2: { /* lower and upper limits */
1138  int l = luaL_checkint(L, 1);
1139  int u = luaL_checkint(L, 2);
1140  luaL_argcheck(L, l<=u, 2, "interval is empty");
1141  lua_pushnumber(L, floor(r*(u-l+1))+l); /* int between `l' and `u' */
1142  break;
1143  }
1144  default: return luaL_error(L, "wrong number of arguments");
1145  }
1146  return 1;
1147 }

1171  * exists, and in such a case, it behaves like in the success case.

1184  if ((de = dictFind(server.lua_scripts,sha)) != NULL) {
1185  sdsfree(sha);
1186  return dictGetKey(de);
1187  }
1188

1190  funcdef = sdscat(funcdef,"function ");
1191  funcdef = sdscatlen(funcdef,funcname,42);
1192  funcdef = sdscatlen(funcdef,"() ",3);
1193  funcdef = sdscatlen(funcdef,body->ptr,sdslen(body->ptr));

1255 void evalGenericCommand(client *c, int evalsha) {
1256  lua_State *lua = server.lua;
1257  char funcname[43];
1258  long long numkeys;
1259  long long initial_server_dirty = server.dirty;
1260  int delhook = 0, err;
1261 
1262  /* When we replicate whole scripts, we want the same PRNG sequence at
1263  * every call so that our PRNG is not affected by external state. */
1264  redisSrand48(0);
1265 
1266  /* We set this flag to zero to remember that so far no random command
1267  * was called. This way we can allow the user to call commands like
1268  * SRANDMEMBER or RANDOMKEY from Lua scripts as far as no write command
1269  * is called (otherwise the replication and AOF would end with non
1270  * deterministic sequences).
1271  *
1272  * Thanks to this flag we'll raise an error every time a write command
1273  * is called after a random command was used. */
1274  server.lua_random_dirty = 0;
1275  server.lua_write_dirty = 0;
1276  server.lua_replicate_commands = server.lua_always_replicate_commands;
1277  server.lua_multi_emitted = 0;
1278  server.lua_repl = PROPAGATE_AOF|PROPAGATE_REPL;
1279 
1280  /* Get the number of arguments that are keys */
1281  if (getLongLongFromObjectOrReply(c,c->argv[2],&numkeys,NULL) != C_OK)
1282  return;
1283  if (numkeys > (c->argc - 3)) {
1284  addReplyError(c,"Number of keys can't be greater than number of args");
1285  return;
1286  } else if (numkeys < 0) {
1287  addReplyError(c,"Number of keys can't be negative");
1288  return;
1289  }
1290 
1291  /* We obtain the script SHA1, then check if this function is already
1292  * defined into the Lua state */
1293  funcname[0] = 'f';
1294  funcname[1] = '_';
1295  if (!evalsha) {
1296  /* Hash the code if this is an EVAL call */
1297  sha1hex(funcname+2,c->argv[1]->ptr,sdslen(c->argv[1]->ptr));
1298  } else {
1299  /* We already have the SHA if it is a EVALSHA */
1300  int j;
1301  char *sha = c->argv[1]->ptr;
1302 
1303  /* Convert to lowercase. We don't use tolower since the function
1304  * managed to always show up in the profiler output consuming
1305  * a non trivial amount of time. */
1306  for (j = 0; j < 40; j++)
1307  funcname[j+2] = (sha[j] >= 'A' && sha[j] <= 'Z') ?
1308  sha[j]+('a'-'A') : sha[j];
1309  funcname[42] = '\0';
1310  }
1311 
1312  /* Push the pcall error handler function on the stack. */
1313  lua_getglobal(lua, "__redis__err__handler");
1314 
1315  /* Try to lookup the Lua function */
1316  lua_getglobal(lua, funcname);
1317  if (lua_isnil(lua,-1)) {
1318  lua_pop(lua,1); /* remove the nil from the stack */
1319  /* Function not defined... let's define it if we have the
1320  * body of the function. If this is an EVALSHA call we can just
1321  * return an error. */
1322  if (evalsha) {
1323  lua_pop(lua,1); /* remove the error handler from the stack. */
1324  addReply(c, shared.noscripterr);
1325  return;
1326  }
1327  if (luaCreateFunction(c,lua,c->argv[1]) == NULL) {
1328  lua_pop(lua,1); /* remove the error handler from the stack. */
1329  /* The error is sent to the client by luaCreateFunction()
1330  * itself when it returns NULL. */
1331  return;
1332  }
1333  /* Now the following is guaranteed to return non nil */
1334  lua_getglobal(lua, funcname);
1335  serverAssert(!lua_isnil(lua,-1));
1336  }
1337 
1338  /* Populate the argv and keys table accordingly to the arguments that
1339  * EVAL received. */
1340  luaSetGlobalArray(lua,"KEYS",c->argv+3,numkeys);
1341  luaSetGlobalArray(lua,"ARGV",c->argv+3+numkeys,c->argc-3-numkeys);
1342 
1343  /* Select the right DB in the context of the Lua client */
1344  selectDb(server.lua_client,c->db->id);
1345 
1346  /* Set a hook in order to be able to stop the script execution if it
1347  * is running for too much time.
1348  * We set the hook only if the time limit is enabled as the hook will
1349  * make the Lua script execution slower.
1350  *
1351  * If we are debugging, we set instead a "line" hook so that the
1352  * debugger is call-back at every line executed by the script. */
1353  server.lua_caller = c;
1354  server.lua_time_start = mstime();
1355  server.lua_kill = 0;
1356  if (server.lua_time_limit > 0 && ldb.active == 0) {
1357  lua_sethook(lua,luaMaskCountHook,LUA_MASKCOUNT,100000);
1358  delhook = 1;
1359  } else if (ldb.active) {
1360  lua_sethook(server.lua,luaLdbLineHook,LUA_MASKLINE|LUA_MASKCOUNT,100000);
1361  delhook = 1;
1362  }
1363 
1364  /* At this point whether this script was never seen before or if it was
1365  * already defined, we can call it. We have zero arguments and expect
1366  * a single return value. */
1367  err = lua_pcall(lua,0,1,-2);
1368 
1369  /* Perform some cleanup that we need to do both on error and success. */
1370  if (delhook) lua_sethook(lua,NULL,0,0); /* Disable hook */
1371  if (server.lua_timedout) {
1372  server.lua_timedout = 0;
1373  /* Restore the client that was protected when the script timeout
1374  * was detected. */
1375  unprotectClient(c);
1376  if (server.masterhost && server.master)
1377  queueClientForReprocessing(server.master);
1378  }
1379  server.lua_caller = NULL;
1380 
1381  /* Call the Lua garbage collector from time to time to avoid a
1382  * full cycle performed by Lua, which adds too latency.
1383  *
1384  * The call is performed every LUA_GC_CYCLE_PERIOD executed commands
1385  * (and for LUA_GC_CYCLE_PERIOD collection steps) because calling it
1386  * for every command uses too much CPU. */
1387  #define LUA_GC_CYCLE_PERIOD 50
1388  {
1389  static long gc_count = 0;
1390 
1391  gc_count++;
1392  if (gc_count == LUA_GC_CYCLE_PERIOD) {
1393  lua_gc(lua,LUA_GCSTEP,LUA_GC_CYCLE_PERIOD);
1394  gc_count = 0;
1395  }
1396  }
1397 
1398  if (err) {
1399  addReplyErrorFormat(c,"Error running script (call to %s): %s\n",
1400  funcname, lua_tostring(lua,-1));
1401  lua_pop(lua,2); /* Consume the Lua reply and remove error handler. */
1402  } else {
1403  /* On success convert the Lua return value into Redis protocol, and
1404  * send it to * the client. */
1405  luaReplyToRedisReply(c,lua); /* Convert and consume the reply. */
1406  lua_pop(lua,1); /* Remove the error handler. */
1407  }
1408 
1409  /* If we are using single commands replication, emit EXEC if there
1410  * was at least a write. */
1411  if (server.lua_replicate_commands) {
1412  preventCommandPropagation(c);
1413  if (server.lua_multi_emitted) {
1414  robj *propargv[1];
1415  propargv[0] = createStringObject("EXEC",4);
1416  alsoPropagate(server.execCommand,c->db->id,propargv,1,
1417  PROPAGATE_AOF|PROPAGATE_REPL);
1418  decrRefCount(propargv[0]);
1419  }
1420  }
1421 
1422  /* EVALSHA should be propagated to Slave and AOF file as full EVAL, unless
1423  * we are sure that the script was already in the context of all the
1424  * attached slaves *and* the current AOF file if enabled.
1425  *
1426  * To do so we use a cache of SHA1s of scripts that we already propagated
1427  * as full EVAL, that's called the Replication Script Cache.
1428  *
1429  * For repliation, everytime a new slave attaches to the master, we need to
1430  * flush our cache of scripts that can be replicated as EVALSHA, while
1431  * for AOF we need to do so every time we rewrite the AOF file. */
1432  if (evalsha && !server.lua_replicate_commands) {
1433  if (!replicationScriptCacheExists(c->argv[1]->ptr)) {
1434  /* This script is not in our script cache, replicate it as
1435  * EVAL, then add it into the script cache, as from now on
1436  * slaves and AOF know about it. */
1437  robj *script = dictFetchValue(server.lua_scripts,c->argv[1]->ptr);
1438 
1439  replicationScriptCacheAdd(c->argv[1]->ptr);
1440  serverAssertWithInfo(c,NULL,script != NULL);
1441 
1442  /* If the script did not produce any changes in the dataset we want
1443  * just to replicate it as SCRIPT LOAD, otherwise we risk running
1444  * an aborted script on slaves (that may then produce results there)
1445  * or just running a CPU costly read-only script on the slaves. */
1446  if (server.dirty == initial_server_dirty) {
1447  rewriteClientCommandVector(c,3,
1448  resetRefCount(createStringObject("SCRIPT",6)),
1449  resetRefCount(createStringObject("LOAD",4)),
1450  script);
1451  } else {
1452  rewriteClientCommandArgument(c,0,
1453  resetRefCount(createStringObject("EVAL",4)));
1454  rewriteClientCommandArgument(c,1,script);
1455  }
1456  forceCommandPropagation(c,PROPAGATE_REPL|PROPAGATE_AOF);
1457  }
1458  }

1642 /* Start a debugging session before calling EVAL implementation.
1643  * The techique we use is to capture the client socket file descriptor,
1644  * in order to perform direct I/O with it from within Lua hooks. This
1645  * way we don't have to re-enter Redis in order to handle I/O.
1646  *
1647  * The function returns 1 if the caller should proceed to call EVAL,
1648  * and 0 if instead the caller should abort the operation (this happens
1649  * for the parent in a forked session, since it's up to the children
1650  * to continue, or when fork returned an error).
1651  *
1652  * The caller should call ldbEndSession() only if ldbStartSession()
1653  * returned 1. */
1654 int ldbStartSession(client *c) {
1655  ldb.forked = (c->flags & CLIENT_LUA_DEBUG_SYNC) == 0;
1656  if (ldb.forked) {
1657  pid_t cp = fork();
1658  if (cp == -1) {
1659  addReplyError(c,"Fork() failed: can't run EVAL in debugging mode.");
1660  return 0;
1661  } else if (cp == 0) {
1662  /* Child. Let's ignore important signals handled by the parent. */
1663  struct sigaction act;
1664  sigemptyset(&act.sa_mask);
1665  act.sa_flags = 0;
1666  act.sa_handler = SIG_IGN;
1667  sigaction(SIGTERM, &act, NULL);
1668  sigaction(SIGINT, &act, NULL);
1669 
1670  /* Log the creation of the child and close the listening
1671  * socket to make sure if the parent crashes a reset is sent
1672  * to the clients. */
1673  serverLog(LL_WARNING,"Redis forked for debugging eval");
1674  closeListeningSockets(0);
1675  } else {
1676  /* Parent */
1677  listAddNodeTail(ldb.children,(void*)(unsigned long)cp);
1678  freeClientAsync(c); /* Close the client in the parent side. */
1679  return 0;
1680  }
1681  } else {
1682  serverLog(LL_WARNING,
1683  "Redis synchronous debugging eval session started");
1684  }
1685 
1686  /* Setup our debugging session. */
1687  anetBlock(NULL,ldb.fd);
1688  anetSendTimeout(NULL,ldb.fd,5000);
1689  ldb.active = 1;
1690 
1691  /* First argument of EVAL is the script itself. We split it into different
1692  * lines since this is the way the debugger accesses the source code. */
1693  sds srcstring = sdsdup(c->argv[1]->ptr);
1694  size_t srclen = sdslen(srcstring);
1695  while(srclen && (srcstring[srclen-1] == '\n' ||
1696  srcstring[srclen-1] == '\r'))
1697  {
1698  srcstring[--srclen] = '\0';
1699  }
1700  sdssetlen(srcstring,srclen);
1701  ldb.src = sdssplitlen(srcstring,sdslen(srcstring),"\n",1,&ldb.lines);
1702  sdsfree(srcstring);
1703  return 1;
1704 }

1907 /* Append an human readable representation of the Lua value at position 'idx'
1908  * on the stack of the 'lua' state, to the SDS string passed as argument.
1909  * The new SDS string with the represented value attached is returned.
1910  * Used in order to implement ldbLogStackValue().
1911  *
1912  * The element is not automatically removed from the stack, nor it is
1913  * converted to a different type. */
1914 #define LDB_MAX_VALUES_DEPTH (LUA_MINSTACK/2)
1915 sds ldbCatStackValueRec(sds s, lua_State *lua, int idx, int level) {
1916  int t = lua_type(lua,idx);
1917 
1918  if (level++ == LDB_MAX_VALUES_DEPTH)
1919  return sdscat(s,"<max recursion level reached! Nested table?>");
1920 
1921  switch(t) {
1922  case LUA_TSTRING:
1923  {
1924  size_t strl;
1925  char *strp = (char*)lua_tolstring(lua,idx,&strl);
1926  s = sdscatrepr(s,strp,strl);
1927  }
1928  break;
1929  case LUA_TBOOLEAN:
1930  s = sdscat(s,lua_toboolean(lua,idx) ? "true" : "false");
1931  break;
1932  case LUA_TNUMBER:
1933  s = sdscatprintf(s,"%g",(double)lua_tonumber(lua,idx));
1934  break;
1935  case LUA_TNIL:
1936  s = sdscatlen(s,"nil",3);
1937  break;
1938  case LUA_TTABLE:
1939  {
1940  int expected_index = 1; /* First index we expect in an array. */
1941  int is_array = 1; /* Will be set to null if check fails. */
1942  /* Note: we create two representations at the same time, one
1943  * assuming the table is an array, one assuming it is not. At the
1944  * end we know what is true and select the right one. */
1945  sds repr1 = sdsempty();
1946  sds repr2 = sdsempty();
1947  lua_pushnil(lua); /* The first key to start the iteration is nil. */
1948  while (lua_next(lua,idx-1)) {
1949  /* Test if so far the table looks like an array. */
1950  if (is_array &&
1951  (lua_type(lua,-2) != LUA_TNUMBER ||
1952  lua_tonumber(lua,-2) != expected_index)) is_array = 0;
1953  /* Stack now: table, key, value */
1954  /* Array repr. */
1955  repr1 = ldbCatStackValueRec(repr1,lua,-1,level);
1956  repr1 = sdscatlen(repr1,"; ",2);
1957  /* Full repr. */
1958  repr2 = sdscatlen(repr2,"[",1);
1959  repr2 = ldbCatStackValueRec(repr2,lua,-2,level);
1960  repr2 = sdscatlen(repr2,"]=",2);
1961  repr2 = ldbCatStackValueRec(repr2,lua,-1,level);
1962  repr2 = sdscatlen(repr2,"; ",2);
1963  lua_pop(lua,1); /* Stack: table, key. Ready for next iteration. */
1964  expected_index++;
1965  }
1966  /* Strip the last " ;" from both the representations. */
1967  if (sdslen(repr1)) sdsrange(repr1,0,-3);
1968  if (sdslen(repr2)) sdsrange(repr2,0,-3);
1969  /* Select the right one and discard the other. */
1970  s = sdscatlen(s,"{",1);
1971  s = sdscatsds(s,is_array ? repr1 : repr2);
1972  s = sdscatlen(s,"}",1);
1973  sdsfree(repr1);
1974  sdsfree(repr2);
1975  }
1976  break;
1977  case LUA_TFUNCTION:
1978  case LUA_TUSERDATA:
1979  case LUA_TTHREAD:
1980  case LUA_TLIGHTUSERDATA:
1981  {
1982  const void *p = lua_topointer(lua,idx);
1983  char *typename = "unknown";
1984  if (t == LUA_TFUNCTION) typename = "function";
1985  else if (t == LUA_TUSERDATA) typename = "userdata";
1986  else if (t == LUA_TTHREAD) typename = "thread";
1987  else if (t == LUA_TLIGHTUSERDATA) typename = "light-userdata";
1988  s = sdscatprintf(s,"\"%s@%p\"",typename,p);
1989  }
1990  break;
1991  default:
1992  s = sdscat(s,"\"<unknown-lua-type>\"");
1993  break;
1994  }
1995  return s;
1996 }
